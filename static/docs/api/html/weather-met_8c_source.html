31static char *
32met_reprocess (char *x, int len)
33{
34 char *p = x;
35 char *o;
36 int spacing = 0;
37 static gchar *buf;
38 static gint buflen = 0;
39 gchar *lastspace = NULL;
40 int count = 0;
41
42 if (buflen < len)
43 {
44 if (buf)
45 g_free (buf);
46 buf = g_malloc (len + 1);
47 buflen = len;
48 }
49
50 o = buf;
51 x += len; /* End mark */
52
53 while (*p && p < x) {
54 if (g_ascii_isspace (*p)) {
55 if (!spacing) {
56 spacing = 1;
57 lastspace = o;
58 count++;
59 *o++ = ' ';
60 }
61 p++;
62 continue;
63 }
64 spacing = 0;
65 if (count > 75 && lastspace) {
66 count = o - lastspace - 1;
67 *lastspace = '\n';
68 lastspace = NULL;
69 }
70
71 if (*p == '&') {
72 if (g_ascii_strncasecmp (p, "&amp;", 5) == 0) {
73 *o++ = '&';
74 count++;
75 p += 5;
76 continue;
77 }
78 if (g_ascii_strncasecmp (p, "&lt;", 4) == 0) {
79 *o++ = '<';
80 count++;
81 p += 4;
82 continue;
83 }
84 if (g_ascii_strncasecmp (p, "&gt;", 4) == 0) {
85 *o++ = '>';
86 count++;
87 p += 4;
88 continue;
89 }
90 }
91 if (*p == '<') {
92 if (g_ascii_strncasecmp (p, "<BR>", 4) == 0) {
93 *o++ = '\n';
94 count = 0;
95 }
96 if (g_ascii_strncasecmp (p, "<B>", 3) == 0) {
97 *o++ = '\n';
98 *o++ = '\n';
99 count = 0;
100 }
101 p++;
102 while (*p && *p != '>')
103 p++;
104 if (*p)
105 p++;
106 continue;
107 }
108 *o++ = *p++;
109 count++;
110 }
111 *o = 0;
112 return buf;
113}

121static gchar *
122met_parse (const gchar *meto, gsize len)
123{
124 gchar *p;
125 gchar *rp;
126 gchar *r = g_strdup ("Met Office Forecast\n");
127 gchar *t;
128 const gchar *end = meto + len;
129
130 g_return_val_if_fail (meto != NULL, r);
131
132 p = xstrnstr (meto, len, "Summary: </b>");
133 g_return_val_if_fail (p != NULL, r);
134
135 rp = xstrnstr (p, end - p, "Text issued at:");
136 g_return_val_if_fail (rp != NULL, r);
137
138 p += 13;
139 /* p to rp is the text block we want but in HTML malformat */
140 t = g_strconcat (r, met_reprocess (p, rp - p), NULL);
141 g_free (r);
142
143 return t;
144}

146static void
147met_finish (GObject *source, GAsyncResult *result, gpointer data)
148{
149 WeatherInfo *info = (WeatherInfo *)data;
150 GError *error = NULL;
151 GBytes *bytes;
152 const char *response_body = NULL;
153 gsize len = 0;
154
155 g_return_if_fail (info != NULL);
156
157 bytes = soup_session_send_and_read_finish (SOUP_SESSION(source),
158 result, &error);
159
160 if (error != NULL) {
161 g_warning ("Failed to get Met Office forecast data: %s.\n",
162 error->message);
163 request_done (info, error);
164 g_error_free (error);
165 return;
166 }
167
168 response_body = g_bytes_get_data (bytes, &len);
169 info->forecast = met_parse (response_body, len);
170 g_bytes_unref (bytes);
171 request_done (info, NULL);
172}
