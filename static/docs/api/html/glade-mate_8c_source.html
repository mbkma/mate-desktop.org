100static int
101stock_compare (const void *a, const void *b)
102{
103 const mateuiinfo_map_t *ga = a;
104 const mateuiinfo_map_t *gb = b;
105
106 return strcmp (ga->extension, gb->extension);
107}

109static gboolean
110get_stock_uiinfo (const char *stock_name, MateUIInfo *info)
111{
112 const int len = strlen ("MATEUIINFO_MENU_");
113 mateuiinfo_map_t *v;
114 mateuiinfo_map_t base;
115
116 /* If an error happens, return this */
117 if (!strncmp (stock_name, "MATEUIINFO_MENU_", len)) {
118 base.extension = stock_name + len;
119 v = bsearch (
120 &base,
121 mate_uiinfo_mapping,
122 sizeof(mate_uiinfo_mapping) /
123 sizeof(mateuiinfo_map_t),
124 sizeof (mate_uiinfo_mapping [0]),
125 stock_compare);
126 if (v) {
127 *info = v->data;
128 return TRUE;
129 } else
130 return FALSE;
131 }
132 return FALSE;
133}

135static void
136menushell_build_children (GladeXML *xml, GtkWidget *w,
137 GladeWidgetInfo *info)
138{
139 int i, j;
140 MateUIInfo infos[2] = {
141 { MATE_APP_UI_ITEM },
142 MATEUIINFO_END
143 };
144
145 for (i = 0; i < info->n_children; i++) {
146 GladeChildInfo *cinfo = &info->children[i];
147 GladeWidgetInfo *cwinfo = cinfo->child;
148 GtkWidget *child;
149 gchar *stock_name = NULL;
150
151 for (j = 0; j < cwinfo->n_properties; j++) {
152 if (!strcmp (cwinfo->properties[j].name, "stock_item")) {
153 stock_name = cwinfo->properties[j].value;
154 break;
155 }
156 }
157 if (!stock_name) {
158 /* this is a normal menu item */
159 child = glade_xml_build_widget (xml, cwinfo);
160 gtk_menu_shell_append (GTK_MENU_SHELL (w), child);
161 continue;
162 }
163 /* load the template MateUIInfo for this item */
164 if (!get_stock_uiinfo (stock_name, &infos[0])) {
165 /* failure ... */
166 if (!strncmp (stock_name, "MATEUIINFO_", 12))
167 stock_name += 12;
168 child = gtk_menu_item_new_with_label (stock_name);
169 glade_xml_set_common_params (xml, child, cwinfo);
170 gtk_menu_shell_append (GTK_MENU_SHELL(w), child);
171 continue;
172 }
173 /* we now have the template for this item. Now fill it in */
174 for (j = 0; j < cwinfo->n_properties; j++) {
175 const char *name = cwinfo->properties[j].name;
176 const char *value = cwinfo->properties[j].value;
177 if (!strcmp (name, "label"))
178 infos[0].label = _(value);
179 else if (!strcmp (name, "tooltip"))
180 infos[0].hint = _(value);
181 }
182 mate_app_fill_menu (GTK_MENU_SHELL(w), infos,
183 glade_xml_ensure_accel(xml), TRUE,
184 i);
185 child = infos[0].widget;
186 gtk_menu_item_remove_submenu(GTK_MENU_ITEM(child));
187 glade_xml_set_common_params(xml, child, cwinfo);
188 }
189
190#if 0
191 if (uline)
192 glade_xml_pop_uline_accel(xml);
193#endif
194#if 0
195 if (strcmp(info->classname, "GtkMenuBar") != 0 &&
196 mate_preferences_get_menus_have_tearoff()) {
197 GtkWidget *tearoff = gtk_tearoff_menu_item_new();
198
199 gtk_menu_prepend(GTK_MENU(w), tearoff);
200 gtk_widget_show(tearoff);
201 }
202#endif
203}

205static void
206mate_add_dock_item (GladeXML *xml,
207 GtkWidget *parent,
208 GladeWidgetInfo *info,
209 GladeChildInfo *childinfo)
210{
211 MateComponentDockPlacement placement;
212 guint band, offset;
213 int position;
214 MateComponentDockItemBehavior behavior;
215 int i;
216 GtkWidget *child;
217 GtkWidget *toplevel;
218
219 band = offset = position = 0;
220 placement = MATECOMPONENT_DOCK_TOP;
221 behavior = MATECOMPONENT_DOCK_ITEM_BEH_NORMAL;
222
223 for (i = 0; i < childinfo->n_properties; i++) {
224 const char *name = childinfo->properties[i].name;
225 const char *value = childinfo->properties[i].value;
226
227 if (!strcmp (name, "placement"))
228 placement = glade_enum_from_string (
229 MATECOMPONENT_TYPE_DOCK_PLACEMENT,
230 value);
231 else if (!strcmp (name, "band"))
232 band = strtoul (value, NULL, 10);
233 else if (!strcmp (name, "position"))
234 position = strtol (value, NULL, 10);
235 else if (!strcmp (name, "offset"))
236 offset = strtoul (value, NULL, 10);
237 else if (!strcmp (name, "behavior"))
238 behavior = glade_flags_from_string (
239 MATECOMPONENT_TYPE_DOCK_ITEM_BEHAVIOR,
240 value);
241 }
242
243 child = glade_xml_build_widget (xml, childinfo->child);
244
245 toplevel = gtk_widget_get_ancestor (parent, MATE_TYPE_APP);
246
247 matecomponent_dock_item_set_behavior (MATECOMPONENT_DOCK_ITEM (child), behavior);
248
249 if (toplevel != NULL) {
250 mate_app_add_dock_item (
251 MATE_APP (toplevel),
252 MATECOMPONENT_DOCK_ITEM (child),
253 placement,
254 band,
255 position,
256 offset);
257 } else {
258 matecomponent_dock_add_item (MATECOMPONENT_DOCK (parent),
259 MATECOMPONENT_DOCK_ITEM (child),
260 placement, band, position, offset,
261 FALSE);
262 }
263}

266static void
267mate_dock_build_children (GladeXML *xml, GtkWidget *w, GladeWidgetInfo *info)
268{
269 int i;
270 GtkWidget *child;
271 GtkWidget *toplevel;
272 GladeChildInfo *childinfo;
273
274 toplevel = gtk_widget_get_ancestor (w, MATE_TYPE_APP);
275
276 for (i = 0; i < info->n_children; i++) {
277 childinfo = &info->children[i];
278
279 if (!strcmp (childinfo->child->classname, "MateComponentDockItem")) {
280 mate_add_dock_item (xml, w, info, childinfo);
281 continue;
282 }
283
284 if (matecomponent_dock_get_client_area (MATECOMPONENT_DOCK (w)))
285 g_warning ("Multiple client areas for MateComponentDock found.");
286
287 child = glade_xml_build_widget (xml, childinfo->child);
288
289 if (toplevel != NULL)
290 mate_app_set_contents (MATE_APP (toplevel), child);
291 else
292 matecomponent_dock_set_client_area (MATECOMPONENT_DOCK (w), child);
293 }

296static void
297dialog_build_children(GladeXML *self, GtkWidget *w,
298 GladeWidgetInfo *info)
299
300{
301 MateDialog *dialog = MATE_DIALOG (w);
302 GtkWidget *aa;
303 GList *children, *list;
304 char *label;
305
306 glade_standard_build_children (self, w, info);
307
308 aa = dialog->action_area;
309 if (!aa)
310 return;
311
312 children = gtk_container_get_children (GTK_CONTAINER (aa));
313 for (list = children; list; list = list->next) {
314 gtk_widget_ref (GTK_WIDGET (list->data));
315 gtk_container_remove (GTK_CONTAINER (aa), GTK_WIDGET (list->data));
316 }
317
318 for (list = children; list; list = list->next) {
319 g_object_get (G_OBJECT (list->data), "label", &label, NULL);
320 if (label) {
321 mate_dialog_append_button (dialog, label);
322 g_free (label);
323 }
324 }
325
326 g_list_foreach (children, (GFunc)gtk_widget_unref, NULL);
327 g_list_free (children);
328}

460static GtkWidget *
461druid_page_edge_new (GladeXML *xml, GType widget_type,
462 GladeWidgetInfo *info)
463{
464 GtkWidget *druid;
465 MateEdgePosition position = MATE_EDGE_OTHER;
466 int i;
467
468 const char *title, *text;
469 char *filename;
470 GdkPixbuf *logo, *watermark, *top_watermark;
471
472 title = text = NULL;
473 logo = watermark = top_watermark = NULL;
474
475 for (i = 0; i < info->n_properties; i++) {
476 const char *name = info->properties[i].name;
477 const char *value = info->properties[i].value;
478
479 if (!strcmp (name, "position"))
480 position = glade_enum_from_string (
481 MATE_TYPE_EDGE_POSITION, value);
482 else if (!strcmp (name, "text"))
483 text = value;
484 else if (!strcmp (name, "title"))
485 title = value;
486 else if (!strcmp (name, "logo")) {
487 if (logo)
488 g_object_unref (G_OBJECT (logo));
489 filename = glade_xml_relative_file (xml, value);
490 logo = gdk_pixbuf_new_from_file (filename, NULL);
491 g_free (filename);
492 } else if (!strcmp (name, "watermark")) {
493 if (watermark)
494 g_object_unref (G_OBJECT (watermark));
495 filename = glade_xml_relative_file (xml, value);
496 watermark = gdk_pixbuf_new_from_file (filename, NULL);
497 g_free (filename);
498 } else if (!strcmp (name, "top_watermark")) {
499 if (top_watermark)
500 g_object_unref (G_OBJECT (top_watermark));
501 filename = glade_xml_relative_file (xml, value);
502 top_watermark = gdk_pixbuf_new_from_file (filename, NULL);
503 g_free (filename);
504 }
505
506 }
507
508 druid = glade_standard_build_widget (xml, widget_type, info);
509
510 mate_druid_page_edge_construct (
511 MATE_DRUID_PAGE_EDGE (druid),
512 position, TRUE, title, text,
513 logo, watermark, top_watermark);
514
515 if (logo)
516 g_object_unref (G_OBJECT (logo));
517 if (watermark)
518 g_object_unref (G_OBJECT (watermark));
519 if (top_watermark)
520 g_object_unref (G_OBJECT (top_watermark));
521
522 return druid;
523}
