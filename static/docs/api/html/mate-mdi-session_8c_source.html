45static GPtrArray *
46config_get_list (const gchar *key)
47{
48 GPtrArray *array;
49 gchar *string, *pos;
50
51 string = mate_config_get_string (key);
52 if (!string) return NULL;
53
54 array = g_ptr_array_new ();
55
56 pos = string;
57 while (*pos) {
58 glong value;
59 gchar *tmp;
60
61 tmp = strchr (pos, ':');
62 if (tmp) *tmp = '\0';
63
64 if (sscanf (pos, "%lx", &value) == 1)
65 g_ptr_array_add (array, (gpointer) value);
66
67 if (tmp)
68 pos = tmp+1;
69 else
70 break;
71 }
72
73 g_free (string);
74 return array;
75}

77static void
78config_set_list (const gchar *key, GList *list, gpointer (*func)(gpointer))
79{
80 gchar value [BUFSIZ];
81
82 value [0] = '\0';
83 while (list) {
84 char buffer [BUFSIZ];
85 gpointer data;
86
87 data = func ? func (list->data) : list->data;
88 g_snprintf (buffer, sizeof(buffer), "%lx", (glong) data);
89 if (*value) strcat (value, ":");
90 strcat (value, buffer);
91
92 list = g_list_next (list);
93 }
94
95 mate_config_set_string (key, value);
96}

100 GHashTable *child_windows, GHashTable *child_views,
101 GHashTable *view_hash, GHashTable *window_hash,
102 glong window, glong child, gboolean *init,

104{
105 GPtrArray *windows, *views;
106 MateMDIChild *mdi_child;
107 guint k;
108
109 windows = g_hash_table_lookup (child_windows, (gpointer) child);
110 if (!windows) return;
111
112 views = g_hash_table_lookup (child_views, (gpointer) child);

114
115 mdi_child = g_hash_table_lookup (child_hash, (gpointer) child);

130 *init = TRUE;
131
132 g_hash_table_insert (window_hash,
133 (gpointer) window,
134 mdi->active_window);
135 }
136
137 g_hash_table_insert (view_hash,
138 views->pdata [k],
139 mdi->active_view);
140 }

230gboolean
231mate_mdi_restore_state (MateMDI *mdi, const gchar *section,
232 MateMDIChildCreator create_child_func)
233{
234 gchar key [BUFSIZ], *string;
235 GPtrArray *window_list, *child_list;
236 GHashTable *child_hash, *child_windows;
237 GHashTable *child_views, *view_hash, *window_hash;
238 glong active_view = 0, active_window = 0;
239 guint i;
240 gint mode;
241
242 g_snprintf (key, sizeof(key), "%s/mdi_mode=-1", section);
243 mode = mate_config_get_int (key);
244 if (mate_config_get_int (key) == -1)
245 return FALSE;
246
247 mate_mdi_set_mode (mdi, mode);
248
249 child_hash = g_hash_table_new (NULL, NULL);
250 child_windows = g_hash_table_new (NULL, NULL);
251 child_views = g_hash_table_new (NULL, NULL);
252 view_hash = g_hash_table_new (NULL, NULL);
253 window_hash = g_hash_table_new (NULL, NULL);
254
255 /* Walk the list of windows. */
256
257 g_snprintf (key, sizeof(key), "%s/mdi_windows", section);
258 window_list = config_get_list (key);
259
260 /* Walk the list of children. */
261
262 g_snprintf (key, sizeof(key), "%s/mdi_children", section);
263 child_list = config_get_list (key);
264
265 /* Get the active view. */
266
267 g_snprintf (key, sizeof(key), "%s/mdi_active_view", section);
268 string = mate_config_get_string (key);
269
270 if (string) {
271 sscanf (string, "%lx", &active_view);
272 g_free (string);
273 }
274
275 /* Get the active window. */
276
277 g_snprintf (key, sizeof(key), "%s/mdi_active_window", section);
278 string = mate_config_get_string (key);
279
280 if (string) {
281 sscanf (string, "%lx", &active_window);
282 g_free (string);
283 }
284
285 /* Read child descriptions. */
286
287 for (i = 0; i < child_list->len; i++) {
288 glong child = (glong) child_list->pdata [i];
289 GPtrArray *windows, *views;
290 MateMDIChild *mdi_child;
291
292 g_snprintf (key, sizeof(key), "%s/mdi_child_config_%lx",
293 section, child);
294 string = mate_config_get_string (key);
295 if (!string) continue;
296
297 mdi_child = create_child_func (string);
298 g_free (string);
299
300 mate_mdi_add_child (mdi, mdi_child);
301
302 g_hash_table_insert (child_hash, (gpointer)child, mdi_child);
303
304 g_snprintf (key, sizeof(key), "%s/mdi_child_windows_%lx",
305 section, child);
306 windows = config_get_list (key);
307
308 g_hash_table_insert (child_windows, (gpointer) child, windows);
309
310 g_snprintf (key, sizeof(key), "%s/mdi_child_views_%lx",
311 section, child);
312 views = config_get_list (key);
313
314 g_hash_table_insert (child_views, (gpointer) child, views);
315 }
316
317 /* Read window descriptions. */
318
319 for (i = 0; i < window_list->len; i++)
320 restore_window (mdi, section, child_list,
321 child_hash, child_windows,
322 child_views, view_hash, window_hash,
323 (glong) window_list->pdata [i]);
324
325 /* For each window, set its active view. */
326
327 for (i = 0; i < window_list->len; i++) {
328 GtkWidget *real_view;
329 glong view;
330 gint ret;
331
332 g_snprintf (key, sizeof(key), "%s/mdi_window_view_%lx",
333 section, (long) window_list->pdata [i]);
334 string = mate_config_get_string (key);
335 if (!string) continue;
336
337 ret = sscanf (string, "%lx", &view);
338 g_free (string);
339 if (ret != 1) continue;
340
341 real_view = g_hash_table_lookup (view_hash, (gpointer) view);
342 if (!real_view) continue;
343
344 mate_mdi_set_active_view (mdi, real_view);
345 }
346
347 /* Finally, set the active window. */
348
349 set_active_window (mdi, window_hash, active_window);
350
351 /* Free allocated memory. */
352
353 for (i = 0; i < child_list->len; i++) {
354 glong child = (glong) child_list->pdata [i];
355 GPtrArray *windows, *views;
356
357 windows = g_hash_table_lookup (child_windows, (gpointer) child);
358 if (windows) g_ptr_array_free (windows, FALSE);
359
360 views = g_hash_table_lookup (child_views, (gpointer) child);
361 if (views) g_ptr_array_free (views, FALSE);
362 }
363
364 g_hash_table_destroy (child_hash);
365 g_hash_table_destroy (child_windows);
366 g_hash_table_destroy (child_views);
367 g_hash_table_destroy (view_hash);
368 g_hash_table_destroy (window_hash);
369
370 return TRUE;
371}

390void
391mate_mdi_save_state (MateMDI *mdi, const gchar *section)
392{
393 gchar key [BUFSIZ], value [BUFSIZ];
394 GList *child, *window;
395 gint x, y, w, h;
396
397 mate_config_clean_section (section);
398
399 /* save MDI mode */
400 g_snprintf (key, sizeof(key), "%s/mdi_mode", section);
401 mate_config_set_int (key, mdi->mode);
402
403 /* Write list of children. */
404
405 g_snprintf (key, sizeof(key), "%s/mdi_children", section);
406 config_set_list (key, mdi->children, NULL);
407
408 /* Write list of windows. */
409
410 g_snprintf (key, sizeof(key), "%s/mdi_windows", section);
411 config_set_list (key, mdi->windows, NULL);
412
413 /* Save active window. */
414
415 g_snprintf (key, sizeof(key), "%s/mdi_active_window", section);
416 g_snprintf (value, sizeof(value), "%lx", (glong) mdi->active_window);
417 mate_config_set_string (key, value);
418
419 /* Save active view. */
420
421 g_snprintf (key, sizeof(key), "%s/mdi_active_view", section);
422 g_snprintf (value, sizeof(value), "%lx", (glong) mdi->active_view);
423 mate_config_set_string (key, value);
424
425 /* Walk list of children. */
426
427 child = mdi->children;
428 while (child) {
429 MateMDIChild *mdi_child;
430 gchar *string;
431
432 mdi_child = MATE_MDI_CHILD (child->data);
433
434 /* Save child configuration. */
435
436 string = mate_mdi_child_get_config_string(mdi_child);
437
438 if (string) {
439 g_snprintf (key, sizeof(key), "%s/mdi_child_config_%lx",
440 section, (long) mdi_child);
441 mate_config_set_string (key, string);
442 g_free (string);
443 }
444
445 /* Save list of views this child has. */
446
447 g_snprintf (key, sizeof(key), "%s/mdi_child_windows_%lx",
448 section, (long) mdi_child);
449 config_set_list (key, mdi_child->views, view_window_func);
450
451 g_snprintf (key, sizeof(key), "%s/mdi_child_views_%lx",
452 section, (long) mdi_child);
453 config_set_list (key, mdi_child->views, NULL);
454
455 child = g_list_next (child);
456 }
457
458 /* Save list of toplevel windows. */
459
460 window = mdi->windows;
461 while (window) {
462 MateApp *app;
463 MateComponentDockLayout *layout;
464 GtkWidget *view;
465 gchar *string;
466
467 app = MATE_APP (window->data);
468
469 gdk_window_get_geometry (GTK_WIDGET (app)->window,
470 &x, &y, &w, &h, NULL);
471
472 gdk_window_get_origin (GTK_WIDGET (app)->window, &x, &y);
473
474 g_snprintf (key, sizeof(key), "%s/mdi_window_%lx",
475 section, (long) app);
476 g_snprintf (value, sizeof(value), "%d/%d/%d/%d", x, y, w, h);
477
478 mate_config_set_string (key, value);
479
480 view = mate_mdi_get_view_from_window (mdi, app);
481
482 g_snprintf (key, sizeof(key), "%s/mdi_window_view_%lx",
483 section, (long) app);
484 g_snprintf (value, sizeof(value), "%lx", (long) view);
485
486 mate_config_set_string (key, value);
487
488 g_snprintf(key, sizeof(key), "%s/mdi_window_layout_%lx",
489 section, (long) app);
490
491 layout = matecomponent_dock_get_layout (MATECOMPONENT_DOCK (app->dock));
492 string = matecomponent_dock_layout_create_string (layout);
493 g_object_unref (G_OBJECT (layout));
494 mate_config_set_string(key, string);
495 g_free(string);
496
497 window = g_list_next (window);
498 }
499
500 mate_config_sync ();
501}
