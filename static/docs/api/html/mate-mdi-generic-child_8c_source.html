305static GtkWidget *mate_mdi_generic_child_create_view (MateMDIGenericChild *child)
306{
307 if(!child->create_view && !child->create_view_cbm) {
308 g_error("MateMDIGenericChild: No method for creating views was provided!");
309 return NULL;
310 }
311
312 if(child->create_view_cbm) {
313 GtkArg args[2];
314 GtkWidget *ret = NULL;
315
316 args[0].name = NULL;
317 args[0].type = MATE_TYPE_MDI_CHILD;
318 GTK_VALUE_OBJECT(args[0]) = GTK_OBJECT(child);
319 args[1].name = NULL;
320 args[1].type = GTK_TYPE_WIDGET;
321 GTK_VALUE_POINTER(args[1]) = &ret;
322 child->create_view_cbm(NULL, child->create_view_data, 1, args);
323 return ret;
324 }
325 else
326 return child->create_view(MATE_MDI_CHILD(child),
327 child->create_view_data);
328}

330static GList *mate_mdi_generic_child_create_menus(MateMDIGenericChild *child, GtkWidget *view)
331{
332 if(!child->create_menus && !child->create_menus_cbm)
333 return NULL;
334
335 if (child->create_menus_cbm) {
336 GtkArg args[3];
337 GList *ret = NULL;
338
339 args[0].name = NULL;
340 args[0].type = MATE_TYPE_MDI_CHILD;
341 GTK_VALUE_OBJECT(args[0]) = GTK_OBJECT(child);
342 args[1].name = NULL;
343 args[1].type = GTK_TYPE_WIDGET;
344 GTK_VALUE_OBJECT(args[0]) = GTK_OBJECT(view);
345 args[2].name = NULL;
346 args[2].type = GTK_TYPE_POINTER; /* should we have a boxed type? */
347 GTK_VALUE_POINTER(args[2]) = &ret;
348 child->create_menus_cbm(NULL, child->create_menus_data, 2, args);
349 return ret;
350 }
351 else
352 return child->create_menus(MATE_MDI_CHILD(child), view,
353 child->create_menus_data);
354}

406static void mate_mdi_generic_child_destroy (GtkObject *obj)
407{
408 MateMDIGenericChild *child = MATE_MDI_GENERIC_CHILD(obj);
409
410 if(child->create_view_dn)
411 child->create_view_dn(child->create_view_data);
412 child->create_view_dn = NULL;
413 if(child->create_menus_dn)
414 child->create_menus_dn(child->create_menus_data);
415 child->create_menus_dn = NULL;
416 if(child->get_config_string_dn)
417 child->get_config_string_dn(child->get_config_string_data);
418 child->get_config_string_dn = NULL;
419 if(child->set_label_dn)
420 child->set_label_dn(child->set_label_data);
421 child->set_label_dn = NULL;
422
423 if(GTK_OBJECT_CLASS(parent_class)->destroy)
424 (* GTK_OBJECT_CLASS(parent_class)->destroy)(obj);
425}
