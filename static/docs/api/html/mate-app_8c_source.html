161static void
162mate_app_set_property (GObject *object,
163 guint param_id,
164 const GValue *value,
165 GParamSpec *pspec)
166{
167 MateApp *app = MATE_APP (object);
168
169 switch (param_id) {

194static void
195mate_app_instance_init (MateApp *app)
196{
197 const char *icons = NULL;
198 MateProgram * program;
199
200 app->_priv = NULL;
201 /* XXX: when there is some private stuff enable this
202 app->_priv = g_new0(MateAppPrivate, 1);
203 */
204 app->accel_group = gtk_accel_group_new ();
205 gtk_window_add_accel_group (GTK_WINDOW (app), app->accel_group);
206
207 app->vbox = gtk_vbox_new (FALSE, 0);
208 gtk_container_add (GTK_CONTAINER (app), app->vbox);
209
210 app->dock = matecomponent_dock_new ();
211 app->layout = matecomponent_dock_layout_new ();
212
213 gtk_box_pack_start (GTK_BOX (app->vbox), app->dock,
214 TRUE, TRUE, 0);
215
216 g_signal_connect (app->dock, "layout_changed", G_CALLBACK (layout_changed), app);
217
218 app->enable_layout_config = TRUE;
219
220 program = mate_program_get ();
221 if (program == NULL)
222 g_error ("You must call mate_program_init() before creating a MateApp");
223 g_object_get (G_OBJECT (program),
224 LIBMATEUI_PARAM_DEFAULT_ICON, &icons,
225 NULL);
226
227 if (icons && *icons) {
228 char **files = g_strsplit (icons, ";", -1);
229 mate_window_icon_set_from_file_list (GTK_WINDOW (app),
230 (const char **)files);
231 g_strfreev (files);
232 }
233}

235static void
236remove_notification_cb (MateComponentDockItem *item, gpointer data)
237{
238 MateConfClient *client;
239 guint notify_id;
240
241 notify_id = GPOINTER_TO_INT (g_object_get_data (G_OBJECT (item), "mate-app-mateconf-notify-id"));
242
243 if (notify_id == 0)
244 return;
245
246 client = mateconf_client_get_default ();
247 mateconf_client_notify_remove (client, notify_id);
248 g_object_unref (client);
249 g_object_set_data (G_OBJECT (item), "mate-app-mateconf-notify-id", NULL);
250
251}

253static void
254dock_item_changed_notify (MateConfClient* client,
255 guint cnxn_id,
256 MateConfEntry *entry,
257 gpointer user_data)
258{
259 MateComponentDockItem *item;
260 guint notify_id;
261
262 item = MATECOMPONENT_DOCK_ITEM (user_data);
263 notify_id = GPOINTER_TO_INT (g_object_get_data (G_OBJECT (item), "mate-app-mateconf-notify-id"));
264
265 if (notify_id == cnxn_id &&
266 entry->value->type == MATECONF_VALUE_BOOL) {
267 gboolean detachable;
268
269 detachable = mateconf_value_get_bool (entry->value);
270
271 /* Update */
272 GDK_THREADS_ENTER();
273 matecomponent_dock_item_set_locked (item, !detachable);
274 GDK_THREADS_LEAVE();
275 }
276}

278static void
279setup_notification_for_dock_item (MateComponentDockItem *item, const char *key_name)
280{
281 guint notify_id;

289 g_object_set_data (G_OBJECT (item), "mate-app-mateconf-notify-id", GINT_TO_POINTER (notify_id));
290 g_signal_connect (item, "destroy", G_CALLBACK (remove_notification_cb), NULL);

293static void
294mate_app_show (GtkWidget *widget)
295{
296 MateApp *app;
297
298 app = MATE_APP (widget);
299
300 if (app->layout != NULL) {
301 if (app->enable_layout_config) {
302 gchar *s;
303
304 /* Override the layout with the user's saved
305 configuration. */
306 s = read_layout_config (app);
307 matecomponent_dock_layout_parse_string (app->layout, s);
308 g_free (s);
309 }
310
311 matecomponent_dock_add_from_layout (MATECOMPONENT_DOCK (app->dock),
312 app->layout);
313
314 if (app->enable_layout_config)
315 write_layout_config (app, app->layout);
316
317 g_object_unref (G_OBJECT (app->layout));
318 app->layout = NULL;
319 }
320
321 gtk_widget_show (app->vbox);
322 gtk_widget_show (app->dock);
323
324 MATE_CALL_PARENT (GTK_WIDGET_CLASS, show, (widget));
325}

329{

342 g_object_unref (G_OBJECT (layout));
343 }

478void
479mate_app_set_contents (MateApp *app, GtkWidget *contents)
480{
481 GtkWidget *new_contents;
482
483 g_return_if_fail (app != NULL);
484 g_return_if_fail (MATE_IS_APP(app));
485 g_return_if_fail (app->dock != NULL);
486 g_return_if_fail (contents != NULL);
487 g_return_if_fail (GTK_IS_WIDGET (contents));
488
489 matecomponent_dock_set_client_area (MATECOMPONENT_DOCK (app->dock), contents);
490
491 /* Re-fetch it in case it did not change */
492 new_contents = matecomponent_dock_get_client_area (MATECOMPONENT_DOCK (app->dock));
493
494 if (new_contents == contents && new_contents != app->contents) {
495 gtk_widget_show (new_contents);
496 g_signal_connect (G_OBJECT (new_contents), "parent_set",
497 G_CALLBACK (contents_parent_set),
498 app);
499
500 app->contents = new_contents;
501 }

513void
514mate_app_set_menus (MateApp *app, GtkMenuBar *menubar)
515{
516 GtkWidget *dock_item;
517 GtkAccelGroup *ag;
518 MateComponentDockItemBehavior behavior;
519
520 g_return_if_fail(app != NULL);
521 g_return_if_fail(MATE_IS_APP(app));
522 g_return_if_fail(app->menubar == NULL);
523 g_return_if_fail(menubar != NULL);
524 g_return_if_fail(GTK_IS_MENU_BAR(menubar));
525
526 behavior = (MATECOMPONENT_DOCK_ITEM_BEH_EXCLUSIVE
527 | MATECOMPONENT_DOCK_ITEM_BEH_NEVER_VERTICAL);
528
529 if ( ! mate_mateconf_get_bool ("/desktop/mate/interface/menubar_detachable"))
530 behavior |= MATECOMPONENT_DOCK_ITEM_BEH_LOCKED;
531
532 dock_item = matecomponent_dock_item_new (MATE_APP_MENUBAR_NAME,
533 behavior);
534 setup_notification_for_dock_item (MATECOMPONENT_DOCK_ITEM (dock_item), "/desktop/mate/interface/menubar_detachable");
535 gtk_container_add (GTK_CONTAINER (dock_item), GTK_WIDGET (menubar));
536
537 app->menubar = GTK_WIDGET (menubar);
538
539 matecomponent_dock_item_set_shadow_type (MATECOMPONENT_DOCK_ITEM (dock_item),
540 GTK_SHADOW_NONE);
541
542 if (app->layout != NULL)
543 matecomponent_dock_layout_add_item (app->layout,
544 MATECOMPONENT_DOCK_ITEM (dock_item),
545 MATECOMPONENT_DOCK_TOP,
546 0, 0, 0);
547 else
548 matecomponent_dock_add_item(MATECOMPONENT_DOCK(app->dock),
549 MATECOMPONENT_DOCK_ITEM (dock_item),
550 MATECOMPONENT_DOCK_TOP,
551 0, 0, 0, TRUE);
552
553 gtk_widget_show (GTK_WIDGET (menubar));
554 gtk_widget_show (GTK_WIDGET (dock_item));
555
556 ag = g_object_get_data(G_OBJECT(app), "GtkAccelGroup");
557 if (ag && !g_slist_find(gtk_accel_groups_from_object (G_OBJECT (app)), ag))
558 gtk_window_add_accel_group(GTK_WINDOW(app), ag);
559}
