121static gint
122mate_calc_application_handle_local_options(GApplication *application,
123 GVariantDict *options)
124{
125 if (g_variant_dict_contains(options, "version"))
126 {
127 g_print("%s %s\n", "mate-calc", VERSION);
128 return 0;
129 }
130
131 const gchar *solve_expr = NULL;
132 if (g_variant_dict_lookup(options, "solve", "&s", &solve_expr))
133 {
134 /* Solve mode - handled locally without initializing the full GUI */
135 MPEquationOptions mp_options;
136 MPErrorCode error;
137 MPNumber result = mp_new();
138 char *result_str;
139
140 memset(&mp_options, 0, sizeof(mp_options));
141 mp_options.base = 10;
142 mp_options.wordlen = 32;
143 mp_options.angle_units = MP_DEGREES;
144 mp_options.convert = do_convert;
145
146 error = mp_equation_parse(solve_expr, &mp_options, &result, NULL);
147
148 if (error == PARSER_ERR_MP)
149 {
150 g_printerr("Error: %s\n", mp_get_error());
151 mp_clear(&result);
152 return 1;
153 }
154 else if (error != 0)
155 {
156 g_printerr("Error: %s\n", mp_error_code_to_string(error));
157 mp_clear(&result);
158 return 1;
159 }
160 else
161 {
162 MpSerializer *serializer = mp_serializer_new(MP_DISPLAY_FORMAT_AUTOMATIC, 10, 9);
163 result_str = mp_serializer_to_string(serializer, &result);
164 g_print("%s\n", result_str);
165 g_free(result_str);
166 g_object_unref(serializer);
167 mp_clear(&result);
168 return 0;
169 }
170 }
171
172 /* Let the default handler continue */
173 return -1;
174}
