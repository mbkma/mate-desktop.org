40/* Our custom marshaller for menu items. We need it so that it can extract the per-attachment
41 * user_data pointer from the parent menu shell and pass it to the callback. This overrides the
42 * user-specified data from the MateUIInfo structures.
43 */
44
45typedef void (* ActivateFunc) (GtkObject *object, gpointer data, GtkWidget *for_widget);
46
47static void
48popup_marshal_func (GtkObject *object, gpointer data, guint n_args, GtkArg *args)
49{
50 ActivateFunc func;
51 gpointer user_data;
52 GObject *tl;
53 GtkWidget *for_widget;
54
55 tl = G_OBJECT (get_toplevel(GTK_WIDGET (object)));
56 user_data = g_object_get_data (tl, "mate_popup_menu_do_popup_user_data");
57 for_widget = g_object_get_data (tl, "mate_popup_menu_do_popup_for_widget");
58
59 g_object_set_data (G_OBJECT (get_toplevel(GTK_WIDGET (object))),
60 "mate_popup_menu_active_item",
61 object);
62
63 func = (ActivateFunc) data;
64 if (func)
65 (* func) (object, user_data, for_widget);
66}

68/* This is our custom signal connection function for popup menu items -- see below for the
69 * marshaller information. We pass the original callback function as the data pointer for the
70 * marshaller (uiinfo->moreinfo).
71 */
72static void
73popup_connect_func (MateUIInfo *uiinfo, const char *signal_name, MateUIBuilderData *uibdata)
74{
75 g_assert (uibdata->is_interp);
76
77 g_object_set_data (G_OBJECT (uiinfo->widget),
78 TOPLEVEL_MENUSHELL_KEY,
79 global_menushell_hack);
80
81 g_signal_connect (uiinfo->widget, signal_name,
82 G_CALLBACK (popup_marshal_func),
83 uiinfo->moreinfo);
84}

97GtkWidget *
98mate_popup_menu_new_with_accelgroup (MateUIInfo *uiinfo,
99 GtkAccelGroup *accelgroup)
100{
101 GtkWidget *menu;
102 GtkAccelGroup *my_accelgroup;
103
104 /* We use our own callback marshaller so that it can fetch the
105 * popup user data from the popup menu and pass it on to the
106 * user-defined callbacks.
107 */
108
109 menu = gtk_menu_new ();
110
111 if(accelgroup)
112 my_accelgroup = accelgroup;
113 else
114 my_accelgroup = gtk_accel_group_new();
115 gtk_menu_set_accel_group (GTK_MENU (menu), my_accelgroup);
116 if(!accelgroup)
117 g_object_unref (G_OBJECT (my_accelgroup));
118
119 mate_popup_menu_append (menu, uiinfo);
120
121 return menu;
122}

166/* Callback used when a button is pressed in a widget attached to a popup
167 * menu. It decides whether the menu should be popped up and does the
168 * appropriate stuff.
169 */
170static gint
171real_popup_button_pressed (GtkWidget *widget, GdkEventButton *event, gpointer data)
172{
173 GtkWidget *popup;
174 gpointer user_data;
175
176 popup = data;
177
178 /*
179 * Fetch the attachment user data from the widget and install
180 * it in the popup menu -- it will be used by the callback
181 * mashaller to pass the data to the callbacks.
182 *
183 */
184
185 user_data = g_object_get_data (G_OBJECT (widget), "mate_popup_menu_attach_user_data");
186
187 mate_popup_menu_do_popup (popup, NULL, NULL, event, user_data, widget);
188
189 return TRUE;
190}

192static gint
193popup_button_pressed (GtkWidget *widget, GdkEventButton *event, gpointer data)
194{
195 if (event->button != 3)
196 return FALSE;
197
198 g_signal_stop_emission_by_name (widget, "button_press_event");
199
200 return real_popup_button_pressed (widget, event, data);

211static gint
212relay_popup_button_pressed (GtkWidget *widget, GdkEventButton *event, gpointer data)
213{
214 GtkWidget *new_widget = NULL;
215
216 if (event->button != 3)
217 return FALSE;
218
219 g_signal_stop_emission_by_name (widget, "button_press_event");
220
221 if (GTK_IS_CONTAINER(widget))
222 {
223
224 do {
225 GList *children;
226 GList *child;
227
228 children = gtk_container_get_children (GTK_CONTAINER (widget));
229
230 for (child = children, new_widget = NULL; child; child = child->next)
231 {
232 GtkWidget *cur;
233
234 cur = (GtkWidget *)child->data;
235
236 if(! GTK_WIDGET_NO_WINDOW(cur) )
237 continue;
238
239 if(cur->allocation.x <= event->x
240 && cur->allocation.y <= event->y
241 && (cur->allocation.x + cur->allocation.width) > event->x
242 && (cur->allocation.y + cur->allocation.height) > event->y
243 && g_object_get_data (G_OBJECT(cur), "mate_popup_menu_nowindow"))
244 {
245 new_widget = cur;
246 break;
247 }
248 }
249 if(new_widget)
250 widget = new_widget;
251 else
252 break;
253 } while(widget && GTK_IS_CONTAINER (widget) && GTK_WIDGET_NO_WINDOW(widget));
254
255 if(!widget || !g_object_get_data (G_OBJECT(widget), "mate_popup_menu"))
256 {
257 return TRUE;
258 }
259 }
260 else
261 new_widget = widget;
262
263 return real_popup_button_pressed (new_widget, event, data);
264}

303void
304mate_popup_menu_attach (GtkWidget *popup, GtkWidget *widget,
305 gpointer user_data)
306{
307 GtkWidget *ev_widget;
308
309 g_return_if_fail (popup != NULL);
310 g_return_if_fail (GTK_IS_MENU (popup));
311 g_return_if_fail (widget != NULL);
312 g_return_if_fail (GTK_IS_WIDGET (widget));
313
314 if(g_object_get_data (G_OBJECT (widget), "mate_popup_menu"))
315 return;
316
317 g_object_set_data (G_OBJECT (widget), "mate_popup_menu", popup);
318
319 /* This operation can fail if someone is trying to set a popup on e.g. an uncontained label, so we do it first. */
320 for(ev_widget = widget; ev_widget && GTK_WIDGET_NO_WINDOW(ev_widget); ev_widget = ev_widget->parent)
321 {
322 g_object_set_data (G_OBJECT (ev_widget), "mate_popup_menu_nowindow", GUINT_TO_POINTER(1));
323 }
324
325 g_return_if_fail (ev_widget);
326
327 /* Ref/sink the popup menu so that we take "ownership" of it */
328
329 g_object_ref_sink (popup);
330
331 /* Store the user data pointer in the widget -- we will use it later when the menu has to be
332 * invoked.
333 */
334
335 g_object_set_data (G_OBJECT (widget), "mate_popup_menu_attach_user_data", user_data);
336 g_object_set_data (G_OBJECT (widget), "mate_popup_menu", user_data);
337
338 /* Prepare the widget to accept button presses -- the proper assertions will be
339 * shouted by gtk_widget_set_events().
340 */
341
342 gtk_widget_add_events (ev_widget, GDK_BUTTON_PRESS_MASK |
343 GDK_KEY_PRESS_MASK);
344
345 g_signal_connect (widget, "button_press_event",
346 G_CALLBACK (popup_button_pressed), popup);
347
348 g_signal_connect (G_OBJECT (widget), "popup_menu",
349 G_CALLBACK (popup_menu_pressed), popup);
350
351 if (ev_widget != widget) {
352 GClosure *closure;
353
354 closure = g_cclosure_new (G_CALLBACK (relay_popup_button_pressed),
355 popup,
356 NULL);
357 g_object_watch_closure (G_OBJECT (widget), closure);
358 g_signal_connect_closure (ev_widget, "button_press_event",
359 closure, FALSE);
360 }
361
362 /* This callback will unref the popup menu when the widget it is attached to gets destroyed. */
363 g_signal_connect (widget, "destroy",
364 G_CALLBACK (popup_attach_widget_destroyed),
365 popup);
366}

391void
392mate_popup_menu_do_popup (GtkWidget *popup, GtkMenuPositionFunc pos_func, gpointer pos_data,
393 GdkEventButton *event, gpointer user_data, GtkWidget *for_widget)
394{
395 guint button;
396 guint32 timestamp;
397
398 g_return_if_fail (popup != NULL);
399 g_return_if_fail (GTK_IS_WIDGET (popup));
400
401 /* Store the user data in the menu for when a callback is activated -- if it is a
402 * Mate-generated menu, then the user data will be passed on to callbacks by our custom
403 * marshaller.
404 */
405
406 g_object_set_data (G_OBJECT (popup), "mate_popup_menu_do_popup_user_data", user_data);
407 g_object_set_data (G_OBJECT (popup), "mate_popup_menu_do_popup_for_widget", for_widget);
408
409 if (event) {
410 button = event->button;
411 timestamp = event->time;
412 } else {
413 button = 0;
414 timestamp = GDK_CURRENT_TIME;
415 }
416
417 gtk_menu_popup (GTK_MENU (popup), NULL, NULL, pos_func, pos_data, button, timestamp);
418}

427/* Returns the index of the active item in the specified menu, or -1 if none is active */
428static int
429get_active_index (GtkMenu *menu)
430{
431 GList *l;
432 GtkWidget *active;
433 int i;
434
435 active = g_object_get_data (G_OBJECT (menu), "mate_popup_menu_active_item");
436
437 for (i = 0, l = GTK_MENU_SHELL (menu)->children; l; l = l->next, i++)
438 if (active == l->data)
439 return i;
440
441 return -1;
442}

458int
459mate_popup_menu_do_popup_modal (GtkWidget *popup, GtkMenuPositionFunc pos_func, gpointer pos_data,
460 GdkEventButton *event, gpointer user_data, GtkWidget *for_widget)
461{
462 guint id;
463 guint button;
464 guint32 timestamp;
465
466 g_return_val_if_fail (popup != NULL, -1);
467 g_return_val_if_fail (GTK_IS_WIDGET (popup), -1);
468
469 /* Connect to the deactivation signal to be able to quit our modal main loop */
470
471 id = g_signal_connect (popup, "deactivate",
472 G_CALLBACK (menu_shell_deactivated),
473 NULL);
474
475 g_object_set_data (G_OBJECT (popup), "mate_popup_menu_active_item", NULL);
476 g_object_set_data (G_OBJECT (popup), "mate_popup_menu_do_popup_user_data", user_data);
477 g_object_set_data (G_OBJECT (popup), "mate_popup_menu_do_popup_for_widget", for_widget);
478
479 if (event) {
480 button = event->button;
481 timestamp = event->time;
482 } else {
483 button = 0;
484 timestamp = GDK_CURRENT_TIME;
485 }
486#ifdef HAVE_GTK_MULTIHEAD
487 gtk_menu_set_screen (GTK_MENU (popup), gtk_widget_get_screen (for_widget));
488#endif
489 gtk_menu_popup (GTK_MENU (popup), NULL, NULL, pos_func, pos_data, button, timestamp);
490 gtk_grab_add (popup);
491 gtk_main ();
492 gtk_grab_remove (popup);
493
494 g_signal_handler_disconnect (G_OBJECT (popup), id);
495
496 return get_active_index (GTK_MENU (popup));
497}

506void
507mate_popup_menu_append (GtkWidget *popup, MateUIInfo *uiinfo)
508{
509 MateUIBuilderData uibdata;
510 gint length;
511
512 g_return_if_fail (uiinfo != NULL);
513
514 uibdata.connect_func = popup_connect_func;
515 uibdata.data = NULL;
516 uibdata.is_interp = TRUE;
517 uibdata.relay_func = NULL;
518 uibdata.destroy_func = NULL;
519
520 for (length = 0; uiinfo[length].type != MATE_APP_UI_ENDOFINFO; length++)
521 if (uiinfo[length].type == MATE_APP_UI_ITEM_CONFIGURABLE)
522 mate_app_ui_configure_configurable (uiinfo + length);
523
524 global_menushell_hack = popup;
525 mate_app_fill_menu_custom (GTK_MENU_SHELL (popup), uiinfo,
526 &uibdata, gtk_menu_get_accel_group(GTK_MENU(popup)), FALSE, 0);
527 global_menushell_hack = NULL;
528}
