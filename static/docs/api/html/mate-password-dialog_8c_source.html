104static void
105mate_password_dialog_set_message (MatePasswordDialog *password_dialog, const gchar *message, gboolean markup)
106{
107 MatePasswordDialogDetails *priv = password_dialog->details;
108
109 if (message != NULL && message[0] != '\0') {
110 gtk_label_set_use_markup (GTK_LABEL (priv->message_label), markup);
111 gtk_label_set_label (GTK_LABEL (priv->message_label), message);
112 gtk_widget_show (priv->message_label);
113 } else {
114 gtk_label_set_label (GTK_LABEL (priv->message_label), "");
115 gtk_widget_hide (priv->message_label);
116 }
117}

119static void
120mate_password_dialog_set_anon (MatePasswordDialog *password_dialog, gboolean anonymous, gboolean explicit)
121{
122 MatePasswordDialogDetails *priv = password_dialog->details;
123
124 priv->anonymous = anonymous;
125 priv->anonymous_set = explicit;
126
127 if (anonymous) {
128 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (priv->connect_with_no_userpass_button), TRUE);

130 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (priv->connect_with_userpass_button), TRUE);
131 }
132
133 gtk_widget_set_sensitive (priv->table_alignment, !anonymous || !priv->show_userpass_buttons);
134
135 g_object_notify (G_OBJECT (password_dialog), "anonymous");

151static GtkWidget *
152add_row (GtkWidget *table, int row, const char *label_text, GtkWidget *entry)
153{
154 GtkWidget *label;
155
156 label = gtk_label_new_with_mnemonic (label_text);
157 gtk_misc_set_alignment (GTK_MISC (label), 0.0, 0.5);
158
159 gtk_table_attach (GTK_TABLE (table), label,
160 0, 1, row, row + 1,
161 GTK_FILL, GTK_EXPAND | GTK_FILL, 0, 0);
162 gtk_table_attach_defaults (GTK_TABLE (table), entry,
163 1, 2, row, row + 1);
164 gtk_label_set_mnemonic_widget (GTK_LABEL (label), entry);
165
166 return label;
167}

175static void
176update_entries_table (MatePasswordDialog *dialog)
177{
178 MatePasswordDialogDetails *priv = dialog->details;
179 int row = 0, n_rows;
180
181 gtk_container_foreach (GTK_CONTAINER (priv->table),
182 (GtkCallback) remove_child, priv->table);
183
184 n_rows = priv->show_username +
185 priv->show_domain +
186 priv->show_password +
187 priv->show_new_password * (2 + priv->show_new_password_quality);
188 if (n_rows == 0) {
189 gtk_widget_hide (priv->table);
190 return;
191 }
192
193 gtk_table_resize (GTK_TABLE (priv->table), n_rows, 2);
194
195 if (priv->show_username) {
196 add_row (priv->table, row++, _("_Username:"), priv->username_entry);
197 }
198 if (priv->show_domain) {
199 add_row (priv->table, row++, _("_Domain:"), priv->domain_entry);
200 }
201 if (priv->show_password) {
202 add_row (priv->table, row++, _("_Password:"), priv->password_entry);
203 }
204 if (priv->show_new_password) {
205 add_row (priv->table, row++, _("_New password:"), priv->new_password_entry);
206 add_row (priv->table, row++, _("Con_firm password:"), priv->confirm_new_password_entry);
207 }
208 if (priv->show_new_password && priv->show_new_password_quality) {
209 add_row (priv->table, row++, _("Password quality:"), priv->quality_meter);
210 }
211
212 gtk_widget_show_all (priv->table);
213}

272static void
273new_password_entries_changed (GtkWidget *entry, MatePasswordDialog *password_dialog)
274{
275 MatePasswordDialogDetails *priv = password_dialog->details;
276 const gchar *new_password, *confirm_password;
277 gdouble quality = 0.0;
278 gboolean accept;
279
280 new_password = gtk_entry_get_text (GTK_ENTRY (priv->new_password_entry));
281 confirm_password = gtk_entry_get_text (GTK_ENTRY (priv->confirm_new_password_entry));
282
283 if (priv->quality_func) {
284 quality = priv->quality_func (password_dialog,
285 new_password,
286 priv->quality_data);
287 }
288
289 accept = strcmp (new_password, confirm_password) == 0;
290 /* && (!priv->show_new_password_quality || quality > 0.5); */
291
292 gtk_progress_bar_set_fraction (GTK_PROGRESS_BAR (priv->quality_meter), quality);
293
294 gtk_dialog_set_response_sensitive (GTK_DIALOG (password_dialog),
295 GTK_RESPONSE_OK, accept);
296}

298static void
299mate_password_dialog_init (MatePasswordDialog *password_dialog)
300{
301 MatePasswordDialogDetails *priv;
302 GtkDialog *dialog = GTK_DIALOG (password_dialog);
303 GtkWindow *window = GTK_WINDOW (password_dialog);
304 GtkWidget *hbox, *main_vbox, *vbox, *icon;
305 GSList *group;
306
307 priv = password_dialog->details = MATE_PASSWORD_DIALOG_GET_PRIVATE (password_dialog);
308
309 priv->show_username = TRUE;
310 priv->show_password = TRUE;
311
312 /* Set the dialog up with HIG properties */
313 gtk_dialog_set_has_separator (dialog, FALSE);
314 gtk_container_set_border_width (GTK_CONTAINER (dialog), 5);
315 gtk_box_set_spacing (GTK_BOX (dialog->vbox), 2); /* 2 * 5 + 2 = 12 */
316 gtk_container_set_border_width (GTK_CONTAINER (dialog->action_area), 5);
317 gtk_box_set_spacing (GTK_BOX (dialog->action_area), 6);
318
319 gtk_window_set_resizable (window, FALSE);
320 gtk_window_set_icon_name (window, GTK_STOCK_DIALOG_AUTHENTICATION);
321
322 gtk_dialog_add_buttons (dialog,
323 GTK_STOCK_CANCEL, GTK_RESPONSE_CANCEL,
324 _("Co_nnect"), GTK_RESPONSE_OK,
325 NULL);
326 gtk_dialog_set_default_response (dialog, GTK_RESPONSE_OK);
327
328 /* Build contents */
329 hbox = gtk_hbox_new (FALSE, 12);
330 gtk_container_set_border_width (GTK_CONTAINER (hbox), 5);
331 gtk_box_pack_start (GTK_BOX (dialog->vbox), hbox, TRUE, TRUE, 0);
332
333 icon = gtk_image_new_from_stock (GTK_STOCK_DIALOG_AUTHENTICATION, GTK_ICON_SIZE_DIALOG);
334 gtk_misc_set_alignment (GTK_MISC (icon), 0.5, 0.0);
335 gtk_box_pack_start (GTK_BOX (hbox), icon, FALSE, FALSE, 0);
336
337 main_vbox = gtk_vbox_new (FALSE, 18);
338 gtk_box_pack_start (GTK_BOX (hbox), main_vbox, TRUE, TRUE, 0);
339

341 gtk_misc_set_alignment (GTK_MISC (priv->message_label), 0.0, 0.5);
342 gtk_label_set_line_wrap (GTK_LABEL (priv->message_label), TRUE);
343 gtk_box_pack_start (GTK_BOX (main_vbox), GTK_WIDGET (priv->message_label),
344 FALSE, FALSE, 0);
345 gtk_widget_set_no_show_all (priv->message_label, TRUE);
346
347 vbox = gtk_vbox_new (FALSE, 6);
348 gtk_box_pack_start (GTK_BOX (main_vbox), vbox, FALSE, FALSE, 0);
349
350 /* the radio buttons for anonymous login */
351 priv->radio_vbox = gtk_vbox_new (FALSE, 6);
352 gtk_box_pack_start (GTK_BOX (vbox), priv->radio_vbox,
353 FALSE, FALSE, 0);
354 gtk_widget_set_no_show_all (priv->radio_vbox, TRUE);
355
356 priv->connect_with_no_userpass_button =
357 gtk_radio_button_new_with_mnemonic (NULL, _("Connect _anonymously"));
358 group = gtk_radio_button_get_group (
359 GTK_RADIO_BUTTON (priv->connect_with_no_userpass_button));
360 priv->connect_with_userpass_button =
361 gtk_radio_button_new_with_mnemonic (
362 group, _("Connect as u_ser:"));
363
364 g_signal_connect (priv->connect_with_no_userpass_button, "toggled",
365 G_CALLBACK (userpass_radio_button_toggled), password_dialog);
366 g_signal_connect (priv->connect_with_userpass_button, "toggled",
367 G_CALLBACK (userpass_radio_button_toggled), password_dialog);
368
369 gtk_box_pack_start (GTK_BOX (priv->radio_vbox),
370 priv->connect_with_no_userpass_button,
371 FALSE, FALSE, 0);
372 gtk_box_pack_start (GTK_BOX (priv->radio_vbox),
373 priv->connect_with_userpass_button,
374 FALSE, FALSE, 0);
375 gtk_widget_show (priv->connect_with_no_userpass_button);
376 gtk_widget_show (priv->connect_with_userpass_button);
377
378 /* The table that holds the entries */
379 priv->table_alignment = gtk_alignment_new (0.0, 0.0, 1.0, 1.0);
380 gtk_box_pack_start (GTK_BOX (vbox), priv->table_alignment,
381 FALSE, FALSE, 0);
382
383 priv->table = gtk_table_new (3, 2, FALSE);
384 gtk_table_set_col_spacings (GTK_TABLE (priv->table), 12);
385 gtk_table_set_row_spacings (GTK_TABLE (priv->table), 6);
386 gtk_container_add (GTK_CONTAINER (priv->table_alignment), priv->table);
387
388 priv->username_entry = g_object_ref_sink (gtk_entry_new ());
389 priv->domain_entry = g_object_ref_sink (gtk_entry_new ());
390 priv->password_entry = g_object_ref_sink (gtk_entry_new ());
391 gtk_entry_set_visibility (GTK_ENTRY (priv->password_entry), FALSE);
392 priv->new_password_entry = g_object_ref_sink (gtk_entry_new ());
393 gtk_entry_set_visibility (GTK_ENTRY (priv->new_password_entry), FALSE);
394 priv->confirm_new_password_entry = g_object_ref_sink (gtk_entry_new ());
395 gtk_entry_set_visibility (GTK_ENTRY (priv->confirm_new_password_entry), FALSE);

397 g_signal_connect (priv->username_entry, "activate",
398 G_CALLBACK (username_entry_activate), password_dialog);

403 g_signal_connect (priv->new_password_entry, "activate",
404 G_CALLBACK (new_password_entry_activate), password_dialog);

406 G_CALLBACK (gtk_window_activate_default),
407 password_dialog);
408 g_signal_connect (priv->new_password_entry, "changed",
409 G_CALLBACK (new_password_entries_changed), password_dialog);

412
413 priv->quality_meter = g_object_ref_sink (gtk_progress_bar_new ());

415 update_entries_table (password_dialog);
416
417 gtk_widget_show_all (hbox);
418
419 priv->remember_box = gtk_vbox_new (FALSE, 6);
420 gtk_box_pack_start (GTK_BOX (vbox), priv->remember_box,

422
423 priv->remember_buttons[0] =
424 gtk_radio_button_new_with_mnemonic (NULL, _("_Forget password immediately"));

437 g_signal_connect (priv->remember_buttons[2], "toggled",

440 gtk_box_pack_start (GTK_BOX (priv->remember_box), priv->remember_buttons[0],

442 gtk_widget_show (priv->remember_buttons[0]);
443 gtk_box_pack_start (GTK_BOX (priv->remember_box), priv->remember_buttons[1],

447 FALSE, FALSE, 0);
448 gtk_widget_show (priv->remember_buttons[2]);
449}

451static GObject *
452mate_password_dialog_constructor (GType type,
453 guint n_construct_properties,
454 GObjectConstructParam *construct_params)
455{
456 GObject *object;
457 MatePasswordDialog *dialog;
458 MatePasswordDialogDetails *priv;
459 const gchar *username;
460
461 object = G_OBJECT_CLASS (mate_password_dialog_parent_class)->constructor
462 (type, n_construct_properties, construct_params);
463
464 dialog = MATE_PASSWORD_DIALOG (object);
465 priv = dialog->details;
466
467 if (priv->show_userpass_buttons_set &&
468 !priv->anonymous_set) {
469 username = gtk_entry_get_text (GTK_ENTRY (priv->username_entry));
470 mate_password_dialog_set_anon (dialog,
471 username == NULL || username[0] == '\0', FALSE);
472 }
473
474 return object;
475}

523 mate_password_dialog_set_show_userpass_buttons (dialog, g_value_get_boolean (value));
524 break;
525 case PROP_SHOW_REMEMBER:
526 mate_password_dialog_set_show_remember (dialog, g_value_get_boolean (value));
527 break;
528 case PROP_READONLY_USERNAME:
529 mate_password_dialog_set_readonly_username (dialog, g_value_get_boolean (value));

538 mate_password_dialog_set_remember (dialog, g_value_get_enum (value));
539 break;
540 case PROP_MESSAGE:
541 mate_password_dialog_set_message (dialog, g_value_get_string (value), FALSE);
542 break;
543 case PROP_MESSAGE_MARKUP:
544 mate_password_dialog_set_message (dialog, g_value_get_string (value), TRUE);
545 break;
546 case PROP_USERNAME:
547 mate_password_dialog_set_username (dialog, g_value_get_string (value));
548 break;
549 case PROP_DOMAIN:
550 mate_password_dialog_set_domain (dialog, g_value_get_string (value));

562 }

565static void
566mate_password_dialog_get_property (GObject *object,
567 guint prop_id,
568 GValue *value,
569 GParamSpec *pspec)
570{
571 MatePasswordDialog *dialog = MATE_PASSWORD_DIALOG (object);
572 MatePasswordDialogDetails *priv = dialog->details;
573
574 switch (prop_id)
575 {
576 case PROP_SHOW_USERNAME:
577 g_value_set_boolean (value, priv->show_username);
578 break;
579 case PROP_SHOW_DOMAIN:
580 g_value_set_boolean (value, priv->show_domain);
581 break;
582 case PROP_SHOW_PASSWORD:
583 g_value_set_boolean (value, priv->show_password);
584 break;
585 case PROP_SHOW_NEW_PASSWORD:
586 g_value_set_boolean (value, priv->show_new_password);
587 break;
588 case PROP_SHOW_NEW_PASSWORD_QUALITY:
589 g_value_set_boolean (value, priv->show_new_password_quality);
590 break;
591 case PROP_SHOW_USERPASS_BUTTONS:
592 g_value_set_boolean (value, priv->show_userpass_buttons);
593 break;
594 case PROP_SHOW_REMEMBER:
595 g_value_set_boolean (value, priv->show_remember);
596 break;
597 case PROP_READONLY_USERNAME:
598 g_value_set_boolean (value, priv->readonly_username);
599 break;
600 case PROP_READONLY_DOMAIN:
601 g_value_set_boolean (value, priv->readonly_domain);
602 break;
603 case PROP_ANONYMOUS:
604 g_value_set_boolean (value, mate_password_dialog_anon_selected (dialog));

926/* GtkDialog callbacks */
927static void
928mate_password_dialog_show (GtkWidget *widget)
929{
930 MatePasswordDialog *password_dialog = MATE_PASSWORD_DIALOG (widget);
931 MatePasswordDialogDetails *priv = password_dialog->details;
932 GtkWidget *focus = NULL;
933 const gchar *text;
934
935 GTK_WIDGET_CLASS (mate_password_dialog_parent_class)->show (widget);
936
937 if (GTK_WIDGET_VISIBLE (priv->password_entry) &&
938 GTK_WIDGET_IS_SENSITIVE (priv->password_entry) &&
939 priv->show_password)
940 /* Password is the default place to focus */
941 focus = priv->password_entry;
942
943 if (GTK_WIDGET_VISIBLE (priv->domain_entry) &&
944 GTK_WIDGET_IS_SENSITIVE (priv->domain_entry) &&
945 priv->show_domain) {
946 /* Empty domain entry gets focus */
947 text = gtk_entry_get_text (GTK_ENTRY (priv->domain_entry));
948 if (!focus || !text || !text[0])
949 focus = priv->domain_entry;
950 }
951
952 if (GTK_WIDGET_VISIBLE (priv->username_entry) &&
953 GTK_WIDGET_IS_SENSITIVE (priv->username_entry) &&
954 priv->show_username) {
955 /* Empty username entry gets focus */
956 text = gtk_entry_get_text (GTK_ENTRY (priv->username_entry));
957 if (!focus || !text || !text[0])
958 focus = priv->username_entry;
959 }
960
961 if (focus) {
962 gtk_widget_grab_focus (focus);
963 }
964}
