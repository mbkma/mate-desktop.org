155GSList*
156proctable_get_columns_order(GtkTreeView *treeview)
157{
158 GList *columns, *col;
159 GSList *order = NULL;
160
161 columns = gtk_tree_view_get_columns(treeview);
162
163 for(col = columns; col; col = col->next)
164 {
165 int id;
166
167 id = gtk_tree_view_column_get_sort_column_id(static_cast<GtkTreeViewColumn*>(col->data));
168 order = g_slist_prepend(order, GINT_TO_POINTER(id));
169 }
170
171 g_list_free(columns);
172
173 order = g_slist_reverse(order);
174
175 return order;
176}

572static void
573get_process_name (ProcInfo *info,
574 const gchar *cmd, const GStrv args)
575{
576 if (args) {
577 // look for /usr/bin/very_long_name
578 // and also /usr/bin/interpreter /usr/.../very_long_name
579 // which may have use prctl to alter 'cmd' name
580 for (int i = 0; i != 2 && args[i]; ++i) {
581 char* basename;
582 basename = g_path_get_basename(args[i]);
583
584 if (g_str_has_prefix(basename, cmd)) {
585 info->name = basename;
586 return;
587 }
588
589 g_free(basename);
590 }
591 }
592
593 info->name = g_strdup (cmd);
594}

596std::string
597ProcInfo::lookup_user(guint uid)
598{
599 typedef std::pair<ProcInfo::UserMap::iterator, bool> Pair;
600 ProcInfo::UserMap::value_type hint(uid, "");
601 Pair p(ProcInfo::users.insert(hint));
602
603 // procman_debug("User lookup for uid %u: %s", uid, (p.second ? "MISS" : "HIT"));
604
605 if (p.second) {
606 struct passwd* pwd;
607 pwd = getpwuid(uid);
608
609 if (pwd && pwd->pw_name)
610 p.first->second = pwd->pw_name;
611 else {
612 char username[16];
613 g_sprintf(username, "%u", uid);
614 p.first->second = username;
615 }
616 }
617
618 return p.first->second;
619}

631static void get_process_memory_writable(ProcInfo *info)
632{
633 glibtop_proc_map buf;
634 glibtop_map_entry *maps;
635
636 maps = glibtop_get_proc_map(&buf, info->pid);
637
638 gulong memwritable = 0;
639 const unsigned number = buf.number;
640
641 for (unsigned i = 0; i < number; ++i) {
642#ifdef __linux__
643 memwritable += maps[i].private_dirty;
644#else
645 if (maps[i].perm & GLIBTOP_MAP_PERM_WRITE)
646 memwritable += maps[i].size;
647#endif
648 }
649
650 info->memwritable = memwritable;
651
652 g_free(maps);
653}

655static void
656get_process_memory_info(ProcInfo *info)
657{
658 glibtop_proc_mem procmem;
659
660#ifdef HAVE_WNCK
661 info->memxserver = 0;
662#ifdef GDK_WINDOWING_X11
663 if (GDK_IS_X11_DISPLAY (gdk_display_get_default ())) {
664 WnckResourceUsage xresources;
665
666 wnck_pid_read_resource_usage (gdk_screen_get_display (gdk_screen_get_default ()),
667 info->pid,
668 &xresources);
669
670 info->memxserver = xresources.total_bytes_estimate;
671 }
672#endif // GDK_WINDOWING_X11
673#endif // HAVE_WNCK
674
675 glibtop_get_proc_mem(&procmem, info->pid);
676
677 info->vmsize = procmem.vsize;
678 info->memres = procmem.resident;
679 info->memshared = procmem.share;
680
681 get_process_memory_writable(info);
682
683 // fake the smart memory column if writable is not available
684 info->mem = info->memwritable ? info->memwritable : info->memres;
685#ifdef HAVE_WNCK
686 info->mem += info->memxserver;
687#endif
688}

775/* Removing a node with children - make sure the children are queued
776** to be readded.
777*/
778template<typename List>
779static void
780remove_info_from_tree (ProcData *procdata, GtkTreeModel *model,
781 ProcInfo *current, List &orphans, unsigned lvl = 0)
782{
783 GtkTreeIter child_node;
784
785 if (std::find(orphans.begin(), orphans.end(), current) != orphans.end()) {
786 procman_debug("[%u] %d already removed from tree", lvl, int(current->pid));
787 return;
788 }
789
790 procman_debug("[%u] pid %d, %d children", lvl, int(current->pid),
791 gtk_tree_model_iter_n_children(model, &current->node));
792
793 // it is not possible to iterate&erase over a treeview so instead we
794 // just pop one child after another and recursively remove it and
795 // its children
796
797 while (gtk_tree_model_iter_children(model, &child_node, &current->node)) {
798 ProcInfo *child = 0;
799 gtk_tree_model_get(model, &child_node, COL_POINTER, &child, -1);
800 remove_info_from_tree(procdata, model, child, orphans, lvl + 1);
801 }
802
803 g_assert(not gtk_tree_model_iter_has_child(model, &current->node));
804
805 if (procdata->selected_process == current)
806 procdata->selected_process = NULL;
807
808 orphans.push_back(current);
809 gtk_tree_store_remove(GTK_TREE_STORE(model), &current->node);
810 procman::poison(current->node, 0x69);
811}

843static void
844update_info (ProcData *procdata, ProcInfo *info)
845{
846 glibtop_proc_state procstate;
847 glibtop_proc_uid procuid;
848 glibtop_proc_time proctime;
849 glibtop_proc_kernel prockernel;
850 glibtop_proc_io procio;
851
852 glibtop_get_proc_kernel(&prockernel, info->pid);
853 g_strlcpy(info->wchan, prockernel.wchan, sizeof info->wchan);
854
855 glibtop_get_proc_state (&procstate, info->pid);
856 info->status = procstate.state;
857
858 glibtop_get_proc_uid (&procuid, info->pid);
859 glibtop_get_proc_time (&proctime, info->pid);
860 glibtop_get_proc_io (&procio, info->pid);
861
862 get_process_memory_info(info);
863
864 info->set_user(procstate.uid);
865
866 // if the cpu time has increased reset the status to running
867 // regardless of kernel state (https://bugzilla.gnome.org/606579)
868 guint64 difference = proctime.rtime - info->cpu_time;
869 if (difference > 0)
870 info->status = GLIBTOP_PROCESS_RUNNING;
871 info->pcpu = difference * 100 / procdata->cpu_total_time;
872 info->pcpu = MIN(info->pcpu, 100);
873
874 if (not procdata->config.solaris_mode)
875 info->pcpu *= procdata->config.num_cpus;
876
877 ProcInfo::cpu_times[info->pid] = info->cpu_time = proctime.rtime;
878 info->nice = procuid.nice;
879
880 gdouble update_interval_seconds = procdata->config.update_interval / 1000.0;
881 difference = procio.disk_wbytes - info->disk_write_bytes_total;
882 info->disk_write_bytes_current = difference > update_interval_seconds ? difference/update_interval_seconds : 0ULL;
883 difference = procio.disk_rbytes - info->disk_read_bytes_total;
884 info->disk_read_bytes_current = difference > update_interval_seconds ? difference/update_interval_seconds : 0ULL;
885
886 info->disk_write_bytes_total = procio.disk_wbytes;
887 info->disk_read_bytes_total = procio.disk_rbytes;
888
889 // set the ppid only if one can exist
890 // i.e. pid=0 can never have a parent
891 if (info->pid > 0) {
892 info->ppid = procuid.ppid;
893 }
894
895 g_assert(info->pid != info->ppid);
896 g_assert(info->ppid != -1 || info->pid == 0);
897
898 /* get cgroup data */
899 get_process_cgroup_info(info);
900
901 get_process_systemd_info(info);
902}

956static void
957refresh_list (ProcData *procdata, const pid_t* pid_list, const guint n)
958{
959 typedef std::list<ProcInfo*> ProcList;
960 ProcList addition;
961
962 GtkTreeModel *model = gtk_tree_view_get_model (GTK_TREE_VIEW (procdata->tree));
963 guint i;
964
965 // Add or update processes in the process list
966 for(i = 0; i < n; ++i) {
967 ProcInfo *info = ProcInfo::find(pid_list[i]);
968
969 if (!info) {
970 info = new ProcInfo(pid_list[i]);
971 ProcInfo::all[info->pid] = info;
972 addition.push_back(info);
973 }
974
975 update_info (procdata, info);
976 }
977
978
979 // Remove dead processes from the process list and from the
980 // tree. children are queued to be readded at the right place
981 // in the tree.
982
983 const std::set<pid_t> pids(pid_list, pid_list + n);
984
985 ProcInfo::Iterator it(ProcInfo::begin());
986
987 while (it != ProcInfo::end()) {
988 ProcInfo * const info = it->second;
989 ProcInfo::Iterator next(it);
990 ++next;
991
992 if (pids.find(info->pid) == pids.end()) {
993 procman_debug("ripping %d", info->pid);
994 remove_info_from_tree(procdata, model, info, addition);
995 addition.remove(info);
996 ProcInfo::all.erase(it);
997 delete info;
998 }
999
1000 it = next;
1001 }
1002
1003 // INVARIANT
1004 // pid_list == ProcInfo::all + addition
1005
1006
1007 if (procdata->config.show_tree) {
1008
1009 // insert process in the tree. walk through the addition list
1010 // (new process + process that have a new parent). This loop
1011 // handles the dependencies because we cannot insert a process
1012 // until its parent is in the tree.
1013
1014 std::set<pid_t> in_tree(pids);
1015
1016 for (ProcList::iterator it(addition.begin()); it != addition.end(); ++it)
1017 in_tree.erase((*it)->pid);
1018
1019
1020 while (not addition.empty()) {
1021 procman_debug("looking for %d parents", int(addition.size()));
1022 ProcList::iterator it(addition.begin());
1023
1024 while (it != addition.end()) {
1025 procman_debug("looking for %d's parent with ppid %d",
1026 int((*it)->pid), int((*it)->ppid));
1027
1028
1029 // inserts the process in the treeview if :
1030 // - it has no parent (ppid = -1),
1031 // ie it is for example the [kernel] on FreeBSD
1032 // - it is init
1033 // - its parent is already in tree
1034 // - its parent is unreachable
1035 //
1036 // rounds == 2 means that addition contains processes with
1037 // unreachable parents
1038 //
1039 // FIXME: this is broken if the unreachable parent becomes active
1040 // i.e. it gets active or changes ower
1041 // so we just clear the tree on __each__ update
1042 // see proctable_update_list (ProcData * const procdata)
1043
1044
1045 if ((*it)->ppid <= 0 or in_tree.find((*it)->ppid) != in_tree.end()) {
1046 insert_info_to_tree(*it, procdata);
1047 in_tree.insert((*it)->pid);
1048 it = addition.erase(it);
1049 continue;
1050 }
1051
1052 ProcInfo *parent = ProcInfo::find((*it)->ppid);
1053 // if the parent is unreachable
1054 if (not parent) {
1055 // or std::find(addition.begin(), addition.end(), parent) == addition.end()) {
1056 insert_info_to_tree(*it, procdata, true);
1057 in_tree.insert((*it)->pid);
1058 it = addition.erase(it);
1059 continue;
1060 }
1061
1062 ++it;
1063 }
1064 }
1065 }
1066 else {
1067 // don't care of the tree
1068 for (ProcList::iterator it(addition.begin()); it != addition.end(); ++it)
1069 insert_info_to_tree(*it, procdata);
1070 }
1071
1072
1073 for (ProcInfo::Iterator it(ProcInfo::begin()); it != ProcInfo::end(); ++it)
1074 update_info_mutable_cols(it->second);
1075}

1078static void
1079proctable_update_list (ProcData * const procdata)
1080{
1081 pid_t* pid_list;
1082 glibtop_proclist proclist;
1083 glibtop_cpu cpu;
1084 gint which, arg;
1085 procman::SelectionMemento selection;
1086
1087 switch (procdata->config.whose_process) {
1088 case ALL_PROCESSES:
1089 which = GLIBTOP_KERN_PROC_ALL;
1090 arg = 0;
1091 break;
1092
1093 case ACTIVE_PROCESSES:
1094 which = GLIBTOP_KERN_PROC_ALL | GLIBTOP_EXCLUDE_IDLE;
1095 arg = 0;
1096 if (procdata->config.show_tree)
1097 {
1098 selection.save(procdata->tree);
1099 }
1100 break;
1101
1102 default:
1103 which = GLIBTOP_KERN_PROC_UID;
1104 arg = getuid ();
1105 if (procdata->config.show_tree)
1106 {
1107 selection.save(procdata->tree);
1108 }
1109 break;
1110 }
1111
1112 pid_list = glibtop_get_proclist (&proclist, which, arg);
1113
1114 /* FIXME: total cpu time elapsed should be calculated on an individual basis here
1115 ** should probably have a total_time_last gint in the ProcInfo structure */
1116 glibtop_get_cpu (&cpu);
1117 procdata->cpu_total_time = MAX(cpu.total - procdata->cpu_total_time_last, 1);
1118 procdata->cpu_total_time_last = cpu.total;
1119
1120 // FIXME: not sure if glibtop always returns a sorted list of pid
1121 // but it is important otherwise refresh_list won't find the parent
1122 std::sort(pid_list, pid_list + proclist.number);
1123
1124 refresh_list (procdata, pid_list, proclist.number);
1125
1126 selection.restore(procdata->tree);
1127
1128 g_free (pid_list);
1129
1130 /* proclist.number == g_list_length(procdata->info) == g_hash_table_size(procdata->pids) */
1131}
