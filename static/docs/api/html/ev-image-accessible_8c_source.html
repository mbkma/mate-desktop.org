118static AtkStateSet *
119ev_image_accessible_ref_state_set (AtkObject *atk_object)
120{
121 AtkStateSet *state_set;
122 AtkStateSet *copy_set;
123 AtkStateSet *page_accessible_state_set;
124 EvImageAccessible *self;
125 EvViewAccessible *view_accessible;
126 gint page;
127
128 self = EV_IMAGE_ACCESSIBLE (atk_object);
129 state_set = ATK_OBJECT_CLASS (ev_image_accessible_parent_class)->ref_state_set (atk_object);
130 atk_state_set_clear_states (state_set);
131
132 page_accessible_state_set = atk_object_ref_state_set (ATK_OBJECT (self->priv->page));
133 copy_set = atk_state_set_or_sets (state_set, page_accessible_state_set);
134 atk_state_set_remove_state (copy_set, ATK_STATE_FOCUSABLE);
135 atk_state_set_remove_state (copy_set, ATK_STATE_FOCUSED);
136
137 view_accessible = ev_page_accessible_get_view_accessible (self->priv->page);
138 page = ev_page_accessible_get_page (self->priv->page);
139 if (!ev_view_accessible_is_doc_rect_showing (view_accessible, page, &self->priv->area))
140 atk_state_set_remove_state (copy_set, ATK_STATE_SHOWING);
141
142 g_object_unref (state_set);
143 g_object_unref (page_accessible_state_set);
144
145 return copy_set;
146}
