94static void
95mate_property_box_instance_init (MatePropertyBox *property_box)
96{
97 GList * button_list;
98
99 property_box->notebook = gtk_notebook_new ();
100
101 mate_dialog_append_buttons (MATE_DIALOG (property_box),
102 MATE_STOCK_BUTTON_HELP,
103 MATE_STOCK_BUTTON_APPLY,
104 MATE_STOCK_BUTTON_CLOSE,
105 MATE_STOCK_BUTTON_OK,
106 NULL);
107
108 mate_dialog_set_default (MATE_DIALOG (property_box), 3);
109
110 /* This is sort of unattractive */
111
112 button_list = MATE_DIALOG(property_box)->buttons;
113
114 property_box->help_button = GTK_WIDGET(button_list->data);
115 button_list = button_list->next;
116
117 property_box->apply_button = GTK_WIDGET(button_list->data);
118 button_list = button_list->next;
119 gtk_widget_set_sensitive (property_box->apply_button, FALSE);
120
121 property_box->cancel_button = GTK_WIDGET(button_list->data);
122 button_list = button_list->next;
123 property_box->ok_button = GTK_WIDGET(button_list->data);
124
125 g_signal_connect (property_box, "clicked",
126 G_CALLBACK(dialog_clicked_cb),
127 NULL);
128
129 gtk_box_pack_start (GTK_BOX(MATE_DIALOG(property_box)->vbox),
130 property_box->notebook,
131 TRUE, TRUE, 0);
132
133 gtk_widget_show (property_box->notebook);
134}

154static void
155dialog_clicked_cb(MateDialog * dialog, gint button, gpointer data)
156{
157 MatePropertyBox *pbox;
158 GtkWidget *page;
159 gboolean dirty = FALSE;
160 gint cur_page_no;
161
162 g_return_if_fail(dialog != NULL);
163 g_return_if_fail(MATE_IS_PROPERTY_BOX(dialog));
164
165 pbox = MATE_PROPERTY_BOX (dialog);
166
167 cur_page_no = gtk_notebook_get_current_page (GTK_NOTEBOOK (pbox->notebook));
168 if (cur_page_no >= 0) {
169 gint page_no = 0;
170
171 page = gtk_notebook_get_nth_page (GTK_NOTEBOOK (pbox->notebook), 0);
172 while (page != NULL) {
173 dirty = GPOINTER_TO_INT(g_object_get_data (G_OBJECT (page), MATE_PROPERTY_BOX_DIRTY));
174 if (dirty)
175 break;
176
177 page = gtk_notebook_get_nth_page (GTK_NOTEBOOK (pbox->notebook),
178 ++page_no);
179 }
180 } else {
181 page = NULL;
182 dirty = FALSE;
183 }
184
185 switch(button) {
186 case 0:
187 help (MATE_PROPERTY_BOX (dialog));
188 break;
189 case 1:
190 global_apply (MATE_PROPERTY_BOX (dialog));
191 break;
192 case 2:
193 just_close (MATE_PROPERTY_BOX (dialog));
194 break;
195 case 3:
196 if (dirty)
197 apply_and_close (MATE_PROPERTY_BOX (dialog));
198 else
199 just_close (MATE_PROPERTY_BOX (dialog));
200 break;
201
202 default:
203 g_assert_not_reached();
204 }
205}

222void
223mate_property_box_changed (MatePropertyBox *property_box)
224{
225 gint cur_page_no;
226 GtkWidget *page;
227
228 g_return_if_fail (property_box != NULL);
229 g_return_if_fail (MATE_IS_PROPERTY_BOX (property_box));
230 g_return_if_fail (property_box->notebook);
231
232 cur_page_no = gtk_notebook_get_current_page (GTK_NOTEBOOK (property_box->notebook));
233 if (cur_page_no < 0)
234 return;
235
236 page = gtk_notebook_get_nth_page (GTK_NOTEBOOK (property_box->notebook),
237 cur_page_no);
238 g_assert (page != NULL);
239
240 g_object_set_data(G_OBJECT(page), MATE_PROPERTY_BOX_DIRTY, GINT_TO_POINTER(1));
241
242 set_sensitive (property_box, 1);
243}

293static void
294global_apply (MatePropertyBox *property_box)
295{
296 GtkWidget *page;
297 gint n;
298
299 g_return_if_fail(GTK_NOTEBOOK(property_box->notebook)->children != NULL);
300
301 n = 0;
302 page = gtk_notebook_get_nth_page(GTK_NOTEBOOK(property_box->notebook), n);
303 while (page != NULL) {
304 if (g_object_get_data(G_OBJECT(page), MATE_PROPERTY_BOX_DIRTY)) {
305 g_signal_emit (property_box, property_box_signals[APPLY], 0,
306 n);
307 g_object_set_data(G_OBJECT(page), MATE_PROPERTY_BOX_DIRTY, GINT_TO_POINTER(0));
308 }
309
310 page = gtk_notebook_get_nth_page(GTK_NOTEBOOK(property_box->notebook), ++n);
311 }
312
313 /* Emit an apply signal with a button of -1. This means we
314 just finished a global apply. Is this a hack? */
315 g_signal_emit (property_box, property_box_signals[APPLY], 0,
316 (gint) -1);
317
318 /* Doesn't matter which item we use. */
319 set_sensitive (property_box, 0);
320}

356gint
357mate_property_box_append_page (MatePropertyBox *property_box,
358 GtkWidget *child,
359 GtkWidget *tab_label)
360{
361 g_return_val_if_fail (property_box != NULL, -1);
362 g_return_val_if_fail (MATE_IS_PROPERTY_BOX (property_box), -1);
363 g_return_val_if_fail (child != NULL, -1);
364 g_return_val_if_fail (GTK_IS_WIDGET (child), -1);
365 g_return_val_if_fail (tab_label != NULL, -1);
366 g_return_val_if_fail (GTK_IS_WIDGET (tab_label), -1);
367
368 gtk_notebook_append_page (GTK_NOTEBOOK (property_box->notebook),
369 child, tab_label);
370
371 return g_list_length (
372 GTK_NOTEBOOK(property_box->notebook)->children) - 1;
373}
