22static GList *
23list_from_char_array (const char **s)
24{
25 GList *list = NULL;
26 int i;
27
28 for (i = 0; s[i]; i++) {
29 GdkPixbuf *pb;
30
31 pb = gdk_pixbuf_new_from_file (s[i], NULL);
32 if (pb)
33 list = g_list_prepend (list, pb);
34 }
35
36 return list;
37}

145void
146mate_window_icon_init (void)

164

167#endif
