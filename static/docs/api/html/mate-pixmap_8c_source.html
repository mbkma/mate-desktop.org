118GtkWidget*
119mate_pixmap_new_from_file_at_size (const gchar *filename, gint width, gint height)
120{
121 GtkWidget *retval = NULL;
122 GdkPixbuf *pixbuf;
123 GError *error = NULL;
124
125 g_return_val_if_fail (filename != NULL, NULL);
126
127 pixbuf = gdk_pixbuf_new_from_file (filename, &error);
128 if (error != NULL) {
129 g_warning (G_STRLOC ": cannot open %s: %s",
130 filename, error->message);
131 g_error_free (error);
132 }
133 if (pixbuf != NULL) {
134 GdkPixbuf *scaled = gdk_pixbuf_scale_simple (pixbuf,
135 width,
136 height,
137 GDK_INTERP_BILINEAR);
138 retval = g_object_new (MATE_TYPE_PIXMAP, NULL);
139 gtk_image_set_from_pixbuf (GTK_IMAGE (retval), scaled);
140 g_object_unref (G_OBJECT (scaled));
141 g_object_unref (G_OBJECT (pixbuf));
142 } else {
143 retval = g_object_new (MATE_TYPE_PIXMAP, NULL);
144 }
145 return retval;
146}
