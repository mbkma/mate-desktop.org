89gpm_session_logout (GpmSession *session)
90{
91 g_return_val_if_fail (GPM_IS_SESSION (session), FALSE);
92
93 /* no mate-session */
94 if (session->priv->proxy == NULL) {
95 g_warning ("no mate-session");
96 return FALSE;
97 }
98
99 /* we have to use no reply, as the SM calls into g-p-m to get the can_suspend property */
100 dbus_g_proxy_call_no_reply (session->priv->proxy, "Shutdown", G_TYPE_INVALID);
101 return TRUE;

137static void

139{
140 gboolean is_idle;

152static gboolean
153gpm_session_is_idle (GpmSession *session)
154{
155 gboolean ret;
156 gboolean is_idle = FALSE;
157 GError *error = NULL;
158 GValue *value;
159
160 /* no mate-session */
161 if (session->priv->proxy_prop == NULL) {
162 g_warning ("no mate-session");
163 goto out;
164 }
165
166 value = g_new0(GValue, 1);
167 /* find out if this change altered the inhibited state */
168 ret = dbus_g_proxy_call (session->priv->proxy_prop, "Get", &error,
169 G_TYPE_STRING, GPM_SESSION_MANAGER_PRESENCE_INTERFACE,
170 G_TYPE_STRING, "status",
171 G_TYPE_INVALID,
172 G_TYPE_VALUE, value,
173 G_TYPE_INVALID);
174 if (!ret) {
175 g_warning ("failed to get idle status: %s", error->message);
176 g_error_free (error);
177 g_free (value);
178 is_idle = FALSE;
179 goto out;
180 }
181 is_idle = (g_value_get_uint (value) == GPM_SESSION_STATUS_ENUM_IDLE);
182 g_free (value);
183out:
184 return is_idle;
185}

315gboolean
316gpm_session_register_client (GpmSession *session, const gchar *app_id, const gchar *client_startup_id)
317{
318 gboolean ret = FALSE;
319 gchar *client_id = NULL;
320 GError *error = NULL;
321 DBusGConnection *connection;
322
323 g_return_val_if_fail (GPM_IS_SESSION (session), FALSE);
324
325 /* no mate-session */
326 if (session->priv->proxy == NULL) {
327 g_warning ("no mate-session");
328 goto out;
329 }
330
331 /* find out if this change altered the inhibited state */
332 ret = dbus_g_proxy_call (session->priv->proxy, "RegisterClient", &error,
333 G_TYPE_STRING, app_id,
334 G_TYPE_STRING, client_startup_id,
335 G_TYPE_INVALID,
336 DBUS_TYPE_G_OBJECT_PATH, &client_id,
337 G_TYPE_INVALID);
338 if (!ret) {
339 g_warning ("failed to register client '%s': %s", client_startup_id, error->message);
340 g_error_free (error);
341 goto out;
342 }
343
344 /* get org.gnome.SessionManager.ClientPrivate interface */
345 connection = dbus_g_bus_get (DBUS_BUS_SESSION, NULL);
346 session->priv->proxy_client_private = dbus_g_proxy_new_for_name_owner (connection, GPM_SESSION_MANAGER_SERVICE,
347 client_id, GPM_SESSION_MANAGER_CLIENT_PRIVATE_INTERFACE, &error);
348 if (session->priv->proxy_client_private == NULL) {
349 g_warning ("DBUS error: %s", error->message);
350 g_error_free (error);
351 goto out;
352 }
353
354 /* get Stop */
355 dbus_g_proxy_add_signal (session->priv->proxy_client_private, "Stop", G_TYPE_INVALID);
356 dbus_g_proxy_connect_signal (session->priv->proxy_client_private, "Stop", G_CALLBACK (gpm_session_stop_cb), session, NULL);
357
358 /* get QueryEndSession */
359 dbus_g_proxy_add_signal (session->priv->proxy_client_private, "QueryEndSession", G_TYPE_UINT, G_TYPE_INVALID);
360 dbus_g_proxy_connect_signal (session->priv->proxy_client_private, "QueryEndSession", G_CALLBACK (gpm_session_query_end_session_cb), session, NULL);
361
362 /* get EndSession */
363 dbus_g_proxy_add_signal (session->priv->proxy_client_private, "EndSession", G_TYPE_UINT, G_TYPE_INVALID);
364 dbus_g_proxy_connect_signal (session->priv->proxy_client_private, "EndSession", G_CALLBACK (gpm_session_end_session_cb), session, NULL);
365
366 g_debug ("registered startup '%s' to client id '%s'", client_startup_id, client_id);
367out:
368 g_free (client_id);
369 return ret;
370}

375static void
376gpm_session_inhibit_changed_cb (DBusGProxy *proxy, const gchar *id, GpmSession *session)
377{
378 gboolean is_idle_inhibited;
379 gboolean is_suspend_inhibited;
380
381 is_idle_inhibited = gpm_session_is_idle_inhibited (session);
382 is_suspend_inhibited = gpm_session_is_suspend_inhibited (session);
383 if (is_idle_inhibited != session->priv->is_idle_inhibited_old || is_suspend_inhibited != session->priv->is_suspend_inhibited_old) {
384 g_debug ("emitting inhibited-changed : idle=(%i), suspend=(%i)", is_idle_inhibited, is_suspend_inhibited);
385 session->priv->is_idle_inhibited_old = is_idle_inhibited;
386 session->priv->is_suspend_inhibited_old = is_suspend_inhibited;
387 g_signal_emit (session, signals [INHIBITED_CHANGED], 0, is_idle_inhibited, is_suspend_inhibited);
388 }
389}

449static void
450gpm_session_init (GpmSession *session)
451{
452 DBusGConnection *connection;
453 GError *error = NULL;
454
455 session->priv = gpm_session_get_instance_private (session);
456 session->priv->is_idle_old = FALSE;
457 session->priv->is_idle_inhibited_old = FALSE;
458 session->priv->is_suspend_inhibited_old = FALSE;
459 session->priv->proxy_client_private = NULL;
460
461 connection = dbus_g_bus_get (DBUS_BUS_SESSION, NULL);
462
463 /* get org.gnome.SessionManager interface */
464 session->priv->proxy = dbus_g_proxy_new_for_name_owner (connection, GPM_SESSION_MANAGER_SERVICE,
465 GPM_SESSION_MANAGER_PATH,
466 GPM_SESSION_MANAGER_INTERFACE, &error);
467 if (session->priv->proxy == NULL) {
468 g_warning ("DBUS error: %s", error->message);
469 g_error_free (error);
470 return;
471 }
472
473 /* get org.gnome.SessionManager.Presence interface */
474 session->priv->proxy_presence = dbus_g_proxy_new_for_name_owner (connection, GPM_SESSION_MANAGER_SERVICE,
475 GPM_SESSION_MANAGER_PRESENCE_PATH,
476 GPM_SESSION_MANAGER_PRESENCE_INTERFACE, &error);
477 if (session->priv->proxy_presence == NULL) {
478 g_warning ("DBUS error: %s", error->message);
479 g_error_free (error);
480 return;
481 }
482
483 /* get properties interface */
484 session->priv->proxy_prop = dbus_g_proxy_new_for_name_owner (connection, GPM_SESSION_MANAGER_SERVICE,
485 GPM_SESSION_MANAGER_PRESENCE_PATH,
486 GPM_DBUS_PROPERTIES_INTERFACE, &error);
487 if (session->priv->proxy_prop == NULL) {
488 g_warning ("DBUS error: %s", error->message);
489 g_error_free (error);
490 return;
491 }
492
493 /* get StatusChanged */
494 dbus_g_proxy_add_signal (session->priv->proxy_presence, "StatusChanged", G_TYPE_UINT, G_TYPE_INVALID);
495 dbus_g_proxy_connect_signal (session->priv->proxy_presence, "StatusChanged", G_CALLBACK (gpm_session_presence_status_changed_cb), session, NULL);
496
497 /* get InhibitorAdded */
498 dbus_g_proxy_add_signal (session->priv->proxy, "InhibitorAdded", DBUS_TYPE_G_OBJECT_PATH, G_TYPE_INVALID);
499 dbus_g_proxy_connect_signal (session->priv->proxy, "InhibitorAdded", G_CALLBACK (gpm_session_inhibit_changed_cb), session, NULL);
500
501 /* get InhibitorRemoved */
502 dbus_g_proxy_add_signal (session->priv->proxy, "InhibitorRemoved", DBUS_TYPE_G_OBJECT_PATH, G_TYPE_INVALID);
503 dbus_g_proxy_connect_signal (session->priv->proxy, "InhibitorRemoved", G_CALLBACK (gpm_session_inhibit_changed_cb), session, NULL);
504
505 /* coldplug */
506 session->priv->is_idle_inhibited_old = gpm_session_is_idle_inhibited (session);
507 session->priv->is_suspend_inhibited_old = gpm_session_is_suspend_inhibited (session);
508 session->priv->is_idle_old = gpm_session_is_idle (session);
509 g_debug ("idle: %i, idle_inhibited: %i, suspend_inhibited: %i", session->priv->is_idle_old, session->priv->is_idle_inhibited_old, session->priv->is_suspend_inhibited_old);
510}

516static void
517gpm_session_finalize (GObject *object)
518{
519 GpmSession *session;
520 g_return_if_fail (object != NULL);
521 g_return_if_fail (GPM_IS_SESSION (object));
522
523 session = GPM_SESSION (object);
524 session->priv = gpm_session_get_instance_private (session);
525
526 g_object_unref (session->priv->proxy);
527 g_object_unref (session->priv->proxy_presence);
528 if (session->priv->proxy_client_private != NULL)
529 g_object_unref (session->priv->proxy_client_private);
530 g_object_unref (session->priv->proxy_prop);
531
532 G_OBJECT_CLASS (gpm_session_parent_class)->finalize (object);
533}
