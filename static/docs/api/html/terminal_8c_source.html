52static char *
53ay_to_string (GVariant *variant,
54 GError **error)
55{
56 gsize len;
57 const char *data;
58
59 data = g_variant_get_fixed_array (variant, &len, sizeof (char));
60 if (len == 0)
61 return NULL;
62
63 /* Make sure there are no embedded NULs */
64 if (memchr (data, '\0', len) != NULL)
65 {
66 g_set_error_literal (error, G_DBUS_ERROR, G_DBUS_ERROR_INVALID_ARGS,
67 "String is shorter than claimed");
68 return NULL;
69 }
70
71 return g_strndup (data, len);
72}

74static char **
75ay_to_strv (GVariant *variant,
76 int *argc)
77{
78 GPtrArray *argv;
79 const char *data, *nullbyte;
80 gsize data_len;
81 gssize len;
82
83 data = g_variant_get_fixed_array (variant, &data_len, sizeof (char));
84 if (data_len == 0 || data_len > G_MAXSSIZE)
85 {
86 if (argc != NULL)
87 *argc = 0;
88 return NULL;
89 }
90
91 argv = g_ptr_array_new ();
92
93 len = data_len;
94 do
95 {
96 gssize string_len;
97
98 nullbyte = memchr (data, '\0', len);
99
100 string_len = nullbyte ? (gssize) (nullbyte - data) : len;
101 g_ptr_array_add (argv, g_strndup (data, string_len));
102
103 len -= string_len + 1;
104 data += string_len + 1;
105 }
106 while (len > 0);
107
108 if (argc)
109 *argc = argv->len;
110
111 /* NULL terminate */
112 g_ptr_array_add (argv, NULL);
113 return (char **) g_ptr_array_free (argv, FALSE);
114}

137static void
138method_call_cb (GDBusConnection *connection,
139 const char *sender,
140 const char *object_path,
141 const char *interface_name,
142 const char *method_name,
143 GVariant *parameters,
144 GDBusMethodInvocation *invocation,
145 gpointer user_data)
146{
147 if (g_strcmp0 (method_name, "HandleArguments") == 0)
148 {
149 TerminalOptions *options = NULL;
150 GVariant *v_wd, *v_display, *v_sid, *v_envv, *v_argv;
151 char *working_directory = NULL, *display_name = NULL, *startup_id = NULL;
152 int initial_workspace = -1;
153 char **envv = NULL, **argv = NULL;
154 int argc;
155 GError *error = NULL;
156
157 g_variant_get (parameters, "(@ay@ay@ay@ayi@ay)",
158 &v_wd, &v_display, &v_sid, &v_envv, &initial_workspace, &v_argv);
159
160 working_directory = ay_to_string (v_wd, &error);
161 if (error)
162 goto out;
163 display_name = ay_to_string (v_display, &error);
164 if (error)
165 goto out;
166 startup_id = ay_to_string (v_sid, &error);
167 if (error)
168 goto out;
169 envv = ay_to_strv (v_envv, NULL);
170 argv = ay_to_strv (v_argv, &argc);
171
172 _terminal_debug_print (TERMINAL_DEBUG_FACTORY,
173 "Factory invoked with working-dir='%s' display='%s' startup-id='%s'"
174 "workspace='%d'\n",
175 working_directory ? working_directory : "(null)",
176 display_name ? display_name : "(null)",
177 startup_id ? startup_id : "(null)",
178 initial_workspace);
179
180 options = terminal_options_parse (working_directory,
181 display_name,
182 startup_id,
183 envv,
184 TRUE,
185 TRUE,
186 &argc, &argv,
187 &error,
188 NULL);
189
190 if (options != NULL)
191 {
192 options->initial_workspace = initial_workspace;
193
194 terminal_app_handle_options (terminal_app_get (), options, FALSE /* no resume */, &error);
195 terminal_options_free (options);
196 }
197
198out:
199 g_variant_unref (v_wd);
200 g_free (working_directory);
201 g_variant_unref (v_display);
202 g_free (display_name);
203 g_variant_unref (v_sid);
204 g_free (startup_id);
205 g_variant_unref (v_envv);
206 g_strfreev (envv);
207 g_variant_unref (v_argv);
208 g_strfreev (argv);
209
210 if (error == NULL)
211 {
212 g_dbus_method_invocation_return_value (invocation, g_variant_new ("()"));
213 }
214 else
215 {
216 g_dbus_method_invocation_return_gerror (invocation, error);
217 g_error_free (error);
218 }
219 }

222static void
223bus_acquired_cb (GDBusConnection *connection,
224 const char *name,
225 gpointer user_data)
226{
227 static const char dbus_introspection_xml[] =
228 "<node name='/org/mate/Terminal'>"
229 "<interface name='org.mate.Terminal.Factory'>"
230 "<method name='HandleArguments'>"
231 "<arg type='ay' name='working_directory' direction='in' />"
232 "<arg type='ay' name='display_name' direction='in' />"
233 "<arg type='ay' name='startup_id' direction='in' />"
234 "<arg type='ay' name='environment' direction='in' />"
235 "<arg type='i' name='workspace' direction='in' />"
236 "<arg type='ay' name='arguments' direction='in' />"
237 "</method>"
238 "</interface>"
239 "</node>";
240
241 static const GDBusInterfaceVTable interface_vtable =
242 {
243 method_call_cb,
244 NULL,
245 NULL,
246 { 0 }
247 };
248
249 OwnData *data = (OwnData *) user_data;
250 GDBusNodeInfo *introspection_data;
251 guint registration_id;
252 GError *error = NULL;
253
254 _terminal_debug_print (TERMINAL_DEBUG_FACTORY,
255 "Bus %s acquired\n", name);
256
257 introspection_data = g_dbus_node_info_new_for_xml (dbus_introspection_xml, NULL);
258 g_assert (introspection_data != NULL);
259
260 registration_id = g_dbus_connection_register_object (connection,
261 TERMINAL_FACTORY_SERVICE_PATH,
262 introspection_data->interfaces[0],
263 &interface_vtable,
264 NULL, NULL,
265 &error);
266 g_dbus_node_info_unref (introspection_data);
267
268 if (registration_id == 0)
269 {
270 g_printerr ("Failed to register object: %s\n", error->message);
271 g_error_free (error);
272 data->exit_code = EXIT_FAILURE;
273 gtk_main_quit ();
274 }

306static void
307name_lost_cb (GDBusConnection *connection,
308 const char *name,
309 gpointer user_data)
310{
311 OwnData *data = (OwnData *) user_data;
312 GError *error = NULL;
313 char **envv;
314 int i;
315 GVariantBuilder builder;
316 GVariant *value;
317 GString *string;
318 char *s;
319 gsize len;
320
321 _terminal_debug_print (TERMINAL_DEBUG_FACTORY,
322 "Lost the name %s on the session bus\n", name);
323
324 /* Couldn't get the connection? No way to continue! */
325 if (connection == NULL)
326 {
327 data->exit_code = EXIT_FAILURE;
328 gtk_main_quit ();
329 return;
330 }
331
332 if (data->options == NULL)
333 {
334 /* Already handled */
335 data->exit_code = EXIT_SUCCESS;
336 gtk_main_quit ();
337 return;
338 }
339
340 _terminal_debug_print (TERMINAL_DEBUG_FACTORY,
341 "Forwarding arguments to existing instance\n");
342
343 g_variant_builder_init (&builder, G_VARIANT_TYPE ("(ayayayayiay)"));
344
345 g_variant_builder_add (&builder, "@ay", string_to_ay (data->options->default_working_dir));
346 g_variant_builder_add (&builder, "@ay", string_to_ay (data->options->display_name));
347 g_variant_builder_add (&builder, "@ay", string_to_ay (data->options->startup_id));
348
349 string = g_string_new (NULL);
350 envv = g_get_environ ();
351 for (i = 0; envv[i]; ++i)
352 {
353 if (i > 0)
354 g_string_append_c (string, '\0');
355
356 g_string_append (string, envv[i]);
357 }
358 g_strfreev (envv);
359
360 len = string->len;
361 s = g_string_free (string, FALSE);
362 g_variant_builder_add (&builder, "@ay",
363 g_variant_new_from_data (G_VARIANT_TYPE ("ay"), s, len, TRUE, g_free, s));
364
365 g_variant_builder_add (&builder, "@i", g_variant_new_int32 (data->options->initial_workspace));
366
367 string = g_string_new (NULL);
368
369 for (i = 0; i < data->argc; ++i)
370 {
371 if (i > 0)
372 g_string_append_c (string, '\0');
373 g_string_append (string, data->argv[i]);
374 }
375
376 len = string->len;
377 s = g_string_free (string, FALSE);
378 g_variant_builder_add (&builder, "@ay",
379 g_variant_new_from_data (G_VARIANT_TYPE ("ay"), s, len, TRUE, g_free, s));
380
381 value = g_dbus_connection_call_sync (connection,
382 data->factory_name,
383 TERMINAL_FACTORY_SERVICE_PATH,
384 TERMINAL_FACTORY_INTERFACE_NAME,
385 "HandleArguments",
386 g_variant_builder_end (&builder),
387 G_VARIANT_TYPE ("()"),
388 G_DBUS_CALL_FLAGS_NONE,
389 -1,
390 NULL,
391 &error);
392 if (value == NULL)
393 {
394 g_printerr ("Failed to forward arguments: %s\n", error->message);
395 g_error_free (error);
396 data->exit_code = EXIT_FAILURE;
397 gtk_main_quit ();
398 }
399 else
400 {
401 g_variant_unref (value);
402 data->exit_code = EXIT_SUCCESS;
403 }
404
405 terminal_options_free (data->options);
406 data->options = NULL;
407
408 gtk_main_quit ();
409}

411static char *
412get_factory_name_for_display (const char *display_name)
413{
414 GString *name;
415 const char *p;
416
417 name = g_string_sized_new (strlen (TERMINAL_FACTORY_SERVICE_NAME_PREFIX) + strlen (display_name) + 1 /* NUL */);
418 g_string_append (name, TERMINAL_FACTORY_SERVICE_NAME_PREFIX);
419
420 for (p = display_name; *p; ++p)
421 {
422 if (g_ascii_isalnum (*p))
423 g_string_append_c (name, *p);
424 else
425 g_string_append_c (name, '_');
426 }
427
428 _terminal_debug_print (TERMINAL_DEBUG_FACTORY,
429 "Factory name is \"%s\"\n", name->str);
430
431 return g_string_free (name, FALSE);
432}

434static int
435get_initial_workspace (void)
436{
437 int ret = -1;
438 GdkWindow *window;
439 guchar *data = NULL;
440 GdkAtom atom;
441 GdkAtom cardinal_atom;
442
443 window = gdk_get_default_root_window();
444
445 atom = gdk_atom_intern_static_string ("_NET_CURRENT_DESKTOP");
446 cardinal_atom = gdk_atom_intern_static_string ("CARDINAL");
447
448 if (gdk_property_get (window, atom, cardinal_atom, 0, 8, FALSE, NULL, NULL, NULL, &data)) {
449 ret = *(int *)data;

456main (int argc, char **argv)
457{
458 int i;
459 char **argv_copy;
460 int argc_copy;
461 const char *startup_id, *home_dir;
462 TerminalOptions *options;
463 GError *error = NULL;
464 char *working_directory;
465 int ret = EXIT_SUCCESS;
466
467 setlocale (LC_ALL, "");
468
469 bindtextdomain (GETTEXT_PACKAGE, TERM_LOCALEDIR);
470 bind_textdomain_codeset (GETTEXT_PACKAGE, "UTF-8");
471 textdomain (GETTEXT_PACKAGE);
472
473 _terminal_debug_init ();
474
475 /* Make a NULL-terminated copy since we may need it later */
476 argv_copy = g_new (char *, argc + 1);
477 for (i = 0; i < argc; ++i)
478 argv_copy [i] = argv [i];
479 argv_copy [i] = NULL;
480 argc_copy = argc;
481
482 startup_id = g_getenv ("DESKTOP_STARTUP_ID");
483
484 working_directory = g_get_current_dir ();
485
486 /* Now change directory to $HOME so we don't prevent unmounting, e.g. if the
487 * factory is started by caja-open-terminal. See bug #565328.
488 * On failure back to /.
489 */
490 home_dir = g_get_home_dir ();
491 if (home_dir == NULL || chdir (home_dir) < 0)
492 if (chdir ("/") < 0)
493 g_warning ("Could not change working directory.");
494
495 options = terminal_options_parse (working_directory,
496 NULL,
497 startup_id,
498 NULL,
499 FALSE,
500 FALSE,
501 &argc, &argv,
502 &error,
503#ifdef HAVE_SMCLIENT
504 gtk_get_option_group (TRUE),
505 egg_sm_client_get_option_group (),
506#endif /* HAVE_SMCLIENT */
507 NULL);
508 g_free (working_directory);
509
510 if (options == NULL)
511 {
512 g_printerr (_("Failed to parse arguments: %s\n"), error->message);
513 g_error_free (error);
514 exit (EXIT_FAILURE);
515 }
516
517 g_set_application_name (_("Terminal"));
518
519 /* Unset the these env variables, so they doesn't end up
520 * in the factory's env and thus in the terminals' envs.
521 */
522 g_unsetenv ("DESKTOP_STARTUP_ID");
523 g_unsetenv ("GIO_LAUNCHED_DESKTOP_FILE_PID");
524 g_unsetenv ("GIO_LAUNCHED_DESKTOP_FILE");
525
526 if (options->startup_id == NULL)
527 {
528 options->startup_id = g_strdup_printf ("_TIME%" G_GINT64_FORMAT,
529 g_get_monotonic_time () / 1000);
530 }
531
532 gdk_init (&argc, &argv);
533 const char *display_name = gdk_display_get_name (gdk_display_get_default ());
534 options->display_name = g_strdup (display_name);
535
536 if (options->use_factory)
537 {
538 OwnData *data;
539 guint owner_id;
540
541 data = g_new (OwnData, 1);
542 data->factory_name = get_factory_name_for_display (display_name);
543 data->options = options;
544 data->exit_code = -1;
545 data->argv = argv_copy;
546 data->argc = argc_copy;
547
548 gtk_init(&argc, &argv);
549 options->initial_workspace = get_initial_workspace ();
550
551 owner_id = g_bus_own_name (G_BUS_TYPE_SESSION,
552 data->factory_name,
553 G_BUS_NAME_OWNER_FLAGS_NONE,
554 bus_acquired_cb,
555 name_acquired_cb,
556 name_lost_cb,
557 data, NULL);
558
559 gtk_main ();
560
561 ret = data->exit_code;
562 g_bus_unown_name (owner_id);
563
564 g_free (data->factory_name);
565 g_free (data);
566
567 }
568 else
569 {
570 gtk_init(&argc, &argv);
571 terminal_app_handle_options (terminal_app_get (), options, TRUE /* allow resume */, &error);
572 terminal_options_free (options);
573
574 if (error == NULL)
575 {
576 gtk_main ();
577 }
578 else
579 {
580 g_printerr ("Error handling options: %s\n", error->message);
581 g_error_free (error);
582 ret = EXIT_FAILURE;
583 }
584 }
585
586 terminal_app_shutdown ();
587
588 g_free (argv_copy);
589
590 return ret;
