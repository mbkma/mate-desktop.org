313static MateUIInfo *copy_ui_info_tree (const MateUIInfo source[])
314{
315 MateUIInfo *copy;
316 int i, count;
317
318 for(count = 0; source[count].type != MATE_APP_UI_ENDOFINFO; count++)
319 ;
320
321 count++;
322
323 copy = g_malloc(count*sizeof(MateUIInfo));
324
325 memcpy(copy, source, count*sizeof(MateUIInfo));
326
327 for(i = 0; i < count; i++) {
328 if( (source[i].type == MATE_APP_UI_SUBTREE) ||
329 (source[i].type == MATE_APP_UI_SUBTREE_STOCK) ||
330 (source[i].type == MATE_APP_UI_RADIOITEMS) )
331 copy[i].moreinfo = copy_ui_info_tree(source[i].moreinfo);
332 }
333
334 return copy;
335}

338static gint count_ui_info_items (const MateUIInfo *ui_info)
339{
340 gint num;
341 gint count=0;
342
343 for(num = 0; ui_info[num].type != MATE_APP_UI_ENDOFINFO; num++)
344 if (ui_info[num].type != MATE_APP_UI_HELP &&
345 ui_info[num].type != MATE_APP_UI_BUILDER_DATA)
346 count++;
347
348 return count;
349}

351static void free_ui_info_tree (MateUIInfo *root)
352{
353 int count;
354
355 for(count = 0; root[count].type != MATE_APP_UI_ENDOFINFO; count++)
356 if( (root[count].type == MATE_APP_UI_SUBTREE) ||
357 (root[count].type == MATE_APP_UI_SUBTREE_STOCK) ||
358 (root[count].type == MATE_APP_UI_RADIOITEMS) )
359 free_ui_info_tree(root[count].moreinfo);
360
361 g_free(root);
362}

433 child = g_list_next(child);
434 }
435
436 gtk_widget_queue_resize(submenu);

542static gint book_button_release (GtkWidget *widget, GdkEventButton *e, gpointer data)
543{
544 gint x = e->x_root, y = e->y_root;
545 MateMDI *mdi;
546
547 mdi = MATE_MDI(data);
548
549 if(mdi->signal_id) {
550 g_signal_handler_disconnect(widget, mdi->signal_id);
551 mdi->signal_id = 0;
552 }
553
554 if(e->button == 1 && e->window == widget->window && mdi->in_drag) {
555 GdkWindow *window;
556 GList *child;
557 MateApp *app;
558 GtkWidget *view, *new_book;
559 GtkNotebook *old_book = GTK_NOTEBOOK(widget);
560
561 mdi->in_drag = FALSE;
562 gdk_pointer_ungrab(GDK_CURRENT_TIME);
563 gtk_grab_remove(widget);
564
565 window = gdk_window_at_pointer(&x, &y);
566 if(window)
567 window = gdk_window_get_toplevel(window);
568
569 child = mdi->windows;
570 while(child) {
571 if(window == GTK_WIDGET(child->data)->window) {
572 int cur_page;
573
574 /* page was dragged to another notebook */
575
576 old_book = GTK_NOTEBOOK(widget);
577 new_book = MATE_APP(child->data)->contents;
578
579 if(old_book == (GtkNotebook *)new_book) /* page has been dropped on the source notebook */
580 return FALSE;
581
582 cur_page = gtk_notebook_get_current_page (GTK_NOTEBOOK (old_book));
583
584 if(cur_page >= 0) {
585 view = gtk_notebook_get_nth_page (GTK_NOTEBOOK (old_book), cur_page);
586 gtk_container_remove(GTK_CONTAINER(old_book), view);
587
588 book_add_view(GTK_NOTEBOOK(new_book), view);
589
590 app = mate_mdi_get_app_from_view(view);
591 gdk_window_raise(GTK_WIDGET(app)->window);
592
593 cur_page = gtk_notebook_get_current_page (GTK_NOTEBOOK (old_book));
594
595 if(cur_page < 0) {
596 mdi->active_window = app;
597 app = MATE_APP(gtk_widget_get_toplevel(GTK_WIDGET(old_book)));
598 mdi->windows = g_list_remove(mdi->windows, app);
599 gtk_widget_destroy(GTK_WIDGET(app));
600 }
601 }
602
603 return FALSE;
604 }
605
606 child = child->next;
607 }
608
609 if(g_list_length(old_book->children) == 1)
610 return FALSE;
611
612 /* create a new toplevel */
613 if(old_book->cur_page) {
614 gint width, height;
615 int cur_page = gtk_notebook_get_current_page (old_book);
616
617 view = gtk_notebook_get_nth_page (old_book, cur_page);
618
619 app = mate_mdi_get_app_from_view(view);
620
621 width = view->allocation.width;
622 height = view->allocation.height;
623
624 gtk_container_remove(GTK_CONTAINER(old_book), view);
625
626 app_clone(mdi, app);
627
628 new_book = book_create(mdi);
629
630 book_add_view(GTK_NOTEBOOK(new_book), view);
631
632 gtk_window_set_position(GTK_WINDOW(mdi->active_window), GTK_WIN_POS_MOUSE);
633
634 gtk_widget_set_size_request (view, width, height);
635
636 if(!GTK_WIDGET_VISIBLE(mdi->active_window))
637 gtk_widget_show(GTK_WIDGET(mdi->active_window));
638 }
639 }
640
641 return FALSE;
642}

685static void book_switch_page(GtkNotebook *book, GtkNotebookPage *pg, gint page_num, MateMDI *mdi)
686{
687 MateApp *app;
688 GtkWidget *page;
689
690#ifdef MATE_ENABLE_DEBUG
691 g_message("MateMDI: switching pages");
692#endif
693 app = MATE_APP(gtk_widget_get_toplevel(GTK_WIDGET(book)));
694
695 page = gtk_notebook_get_nth_page (book, page_num);
696
697 if(page != NULL) {
698 if(page != mdi->active_view)

774static gint app_close_book (MateApp *app, GdkEventAny *event, MateMDI *mdi)
775{
776 MateMDIChild *child;
777 GtkWidget *view;
778 gint handler_ret = TRUE;
779
780 if(g_list_length(mdi->windows) == 1) {
781 if(!mate_mdi_remove_all(mdi, FALSE))
782 return TRUE;
783
784 mdi->windows = g_list_remove(mdi->windows, app);
785 gtk_widget_destroy(GTK_WIDGET(app));
786
787 /* only destroy mdi if there are no non-MDI windows registered
788 with it. */
789 if(mdi->registered == NULL)
790 gtk_object_destroy(GTK_OBJECT(mdi));
791 }
792 else {
793 GList *children = gtk_container_get_children (GTK_CONTAINER (app->contents));
794 GList *li;
795
796 /* first check if all the children in this notebook can be removed */
797 if (children == NULL) {
798 mdi->windows = g_list_remove(mdi->windows, app);
799 gtk_widget_destroy(GTK_WIDGET(app));
800 return FALSE;
801 }
802
803 for (li = children; li != NULL; li = li->next) {
804 GList *node;
805 view = li->data;
806
807 child = mate_mdi_get_child_from_view(view);
808
809 node = child->views;
810 while(node) {
811 if(mate_mdi_get_app_from_view(node->data) != app)
812 break;
813
814 node = node->next;
815 }
816
817 if(node == NULL) { /* all the views reside in this MateApp */
818 g_signal_emit(mdi, mdi_signals[REMOVE_CHILD], 0,
819 child,
820 &handler_ret);
821 if(handler_ret == FALSE) {
822 g_list_free (children);
823 return TRUE;
824 }
825 }
826 }
827
828 /* now actually remove all children/views! */
829 for (li = children; li != NULL; li = li->next) {
830 view = li->data;
831
832 child = mate_mdi_get_child_from_view(view);
833
834 /* if this is the last view, remove the child */
835 if(g_list_length(child->views) == 1)
836 mate_mdi_remove_child(mdi, child, TRUE);
837 else
838 mate_mdi_remove_view(mdi, view, TRUE);
839 }
840
841 g_list_free (children);
842 }
843
844 return FALSE;
845}

847static void app_set_view (MateMDI *mdi, MateApp *app, GtkWidget *view)
848{
849 GList *menu_list = NULL, *children;
850 GtkWidget *parent = NULL;
851 MateMDIChild *child;
852 MateUIInfo *ui_info;
853 gint pos, items;
854
855 /* free previous child ui-info */
856 ui_info = g_object_get_data(G_OBJECT(app), MATE_MDI_CHILD_MENU_INFO_KEY);
857 if(ui_info != NULL) {
858 free_ui_info_tree(ui_info);
859 g_object_set_data(G_OBJECT(app), MATE_MDI_CHILD_MENU_INFO_KEY, NULL);
860 }
861 ui_info = NULL;
862
863 if(mdi->child_menu_path)
864 parent = mate_app_find_menu_pos(app->menubar, mdi->child_menu_path, &pos);
865
866 /* remove old child-specific menus */
867 items = GPOINTER_TO_INT(g_object_get_data(G_OBJECT(app), MATE_MDI_ITEM_COUNT_KEY));
868 if(items > 0 && parent) {
869 GtkWidget *widget;
870
871 /* remove items; should be kept in sync with mate_app_remove_menus! */
872 children = g_list_nth(GTK_MENU_SHELL(parent)->children, pos);
873 while(children && items > 0) {
874 widget = GTK_WIDGET(children->data);
875 children = children->next;
876
877 /* if this item contains a gtkaccellabel, we have to set its
878 accel_widget to NULL so that the item gets unrefed. */
879 if(GTK_IS_ACCEL_LABEL(GTK_BIN(widget)->child))
880 gtk_accel_label_set_accel_widget(GTK_ACCEL_LABEL(GTK_BIN(widget)->child), NULL);
881
882 gtk_container_remove(GTK_CONTAINER(parent), widget);
883 items--;
884 }
885 }
886
887 items = 0;
888 if(view) {
889 child = mate_mdi_get_child_from_view(view);
890
891 /* set the title */
892 if( (mdi->mode == MATE_MDI_MODAL) || (mdi->mode == MATE_MDI_TOPLEVEL) ) {
893 /* in MODAL and TOPLEVEL modes the window title includes the active
894 child name: "child_name - mdi_title" */
895 gchar *fullname;
896
897 fullname = g_strconcat(child->name, " - ", mdi->title, NULL);
898 gtk_window_set_title(GTK_WINDOW(app), fullname);
899 g_free(fullname);
900 }
901 else
902 gtk_window_set_title(GTK_WINDOW(app), mdi->title);
903
904 /* create new child-specific menus */
905 if(parent) {
906 if( child->menu_template &&
907 ( (ui_info = copy_ui_info_tree(child->menu_template)) != NULL) ) {
908 mate_app_insert_menus_with_data(app, mdi->child_menu_path, ui_info, child);
909 g_object_set_data(G_OBJECT(app), MATE_MDI_CHILD_MENU_INFO_KEY, ui_info);
910 items = count_ui_info_items(ui_info);
911 }
912 else {
913 menu_list = child_create_menus(child, view);
914
915 if(menu_list) {
916 GList *menu;
917
918 items = 0;
919 menu = menu_list;
920 while(menu) {
921 gtk_menu_shell_insert(GTK_MENU_SHELL(parent), GTK_WIDGET(menu->data), pos);
922 menu = menu->next;
923 pos++;
924 items++;
925 }
926
927 g_list_free(menu_list);
928 }
929 else
930 items = 0;
931 }
932 }
933 }
934 else
935 gtk_window_set_title(GTK_WINDOW(app), mdi->title);
936
937 g_object_set_data(G_OBJECT(app), MATE_MDI_ITEM_COUNT_KEY, GINT_TO_POINTER(items));
938
939 if(parent)
940 gtk_widget_queue_resize(parent);
941
942 set_active_view(mdi, view);
943}
