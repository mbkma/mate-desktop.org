39static void G_GNUC_UNUSED
40add_sized (GtkIconFactory *factory,
41 const guchar *inline_data,
42 GtkIconSize size,
43 const gchar *stock_id)
44{
45 GtkIconSet *set;
46 GtkIconSource *source;
47 GdkPixbuf *pixbuf;
48
49 pixbuf = gdk_pixbuf_new_from_inline (-1, inline_data, FALSE, NULL);
50
51 source = gtk_icon_source_new ();
52 gtk_icon_source_set_pixbuf (source, pixbuf);
53 gtk_icon_source_set_size (source, size);
54
55 set = gtk_icon_set_new ();
56 gtk_icon_set_add_source (set, source);
57
58 gtk_icon_factory_add (factory, stock_id, set);
59
60 g_object_unref (G_OBJECT (pixbuf));
61 gtk_icon_source_free (source);
62 gtk_icon_set_unref (set);
63}
