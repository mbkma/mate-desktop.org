64/* Check if any window with _NET_WM_WINDOW_TYPE_DESKTOP exists */
65static gboolean
66desktop_window_exists (void)
67{
68 Display *display = gdk_x11_get_default_xdisplay ();
69 Window window = gdk_x11_get_default_root_xwindow ();
70 Atom client_list, desktop_type, window_type, type;
71 int format;
72 unsigned long nitems, after;
73 unsigned char *data;
74 GdkDisplay *gdk_display;
75 gboolean running = FALSE;
76
77 /* This atom will be present if a window owns the desktop (e.g. caja) */
78 desktop_type = XInternAtom (display, "_NET_WM_WINDOW_TYPE_DESKTOP", True);
79 if (desktop_type == None)
80 return FALSE;
81
82 /* This atom will contain the list of all windows managed by the window manager */
83 client_list = XInternAtom (display, "_NET_CLIENT_LIST", True);
84 if (client_list == None)
85 return FALSE;
86
87 XGetWindowProperty (display, window, client_list, 0, 1024, False,
88 XA_WINDOW, &type, &format, &nitems, &after, &data);
89
90 if (data == NULL)
91 return FALSE;
92
93 if (type != XA_WINDOW || format != 32) {
94 XFree (data);
95 return FALSE;
96 }
97
98 window_type = XInternAtom (display, "_NET_WM_WINDOW_TYPE", True);
99 if (window_type == None) {
100 XFree (data);
101 return FALSE;
102 }
103
104 gdk_display = gdk_display_get_default ();
105 Window *windows = (Window *) data;
106
107 /* Read all the windows and determine if any of them own the desktop */
108 for (unsigned long i = 0; i < nitems && !running; i++) {
109 Atom wm_type;
110 int wm_format;
111 unsigned long wm_nitems, wm_after;
112 unsigned char *wm_data;
113
114 gdk_x11_display_error_trap_push (gdk_display);
115
116 XGetWindowProperty (display, windows[i], window_type, 0, 32, False,
117 XA_ATOM, &wm_type, &wm_format, &wm_nitems, &wm_after, &wm_data);
118
119 if (wm_data != NULL) {
120 Atom *types = (Atom *) wm_data;
121 for (unsigned long j = 0; j < wm_nitems; j++) {
122 /* Found a window managing the desktop. Bail out. */
123 if (types[j] == desktop_type) {
124 running = TRUE;
125 break;
126 }
127 }
128 XFree (wm_data);
129 }
130
131 gdk_x11_display_error_trap_pop_ignored (gdk_display);
132 }
133 XFree (data);
134
135 return running;
136}

147static void
148draw_background (MsdBackgroundManager *manager)
149{
150 if (manager->draw_in_progress || desktop_window_exists ())
151 return;
152
153 GdkDisplay *display = gdk_display_get_default ();
154 GdkScreen *screen = gdk_display_get_default_screen (display);
155 GdkWindow *window = gdk_screen_get_root_window (screen);
156 gint scale = gdk_window_get_scale_factor (window);
157 gint width = WidthOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
158 gint height = HeightOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
159
160 manager->draw_in_progress = TRUE;
161
162 free_bg_surface (manager);
163 manager->surface = mate_bg_create_surface_scale (manager->bg, window, width, height, scale, TRUE);
164 mate_bg_set_surface_as_root (screen, manager->surface);
165
166 manager->draw_in_progress = FALSE;
167}

237gboolean
238msd_background_manager_start (MsdBackgroundManager *manager,
239 GError **error)
240{
241 g_debug ("Starting background manager");
242 mate_settings_profile_start (NULL);
243
244 manager->settings = g_settings_new (MATE_BG_SCHEMA);
245
246 manager->bg = mate_bg_new();
247
248 g_signal_connect(manager->bg, "changed", G_CALLBACK (on_bg_changed), manager);
249
250 g_signal_connect(manager->bg, "transitioned", G_CALLBACK (on_bg_transitioned), manager);
251
252 mate_bg_load_from_gsettings (manager->bg, manager->settings);
253
254 connect_screen_signals (manager);
255
256 g_signal_connect (manager->settings, "change-event",
257 G_CALLBACK (settings_change_event_cb), manager);
258
259 mate_settings_profile_end (NULL);
260 return TRUE;
261}

263void
264msd_background_manager_stop (MsdBackgroundManager *manager)
265{
266 g_debug ("Stopping background manager");
267
268 disconnect_screen_signals (manager);
269
270 if (manager->settings != NULL) {
271 g_signal_handlers_disconnect_by_func (manager->settings, settings_change_event_cb, manager);
272 g_object_unref (manager->settings);
273 manager->settings = NULL;
274 }
275
276 if (manager->bg != NULL) {
277 g_object_unref (manager->bg);
278 manager->bg = NULL;
279 }
280
281 free_bg_surface (manager);
282}
