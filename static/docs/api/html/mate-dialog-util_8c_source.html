155static void
156dialog_reply_callback (MateMessageBox * mbox, gint button, callback_info* data)
157{
158 MateReplyCallback func = (MateReplyCallback) data->function;
159
160 if (button == 0)
161 button = 1;
162 else
163 button = 0;
164
165 (* func)(button, data->data);
166}

168static GtkWidget *
169reply_dialog (const gchar * question, MateReplyCallback callback, gpointer data,
170 gboolean yes_or_ok, gboolean modal, GtkWindow * parent)
171{
172 GtkWidget * mbox;
173 callback_info * info;
174
175 if (yes_or_ok) {
176 mbox = mate_message_box_new(question, MATE_MESSAGE_BOX_QUESTION,
177 MATE_STOCK_BUTTON_NO,
178 MATE_STOCK_BUTTON_YES, NULL);
179 }
180 else {
181 mbox = mate_message_box_new(question, MATE_MESSAGE_BOX_QUESTION,
182 MATE_STOCK_BUTTON_CANCEL,
183 MATE_STOCK_BUTTON_OK, NULL);
184 }
185
186 if (callback != NULL) {
187 info = g_new(callback_info, 1);
188
189 info->function = callback;
190 info->data = data;
191
192 g_signal_connect_data (mbox, "clicked",
193 G_CALLBACK (dialog_reply_callback),
194 info,
195 (GClosureNotify) g_free,
196 0);
197 }
198
199 if (modal) {
200 gtk_window_set_modal(GTK_WINDOW(mbox),TRUE);
201 }
202
203 if (parent != NULL) {
204 mate_dialog_set_parent(MATE_DIALOG(mbox),parent);
205 }
206
207 gtk_widget_show(mbox);
208 return mbox;
209}

396static void
397dialog_string_callback (MateMessageBox * mbox, gint button, callback_info * data)
398{
399 gchar * s = NULL;
400 const char * tmp;
401 MateStringCallback func = (MateStringCallback)data->function;
402
403 if (button == 1) {
404 tmp = gtk_entry_get_text (data->entry);
405 if (tmp) s = g_strdup(tmp);
406 }
407
408 (* func)(s, data->data);
409}

411static GtkWidget *
412request_dialog (const gchar * request, const gchar * default_text, const guint16 max_length,
413 MateStringCallback callback, gpointer data,
414 gboolean password,
415 GtkWindow * parent)
416{
417 GtkWidget * mbox;
418 callback_info * info;
419 GtkWidget * entry;
420
421 mbox = mate_message_box_new ( request, MATE_MESSAGE_BOX_QUESTION,
422 MATE_STOCK_BUTTON_CANCEL,
423 MATE_STOCK_BUTTON_OK,
424 NULL );
425 mate_dialog_set_default ( MATE_DIALOG(mbox), 1 );
426
427 /* set up text entry widget */
428 entry = gtk_entry_new();
429 if (password) gtk_entry_set_visibility (GTK_ENTRY(entry), FALSE);
430 if ((default_text != NULL) && (*default_text))
431 gtk_entry_set_text(GTK_ENTRY(entry), default_text);
432 if (max_length > 0)
433 gtk_entry_set_max_length(GTK_ENTRY(entry), max_length);
434
435 gtk_box_pack_end ( GTK_BOX(MATE_DIALOG(mbox)->vbox),
436 entry, FALSE, FALSE, MATE_PAD_SMALL );
437
438 /* If Return is pressed in the text entry, propagate to the buttons */
439 mate_dialog_editable_enters(MATE_DIALOG(mbox), GTK_EDITABLE(entry));
440
441 info = g_new(callback_info, 1);
442
443 info->function = callback;
444 info->data = data;
445 info->entry = GTK_ENTRY(entry);
446
447 g_signal_connect_data (mbox, "clicked",
448 G_CALLBACK (dialog_string_callback),
449 info,
450 (GClosureNotify) g_free,
451 0);
452
453 if (parent != NULL) {
454 mate_dialog_set_parent(MATE_DIALOG(mbox),parent);
455 }
456
457 gtk_widget_grab_focus (entry);
458
459 gtk_widget_show (entry);
460 gtk_widget_show (mbox);
461 return mbox;
462}
