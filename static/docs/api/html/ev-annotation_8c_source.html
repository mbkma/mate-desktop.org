139static void
140ev_annotation_finalize (GObject *object)
141{
142 EvAnnotation *annot = EV_ANNOTATION (object);
143
144 if (annot->page) {
145 g_object_unref (annot->page);
146 annot->page = NULL;
147 }
148
149 if (annot->contents) {
150 g_free (annot->contents);
151 annot->contents = NULL;
152 }
153
154 if (annot->name) {
155 g_free (annot->name);
156 annot->name = NULL;
157 }
158
159 if (annot->modified) {
160 g_free (annot->modified);
161 annot->modified = NULL;
162 }
163
164 G_OBJECT_CLASS (ev_annotation_parent_class)->finalize (object);
165}

173static void
174ev_annotation_set_property (GObject *object,
175 guint prop_id,

178{
179 EvAnnotation *annot = EV_ANNOTATION (object);
180

185 case PROP_ANNOT_CONTENTS:

187 break;
188 case PROP_ANNOT_NAME:
189 ev_annotation_set_name (annot, g_value_get_string (value));
190 break;
191 case PROP_ANNOT_MODIFIED:
192 ev_annotation_set_modified (annot, g_value_get_string (value));
193 break;
194 case PROP_ANNOT_COLOR:
195 ev_annotation_set_color (annot, g_value_get_pointer (value));
196 break;
197 case PROP_ANNOT_RGBA:
198 ev_annotation_set_rgba (annot, g_value_get_boxed (value));
199 break;
200 default:
201 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
202 }

548void
549ev_annotation_get_color (EvAnnotation *annot,
550 GdkColor *color)
551{
552 GdkRGBA rgba;
553
554 g_return_if_fail (EV_IS_ANNOTATION (annot));
555 g_return_if_fail (color != NULL);
556
557 ev_annotation_get_rgba (annot, &rgba);
558
559 color->pixel = 0;
560 color->red = CLAMP (rgba.red * 65535. + 0.5, 0, 65535);
561 color->green = CLAMP (rgba.green * 65535. + 0.5, 0, 65535);
562 color->blue = CLAMP (rgba.blue * 65535. + 0.5, 0, 65535);
563}
564
578gboolean
579ev_annotation_set_color (EvAnnotation *annot,
580 const GdkColor *color)
581{
582 GdkColor annot_color;
583 GdkRGBA rgba;
584
585 g_return_val_if_fail (EV_IS_ANNOTATION (annot), FALSE);
586
587 ev_annotation_get_color (annot, &annot_color);
588 if (color == NULL ||
589 ((color->red == annot_color.red) &&
590 (color->green == annot_color.green) &&
591 (color->blue == annot_color.blue)))
592 return FALSE;
593
594 rgba.red = color->red / 65535.;
595 rgba.green = color->green / 65535.;
596 rgba.blue = color->blue / 65535.;
597 rgba.alpha = 1.;
598
599 ev_annotation_set_rgba (annot, &rgba);
600
601 return TRUE;
602}

712static EvAnnotationMarkupProps *
713ev_annotation_markup_get_properties (EvAnnotationMarkup *markup)
714{
715 EvAnnotationMarkupProps *props;
716 static GQuark props_key = 0;
717
718 if (!props_key)
719 props_key = g_quark_from_static_string ("ev-annotation-markup-props");
720
721 props = g_object_get_qdata (G_OBJECT (markup), props_key);
722 if (!props) {
723 props = g_slice_new0 (EvAnnotationMarkupProps);
724 g_object_set_qdata_full (G_OBJECT (markup),
725 props_key, props,
726 (GDestroyNotify) ev_annotation_markup_props_free);
727 }
728
729 return props;
730}
