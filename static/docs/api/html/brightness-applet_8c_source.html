129static gboolean
130gpm_applet_get_brightness (GpmBrightnessApplet *applet)
131{
132 GError *error = NULL;
133 gboolean ret;
134 guint policy_brightness;
135
136 if (applet->proxy == NULL) {
137 g_warning ("not connected\n");
138 return FALSE;
139 }
140
141 ret = dbus_g_proxy_call (applet->proxy, "GetBrightness", &error,
142 G_TYPE_INVALID,
143 G_TYPE_UINT, &policy_brightness,
144 G_TYPE_INVALID);
145 if (error) {
146 g_debug ("ERROR: %s\n", error->message);
147 g_error_free (error);
148 }
149 if (ret) {
150 applet->level = policy_brightness;
151 } else {
152 /* abort as the DBUS method failed */
153 g_warning ("GetBrightness failed!\n");
154 }
155
156 return ret;
157}

163static gboolean
164gpm_applet_set_brightness (GpmBrightnessApplet *applet)
165{
166 GError *error = NULL;
167 gboolean ret;
168
169 if (applet->proxy == NULL) {
170 g_warning ("not connected");
171 return FALSE;
172 }
173
174 ret = dbus_g_proxy_call (applet->proxy, "SetBrightness", &error,
175 G_TYPE_UINT, applet->level,
176 G_TYPE_INVALID,
177 G_TYPE_INVALID);
178 if (error) {
179 g_debug ("ERROR: %s", error->message);
180 g_error_free (error);
181 }
182 if (!ret) {
183 /* abort as the DBUS method failed */
184 g_warning ("SetBrightness failed!");
185 }
186
187 return ret;
188}

196static void
197gpm_applet_get_icon (GpmBrightnessApplet *applet)
198{
199 const gchar *icon;
200
201 /* free */
202 if (applet->icon != NULL) {
203 g_object_unref (applet->icon);
204 applet->icon = NULL;
205 }
206
207 if (applet->size <= 2) {
208 return;
209 }
210
211 /* get icon */
212 if (applet->proxy == NULL) {
213 icon = GPM_BRIGHTNESS_APPLET_ICON_INVALID;
214 } else if (applet->call_worked == FALSE) {
215 icon = GPM_BRIGHTNESS_APPLET_ICON_DISABLED;
216 } else {
217 icon = GPM_BRIGHTNESS_APPLET_ICON;
218 }
219
220 applet->icon = gtk_icon_theme_load_icon (gtk_icon_theme_get_default (),
221 icon, applet->size - 2, 0, NULL);
222
223 if (applet->icon == NULL) {
224 g_warning ("Cannot find %s!\n", icon);
225 } else {
226 g_debug ("got icon %s!\n", icon);
227 /* update size cache */
228 applet->icon_height = gdk_pixbuf_get_height (applet->icon);
229 applet->icon_width = gdk_pixbuf_get_width (applet->icon);
230 }
231}

239static void
240gpm_applet_check_size (GpmBrightnessApplet *applet)
241{
242 GtkAllocation allocation;
243
244 /* we don't use the size function here, but the yet allocated size because the
245 size value is false (kind of rounded) */
246 gtk_widget_get_allocation (GTK_WIDGET (applet), &allocation);
247 if (MATE_PANEL_APPLET_VERTICAL(mate_panel_applet_get_orient (MATE_PANEL_APPLET (applet)))) {
248 if (applet->size != allocation.width) {
249 applet->size = allocation.width;
250 gpm_applet_get_icon (applet);
251 gtk_widget_set_size_request (GTK_WIDGET(applet), applet->size, applet->icon_height + 2);
252 }
253 } else {
254 if (applet->size != allocation.height) {
255 applet->size = allocation.height;
256 gpm_applet_get_icon (applet);
257 gtk_widget_set_size_request (GTK_WIDGET(applet), applet->icon_width + 2, applet->size);
258 }
259 }
260}

268static gboolean
269gpm_applet_draw_cb (GpmBrightnessApplet *applet)
270{
271 gint w, h, bg_type;
272 GdkRGBA color;
273 cairo_t *cr;
274 cairo_pattern_t *pattern;
275 GtkStyleContext *context;
276 GtkAllocation allocation;
277
278 if (gtk_widget_get_window (GTK_WIDGET(applet)) == NULL) {
279 return FALSE;
280 }
281
282 /* retrieve applet size */
283 gpm_applet_get_icon (applet);
284 gpm_applet_check_size (applet);
285 if (applet->size <= 2) {
286 return FALSE;
287 }
288
289 /* if no icon, then don't try to display */
290 if (applet->icon == NULL) {
291 return FALSE;
292 }
293
294 gtk_widget_get_allocation (GTK_WIDGET (applet), &allocation);
295 w = allocation.width;
296 h = allocation.height;
297
298 cr = gdk_cairo_create (gtk_widget_get_window (GTK_WIDGET(applet)));
299
300 /* draw pixmap background */
301 bg_type = mate_panel_applet_get_background (MATE_PANEL_APPLET (applet), &color, &pattern);
302 if (bg_type == PANEL_PIXMAP_BACKGROUND && !applet->popped) {
303 /* fill with given background pixmap */
304 cairo_set_source (cr, pattern);
305 cairo_rectangle (cr, 0, 0, w, h);
306 cairo_fill (cr);
307 }
308
309 /* draw color background */
310 if (bg_type == PANEL_COLOR_BACKGROUND && !applet->popped) {
311 gdk_cairo_set_source_rgba (cr, &color);
312 cairo_rectangle (cr, 0, 0, w, h);
313 cairo_fill (cr);
314 }
315
316 /* fill with selected color if popped */
317 if (applet->popped) {
318 context = gtk_widget_get_style_context (GTK_WIDGET(applet));
319 gtk_style_context_get_color (context, GTK_STATE_FLAG_SELECTED, &color);
320 gdk_cairo_set_source_rgba (cr, &color);
321 cairo_rectangle (cr, 0, 0, w, h);
322 cairo_fill (cr);
323 }
324
325 /* draw icon at center */
326 gdk_cairo_set_source_pixbuf (cr, applet->icon, (w - applet->icon_width)/2, (h - applet->icon_height)/2);
327 cairo_paint (cr);
328
329 cairo_destroy (cr);
330
331 return TRUE;
332}

371static void
372gpm_applet_update_tooltip (GpmBrightnessApplet *applet)
373{
374 gchar *buf = NULL;
375 if (applet->popped == FALSE) {
376 if (applet->proxy == NULL) {
377 buf = g_strdup (_("Cannot connect to mate-power-manager"));
378 } else if (applet->call_worked == FALSE) {
379 buf = g_strdup (_("Cannot get laptop panel brightness"));
380 } else {
381 buf = g_strdup_printf (_("LCD brightness : %u%%"), applet->level);
382 }
383 gtk_widget_set_tooltip_text (GTK_WIDGET(applet), buf);
384 } else {
385 gtk_widget_set_tooltip_text (GTK_WIDGET(applet), NULL);
386 }
387 g_free (buf);

422static void
423gpm_applet_adjust_brightness (GpmBrightnessApplet *applet, guint step, gboolean increase)
424{
425 if (increase) {
426 applet->level = (applet->level <= 100 - step) ? applet->level + step : 100;
427 } else {
428 applet->level = (applet->level >= step) ? applet->level - step : 0;
429 }
430 applet->call_worked = gpm_applet_set_brightness (applet);
431 gpm_applet_update_popup_level (applet);
432}

507static gboolean
508gpm_applet_key_press_cb (GtkWidget *popup, GdkEventKey *event, GpmBrightnessApplet *applet)
509{
510 switch (event->keyval) {
511 case GDK_KEY_KP_Enter:
512 case GDK_KEY_ISO_Enter:
513 case GDK_KEY_3270_Enter:
514 case GDK_KEY_Return:
515 case GDK_KEY_space:
516 case GDK_KEY_KP_Space:
517 case GDK_KEY_Escape:
518 /* if yet popped, hide */
519 if (applet->popped) {
520 gtk_widget_hide (applet->popup);
521 applet->popped = FALSE;
522 gpm_applet_update_tooltip (applet);
523 return TRUE;
524 } else {
525 return FALSE;
526 }
527 break;
528 case GDK_KEY_Page_Up:
529 gpm_applet_adjust_brightness (applet, GPM_BRIGHTNESS_APPLET_PERCENT_CHANGE_PAGE, TRUE);
530 return TRUE;
531 break;
532 case GDK_KEY_Left:
533 case GDK_KEY_Up:
534 gpm_applet_adjust_brightness (applet, GPM_BRIGHTNESS_APPLET_PERCENT_CHANGE_ARROW, TRUE);
535 return TRUE;
536 break;
537 case GDK_KEY_Page_Down:
538 gpm_applet_adjust_brightness (applet, GPM_BRIGHTNESS_APPLET_PERCENT_CHANGE_PAGE, FALSE);
539 return TRUE;
540 break;
541 case GDK_KEY_Right:
542 case GDK_KEY_Down:
543 gpm_applet_adjust_brightness (applet, GPM_BRIGHTNESS_APPLET_PERCENT_CHANGE_ARROW, FALSE);
544 return TRUE;
545 break;
546 default:
547 return FALSE;
548 break;
549 }
550
551 return FALSE;
552}

613static void
614gpm_applet_create_popup (GpmBrightnessApplet *applet)
615{
616 static GtkWidget *box, *frame;
617 GtkWidget *toplevel;
618 GtkStyleContext *context;
619 GdkScreen *screen;
620 GdkVisual *visual;
621 gint orientation = mate_panel_applet_get_orient (MATE_PANEL_APPLET (MATE_PANEL_APPLET (applet)));
622
623 gpm_applet_destroy_popup_cb (applet);
624
625 /* slider */
626 if (MATE_PANEL_APPLET_VERTICAL(orientation)) {
627 applet->slider = gtk_scale_new_with_range (GTK_ORIENTATION_HORIZONTAL, 0, 100, 1);
628 gtk_widget_set_size_request (applet->slider, 100, -1);
629 } else {
630 applet->slider = gtk_scale_new_with_range (GTK_ORIENTATION_VERTICAL, 0, 100, 1);
631 gtk_widget_set_size_request (applet->slider, -1, 100);
632 }
633 gtk_range_set_inverted (GTK_RANGE(applet->slider), TRUE);
634 gtk_scale_set_draw_value (GTK_SCALE(applet->slider), FALSE);
635 gtk_range_set_value (GTK_RANGE(applet->slider), applet->level);
636 g_signal_connect (G_OBJECT(applet->slider), "value-changed", G_CALLBACK(gpm_applet_slide_cb), applet);
637
638 /* minus button */
639 applet->btn_minus = gtk_button_new_with_label ("\342\210\222"); /* U+2212 MINUS SIGN */
640 gtk_button_set_relief (GTK_BUTTON(applet->btn_minus), GTK_RELIEF_NONE);
641 gtk_widget_set_can_focus (applet->btn_minus, FALSE);
642 g_signal_connect (G_OBJECT(applet->btn_minus), "pressed", G_CALLBACK(gpm_applet_minus_cb), applet);
643
644 /* plus button */
645 applet->btn_plus = gtk_button_new_with_label ("+");
646 gtk_button_set_relief (GTK_BUTTON(applet->btn_plus), GTK_RELIEF_NONE);
647 gtk_widget_set_can_focus (applet->btn_plus, FALSE);
648 g_signal_connect (G_OBJECT(applet->btn_plus), "pressed", G_CALLBACK(gpm_applet_plus_cb), applet);
649
650 /* box */
651 if (MATE_PANEL_APPLET_VERTICAL(orientation)) {
652 box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 1);
653 } else {
654 box = gtk_box_new (GTK_ORIENTATION_VERTICAL, 1);
655 }
656 gtk_box_pack_start (GTK_BOX(box), applet->btn_plus, FALSE, FALSE, 0);
657 gtk_box_pack_start (GTK_BOX(box), applet->slider, TRUE, TRUE, 0);
658 gtk_box_pack_start (GTK_BOX(box), applet->btn_minus, FALSE, FALSE, 0);
659
660 /* frame */
661 frame = gtk_frame_new (NULL);
662 gtk_frame_set_shadow_type (GTK_FRAME(frame), GTK_SHADOW_OUT);
663 gtk_container_add (GTK_CONTAINER(frame), box);
664
665 /* window */
666 applet->popup = gtk_window_new (GTK_WINDOW_POPUP);
667 gtk_window_set_type_hint (GTK_WINDOW(applet->popup), GDK_WINDOW_TYPE_HINT_UTILITY);
668 gtk_container_add (GTK_CONTAINER(applet->popup), frame);
669
670 /* window events */
671 g_signal_connect (G_OBJECT(applet->popup), "button-press-event",
672 G_CALLBACK (on_popup_button_press), applet);
673
674 g_signal_connect (G_OBJECT(applet->popup), "key-press-event",
675 G_CALLBACK(gpm_applet_key_press_cb), applet);
676
677 /* Set volume control frame, slider and toplevel window to follow panel volume control theme */
678 toplevel = gtk_widget_get_toplevel (frame);
679 context = gtk_widget_get_style_context (GTK_WIDGET(toplevel));
680 gtk_style_context_add_class(context,"mate-panel-applet-slider");
681 /*Make transparency possible in gtk3 theme3 */
682 screen = gtk_widget_get_screen(GTK_WIDGET(toplevel));
683 visual = gdk_screen_get_rgba_visual(screen);
684 gtk_widget_set_visual(GTK_WIDGET(toplevel), visual);
685}

693static gboolean
694gpm_applet_popup_cb (GpmBrightnessApplet *applet, GdkEventButton *event)
695{
696 GtkAllocation allocation, popup_allocation;
697 gint orientation, x, y;
698 GdkWindow *window;
699 GdkDisplay *display;
700 GdkSeat *seat;
701
702 /* react only to left mouse button */
703 if (event->button != 1) {
704 return FALSE;
705 }
706
707 /* if yet popped, release focus and hide */
708 if (applet->popped) {
709 gtk_widget_hide (applet->popup);
710 applet->popped = FALSE;
711 gpm_applet_update_tooltip (applet);
712 return TRUE;
713 }
714
715 /* otherwise pop */
716 applet->popped = TRUE;
717
718 /* create a new popup (initial or if panel parameters changed) */
719 if (applet->popup == NULL) {
720 gpm_applet_create_popup (applet);
721 }
722
723 /* update UI for current brightness */
724 gpm_applet_update_popup_level (applet);
725
726 gtk_widget_show_all (applet->popup);

728 /* retrieve geometry parameters and move window appropriately */
729 orientation = mate_panel_applet_get_orient (MATE_PANEL_APPLET (MATE_PANEL_APPLET (applet)));
730 gdk_window_get_origin (gtk_widget_get_window (GTK_WIDGET(applet)), &x, &y);
731

736 x += allocation.x + allocation.width/2;

738 x -= popup_allocation.width/2;
739 break;
740 case MATE_PANEL_APPLET_ORIENT_UP:
741 x += allocation.x + allocation.width/2;
742 y += allocation.y;
743 x -= popup_allocation.width/2;
744 y -= popup_allocation.height;
745 break;
746 case MATE_PANEL_APPLET_ORIENT_RIGHT:
747 y += allocation.y + allocation.height/2;
748 x += allocation.x + allocation.width;
749 y -= popup_allocation.height/2;
750 break;
751 case MATE_PANEL_APPLET_ORIENT_LEFT:
752 y += allocation.y + allocation.height/2;
753 x += allocation.x;
754 x -= popup_allocation.width;
755 y -= popup_allocation.height/2;
756 break;
757 default:
758 g_assert_not_reached ();
759 }
760
761 gtk_window_move (GTK_WINDOW (applet->popup), x, y);
762
763 /* grab input */
764 window = gtk_widget_get_window (GTK_WIDGET (applet->popup));
765 display = gdk_window_get_display (window);
766 seat = gdk_display_get_default_seat (display);
767 gdk_seat_grab (seat,
768 window,
769 GDK_SEAT_CAPABILITY_ALL,
770 TRUE,
771 NULL,
772 NULL,
773 NULL,
774 NULL);

914static gboolean
915gpm_brightness_applet_dbus_connect (GpmBrightnessApplet *applet)
916{
917 GError *error = NULL;
918
919 if (applet->connection == NULL) {
920 g_debug ("get connection\n");
921 g_clear_error (&error);
922 applet->connection = dbus_g_bus_get (DBUS_BUS_SESSION, &error);
923 if (error != NULL) {
924 g_warning ("Could not connect to DBUS daemon: %s", error->message);
925 g_error_free (error);
926 applet->connection = NULL;
927 return FALSE;
928 }
929 }
930 if (applet->proxy == NULL) {
931 g_debug ("get proxy\n");
932 g_clear_error (&error);
933 applet->proxy = dbus_g_proxy_new_for_name_owner (applet->connection,
934 GPM_DBUS_SERVICE,
935 GPM_DBUS_PATH_BACKLIGHT,
936 GPM_DBUS_INTERFACE_BACKLIGHT,
937 &error);
938 if (error != NULL) {
939 g_warning ("Cannot connect, maybe the daemon is not running: %s\n", error->message);
940 g_error_free (error);
941 applet->proxy = NULL;
942 return FALSE;
943 }
944 dbus_g_proxy_add_signal (applet->proxy, "BrightnessChanged",
945 G_TYPE_UINT, G_TYPE_INVALID);
946 dbus_g_proxy_connect_signal (applet->proxy, "BrightnessChanged",
947 G_CALLBACK (brightness_changed_cb),
948 applet, NULL);
949 /* reset, we might be starting race */
950 applet->call_worked = gpm_applet_get_brightness (applet);
951 }
952 return TRUE;
953}
