49/* MateMDIChild
50 * is an abstract class. In order to use it, you have to either derive a
51 * new class from it and set the proper virtual functions in its parent
52 * MateMDIChildClass structure or use the MateMDIGenericChild class
53 * that allows to specify the relevant functions on a per-instance basis
54 * and can directly be used with MateMDI.
55 */
56struct _MateMDIChild
57{
58 GtkObject object;
59
60 GtkObject *parent;
61
62 gchar *name;
63
64 GList *views;
65
66 MateUIInfo *menu_template;
67
68 gpointer reserved;
69};
