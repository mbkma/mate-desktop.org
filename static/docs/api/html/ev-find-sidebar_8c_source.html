141static gboolean
142sidebar_tree_button_press_cb (GtkTreeView *view,
143 GdkEventButton *event,
144 EvFindSidebar *sidebar)
145{
146 EvFindSidebarPrivate *priv = GET_PRIVATE (sidebar);
147 GtkTreeModel *model;
148 GtkTreePath *path;
149 GtkTreeIter iter;
150
151 gtk_tree_view_get_path_at_pos (view, event->x, event->y, &path,
152 NULL, NULL, NULL);
153 if (!path)
154 return FALSE;
155
156 if (priv->highlighted_result &&
157 gtk_tree_path_compare (priv->highlighted_result, path) != 0) {
158 gtk_tree_path_free (path);
159 return FALSE;
160 }
161
162 model = gtk_tree_view_get_model (view);
163 gtk_tree_model_get_iter (model, &iter, path);
164 gtk_tree_path_free (path);
165
166 ev_find_sidebar_activate_result_at_iter (sidebar, model, &iter);
167
168 /* Always return FALSE so the tree view gets the event and can update
169 * the selection etc.
170 */
171 return FALSE;
172}

282static gchar *
283sanitized_substring (const gchar *text,
284 gint start,
285 gint end)
286{
287 const gchar *p;
288 const gchar *start_ptr;
289 const gchar *end_ptr;
290 guint len = 0;
291 gchar *retval;
292
293 if (end - start <= 0)
294 return NULL;
295
296 start_ptr = g_utf8_offset_to_pointer (text, start);
297 end_ptr = g_utf8_offset_to_pointer (start_ptr, end - start);
298
299 retval = g_malloc (end_ptr - start_ptr + 1);
300 p = start_ptr;
301
302 while (p != end_ptr) {
303 const gchar *next;
304
305 next = g_utf8_next_char (p);
306
307 if (next != end_ptr) {
308 GUnicodeBreakType break_type;
309
310 break_type = g_unichar_break_type (g_utf8_get_char (p));
311 if (break_type == G_UNICODE_BREAK_HYPHEN && *next == '\n') {
312 p = g_utf8_next_char (next);
313 continue;
314 }
315 }
316
317 if (*p != '\n') {
318 strncpy (retval + len, p, next - p);
319 len += next - p;
320 } else {
321 *(retval + len) = ' ';
322 len++;
323 }
324
325 p = next;
326 }
327
328 if (len == 0) {
329 g_free (retval);
330
331 return NULL;
332 }
333
334 retval[len] = 0;
335
336 return retval;
337}

339static gchar *
340get_surrounding_text_markup (const gchar *text,
341 const gchar *find_text,
342 gboolean case_sensitive,
343 PangoLogAttr *log_attrs,
344 gint log_attrs_length,
345 gint offset)
346{
347 gint iter;
348 gchar *prec = NULL;
349 gchar *succ = NULL;
350 gchar *match = NULL;
351 gchar *markup;
352 gint max_chars;
353
354 iter = MAX (0, offset - 1);
355 while (!log_attrs[iter].is_word_start && iter > 0)
356 iter--;
357
358 prec = sanitized_substring (text, iter, offset);
359
360 iter = offset;
361 offset += g_utf8_strlen (find_text, -1);
362 if (!case_sensitive)
363 match = g_utf8_substring (text, iter, offset);
364
365 iter = MIN (log_attrs_length, offset + 1);
366 max_chars = MIN (log_attrs_length - 1, iter + 100);
367 while (TRUE) {
368 gint word = iter;
369
370 while (!log_attrs[word].is_word_end && word < max_chars)
371 word++;
372
373 if (word > max_chars)
374 break;
375
376 iter = word + 1;
377 }
378
379 succ = sanitized_substring (text, offset, iter);
380
381 markup = g_markup_printf_escaped ("%s<span weight=\"bold\">%s</span>%s",
382 prec ? prec : "", match ? match : find_text, succ ? succ : "");
383 g_free (prec);
384 g_free (succ);
385 g_free (match);
386
387 return markup;
388}

412static gint
413get_match_offset (EvRectangle *areas,
414 guint n_areas,
415 EvRectangle *match,
416 gint offset)
417{
418 gdouble x, y;
419 gint i;
420
421 x = match->x1;
422 y = (match->y1 + match->y2) / 2;
423
424 i = offset;
425
426 do {
427 EvRectangle *area = areas + i;
428
429 if (x >= area->x1 && x < area->x2 &&
430 y >= area->y1 && y <= area->y2) {
431 return i;
432 }
433
434 i = (i + 1) % n_areas;
435 } while (i != offset);
436
437 return -1;
438}

440static gboolean
441process_matches_idle (EvFindSidebar *sidebar)
442{
443 EvFindSidebarPrivate *priv = GET_PRIVATE (sidebar);
444 GtkTreeModel *model;
445 gint current_page;
446 EvDocument *document;
447
448 priv->process_matches_idle_id = 0;
449
450 if (!ev_job_find_has_results (priv->job)) {
451 if (ev_job_is_finished (EV_JOB (priv->job)))
452 g_clear_object (&priv->job);
453 return FALSE;
454 }
455
456 document = EV_JOB (priv->job)->document;
457 model = gtk_tree_view_get_model (GTK_TREE_VIEW (priv->tree_view));
458
459 do {
460 GList *matches, *l;
461 EvPage *page;
462 gint result;
463 gchar *page_label;
464 gchar *page_text;
465 EvRectangle *areas = NULL;
466 guint n_areas;
467 PangoLogAttr *text_log_attrs;
468 gulong text_log_attrs_length;
469 gint offset;
470
471 current_page = priv->current_page;
472 priv->current_page = (priv->current_page + 1) % priv->job->n_pages;
473
474 matches = priv->job->pages[current_page];
475 if (!matches)
476 continue;
477
478 page = ev_document_get_page (document, current_page);
479 page_label = ev_document_get_page_label (document, current_page);
480 page_text = get_page_text (document, page, &areas, &n_areas);
481 g_object_unref (page);
482 if (!page_text)
483 continue;
484
485 text_log_attrs_length = g_utf8_strlen (page_text, -1);
486 text_log_attrs = g_new0 (PangoLogAttr, text_log_attrs_length + 1);
487 pango_get_log_attrs (page_text, -1, -1, NULL, text_log_attrs, text_log_attrs_length + 1);
488
489 if (priv->first_match_page == -1)
490 priv->first_match_page = current_page;
491
492 offset = 0;
493
494 for (l = matches, result = 0; l; l = g_list_next (l), result++) {
495 EvRectangle *match = (EvRectangle *)l->data;
496 gchar *markup;
497 GtkTreeIter iter;
498 gint new_offset;
499
500 new_offset = get_match_offset (areas, n_areas, match, offset);
501 if (new_offset == -1) {
502 g_warning ("No offset found for match \"%s\" at page %d after processing %d results\n",
503 priv->job->text, current_page, result);
504 /* It may happen that a vertical text match has no corresponding text area, skip
505 * that but keep iterating to show any other matches in page (issue #1545) */
506 continue;
507 }
508 offset = new_offset;
509
510 if (current_page >= priv->job->start_page) {
511 gtk_list_store_append (GTK_LIST_STORE (model), &iter);
512 } else {
513 gtk_list_store_insert (GTK_LIST_STORE (model), &iter,
514 priv->insert_position);
515 priv->insert_position++;
516 }
517
518 markup = get_surrounding_text_markup (page_text,
519 priv->job->text,
520 priv->job->case_sensitive,
521 text_log_attrs,
522 text_log_attrs_length,
523 offset);
524
525 gtk_list_store_set (GTK_LIST_STORE (model), &iter,
526 TEXT_COLUMN, markup,
527 PAGE_LABEL_COLUMN, page_label,
528 PAGE_COLUMN, current_page + 1,
529 RESULT_COLUMN, result,
530 -1);
531 g_free (markup);
532 }
533
534 g_free (page_label);
535 g_free (page_text);
536 g_free (text_log_attrs);
537 g_free (areas);
538 } while (current_page != priv->job_current_page);
539
540 if (ev_job_is_finished (EV_JOB (priv->job)) && priv->current_page == priv->job->start_page)
541 ev_find_sidebar_highlight_first_match_of_page (sidebar, priv->first_match_page);
542
543 return FALSE;
544}

586void
587ev_find_sidebar_restart (EvFindSidebar *sidebar,
588 gint page)
589{
590 EvFindSidebarPrivate *priv = GET_PRIVATE (sidebar);
591 gint first_match_page = -1;
592 gint i;
593
594 if (!priv->job)
595 return;
596
597 for (i = 0; i < priv->job->n_pages; i++) {
598 int index;
599
600 index = page + i;
601
602 if (index >= priv->job->n_pages)
603 index -= priv->job->n_pages;
604
605 if (priv->job->pages[index]) {
606 first_match_page = index;
607 break;
608 }
609 }
610
611 if (first_match_page != -1)
612 ev_find_sidebar_highlight_first_match_of_page (sidebar, first_match_page);
613}
