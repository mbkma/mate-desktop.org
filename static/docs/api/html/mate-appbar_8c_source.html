46struct _MateAppBarPrivate
47{
48 /* Private; there's no guarantee on the type of these in the
49 future. Statusbar could be a label, entry, GtkStatusbar, or
50 something else; progress could be a label or progress bar; it's
51 all up in the air for now. */
52 /* there is no reason for these not to be properties */
53 GtkWidget * progress;
54 GtkWidget * status;
55 gchar * prompt; /* The text of a prompt, if any. */
56
57 /* Keep it simple; no contexts.
58 if (status_stack) display_top_of_stack;
59 else if (default_status) display_default;
60 else display_nothing; */
61 /* Also private by the way */
62 GSList * status_stack;
63 gchar * default_status;
64
65 gint16 editable_start; /* The first editable position in the interactive
66 buffer. */
67
68 /* these are construct only currently. there is no reason this
69 * couldn't be changed. */
70
71 MatePreferencesType interactivity;
72 guint interactive : 1; /* This means status is an entry rather than a
73 label, for the moment. */
74 guint has_progress : 1;
75 guint has_status : 1;
76};

96static GSList *
97stringstack_push(GSList * stringstack, const gchar * s)
98{
99 return g_slist_prepend(stringstack, g_strdup(s));
100}

102static GSList *
103stringstack_pop(GSList * stringstack)
104{
105 /* Could be optimized */
106 if (stringstack) {
107 g_free(stringstack->data);
108 return g_slist_remove(stringstack, stringstack->data);
109 }
110 else return NULL;
111}

113static const gchar *
114stringstack_top(GSList * stringstack)
115{
116 if (stringstack) return stringstack->data;
117 else return NULL;
118}

120static void
121stringstack_free(GSList * stringstack)
122{
123 GSList * tmp = stringstack;
124 while (tmp) {
125 g_free(tmp->data);
126 tmp = g_slist_next(tmp);
127 }
128 g_slist_free(stringstack);
129}

131static void
132entry_delete_text_cb(GtkWidget * entry, gint start,
133 gint stop, MateAppBar * ab)
134{
135 g_return_if_fail(MATE_IS_APPBAR(ab));
136 g_return_if_fail(GTK_IS_ENTRY(entry));
137 g_return_if_fail(ab->_priv->interactive);
138
139 if (ab->_priv->prompt == NULL) return; /* not prompting, so don't
140 interfere. */
141 if (start < ab->_priv->editable_start) {
142 /* Block the signal, since it's trying to delete text
143 that shouldn't be deleted. */
144 g_signal_stop_emission_by_name(entry, "delete_text");
145 gdk_beep();
146 }
147}

170static gint
171entry_key_press_cb(GtkWidget * entry, GdkEventKey * e, MateAppBar * ab)
172{
173 /* This should be some kind of configurable binding
174 or accelerator, probably. */
175 if ( (e->keyval == GDK_g) &&
176 (e->state & GDK_CONTROL_MASK) ) {

256{

335void
336mate_appbar_refresh (MateAppBar * appbar)
337{
338 g_return_if_fail(appbar != NULL);
339 g_return_if_fail(MATE_IS_APPBAR(appbar));
340
341 if (appbar->_priv->prompt) {
342 g_return_if_fail(appbar->_priv->interactive); /* Just a consistency check */
343 gtk_editable_set_editable(GTK_EDITABLE(appbar->_priv->status), TRUE);
344 /* Allow insert_text to work, so we can set the prompt. */
345 appbar->_priv->editable_start = 0;
346 gtk_entry_set_text(GTK_ENTRY(appbar->_priv->status), appbar->_priv->prompt);
347 /* This has to be after setting the text. */
348 appbar->_priv->editable_start = strlen(appbar->_priv->prompt);
349 gtk_editable_set_position(GTK_EDITABLE(appbar->_priv->status),
350 appbar->_priv->editable_start);
351 gtk_widget_grab_focus(appbar->_priv->status);
352 }
353 else {
354 if (appbar->_priv->interactive) {
355 appbar->_priv->editable_start = 0;
356 gtk_editable_set_editable(GTK_EDITABLE(appbar->_priv->status), FALSE);
357 gtk_grab_remove(appbar->_priv->status); /* In case */
358 }
359
360 if (appbar->_priv->status_stack)
361 mate_appbar_set_status(appbar, stringstack_top(appbar->_priv->status_stack));
362 else if (appbar->_priv->default_status)
363 mate_appbar_set_status(appbar, appbar->_priv->default_status);
364 else
365 mate_appbar_set_status(appbar, "");
366 }
367}

545static void

547{
548 MateAppBar *ab;

571static GObject *
572mate_appbar_constructor (GType type,
573 guint n_properties,
574 GObjectConstructParam *properties)
575{
576 GObject *object;
577 MateAppBar *ab;
578 GtkBox *box;
579 gboolean has_status, has_progress, interactive;
580
581 object = G_OBJECT_CLASS (parent_class)->constructor (type,
582 n_properties,
583 properties);
584
585 ab = MATE_APPBAR (object);
586
587 has_status = ab->_priv->has_status;
588 has_progress = ab->_priv->has_progress;
589 interactive = ab->_priv->interactive;
590
591 box = GTK_BOX (ab);
592
593 box->spacing = MATE_PAD_SMALL;
594 box->homogeneous = FALSE;
595
596 if (has_progress)
597 ab->_priv->progress = gtk_progress_bar_new ();
598
599 /*
600 * If the progress meter goes on the right then we place it after we
601 * create the status line.
602 */
603 if (has_progress &&
604 /* FIXME: this should listen to changes! */
605 ! mate_mateconf_get_bool ("/desktop/mate/interface/statusbar_meter_on_right"))
606 gtk_box_pack_start (box, ab->_priv->progress, FALSE, FALSE, 0);
607
608 if ( has_status ) {
609 if ( interactive ) {
610 ab->_priv->status = gtk_entry_new();
611
612 g_signal_connect (ab->_priv->status, "delete_text",
613 G_CALLBACK(entry_delete_text_cb),
614 ab);
615 g_signal_connect (ab->_priv->status, "insert_text",
616 G_CALLBACK(entry_insert_text_cb),
617 ab);
618 g_signal_connect_after (ab->_priv->status, "key_press_event",
619 G_CALLBACK(entry_key_press_cb),
620 ab);
621 g_signal_connect (ab->_priv->status, "activate",
622 G_CALLBACK(entry_activate_cb),
623 ab);
624
625 /* no prompt now */
626 gtk_editable_set_editable(GTK_EDITABLE(ab->_priv->status), FALSE);
627
628 gtk_box_pack_start (box, ab->_priv->status, TRUE, TRUE, 0);
629 } else {
630 GtkWidget * frame;
631
632 frame = gtk_frame_new (NULL);
633 gtk_frame_set_shadow_type (GTK_FRAME(frame), GTK_SHADOW_IN);
634
635 ab->_priv->status = gtk_label_new (NULL);
636 gtk_misc_set_alignment (GTK_MISC (ab->_priv->status), 0.0, 0.0);
637 gtk_widget_set_size_request (ab->_priv->status, 1, -1);
638
639 gtk_box_pack_start (box, frame, TRUE, TRUE, 0);
640 gtk_container_add (GTK_CONTAINER(frame), ab->_priv->status);
641
642 gtk_widget_show (frame);
643 }
644 }
645
646 if (has_progress &&
647 /* FIXME: this should listen to changes! */
648 mate_mateconf_get_bool ("/desktop/mate/interface/statusbar_meter_on_right"))
649 gtk_box_pack_start (box, ab->_priv->progress, FALSE, FALSE, 0);
650
651 if (ab->_priv->status) gtk_widget_show (ab->_priv->status);
652 if (ab->_priv->progress) gtk_widget_show(ab->_priv->progress);
653
654 return object;
655}

657static void
658mate_appbar_set_property (GObject *object,
659 guint prop_id,
660 const GValue *value,
661 GParamSpec *pspec)
662{
663 MateAppBarPrivate *priv = MATE_APPBAR (object)->_priv;
664 switch (prop_id) {
665 case PROP_HAS_STATUS:
666 priv->has_status = g_value_get_boolean (value);
667 break;
668 case PROP_HAS_PROGRESS:
669 priv->has_progress = g_value_get_boolean (value);
670 break;
671 case PROP_INTERACTIVITY:
672 priv->interactivity = g_value_get_enum (value);
673 switch (priv->interactivity) {
674 case MATE_PREFERENCES_NEVER:
675 priv->interactive = FALSE;
676 break;
677 case MATE_PREFERENCES_ALWAYS:
678 priv->interactive = TRUE;
679 break;
680 default:
681 priv->interactive = mate_mateconf_get_bool ("/desktop/mate/interface/statusbar-interactive");
682 break;
683 }
684 }
685}
