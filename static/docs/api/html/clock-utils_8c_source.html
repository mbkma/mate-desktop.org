41gboolean
42clock_locale_supports_am_pm (void)
43{
44#ifdef HAVE_NL_LANGINFO
45 const char *am;
46
47 am = nl_langinfo (AM_STR);
48 return (am[0] != '\0');

50 return TRUE;

61void
62clock_utils_display_help (GtkWidget *widget,
63 const char *doc_id,
64 const char *link_id)
65{
66 GError *error = NULL;
67 char *uri;
68
69 if (link_id)
70 uri = g_strdup_printf ("help:%s/%s", doc_id, link_id);
71 else
72 uri = g_strdup_printf ("help:%s", doc_id);
73
74 gtk_show_uri_on_window (NULL, uri,
75 gtk_get_current_event_time (), &error);
76 g_free (uri);
77
78 if (error &&
79 g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
80 g_error_free (error);
81 else if (error) {
82 GtkWidget *parent;
83 GtkWidget *dialog;
84 char *primary;
85
86 if (GTK_IS_WINDOW (widget))
87 parent = widget;
88 else
89 parent = NULL;
90
91 primary = g_markup_printf_escaped (
92 _("Could not display help document '%s'"),
93 doc_id);
94 dialog = gtk_message_dialog_new (
95 parent ? GTK_WINDOW (parent) : NULL,
96 GTK_DIALOG_MODAL|GTK_DIALOG_DESTROY_WITH_PARENT,
97 GTK_MESSAGE_ERROR,
98 GTK_BUTTONS_CLOSE,
99 "%s", primary);
100
101 gtk_message_dialog_format_secondary_text (
102 GTK_MESSAGE_DIALOG (dialog),
103 "%s", error->message);
104
105 g_error_free (error);
106 g_free (primary);
107
108 g_signal_connect (dialog, "response",
109 G_CALLBACK (gtk_widget_destroy),
110 NULL);
111
112 gtk_window_set_icon_name (GTK_WINDOW (dialog), CLOCK_ICON);
113 gtk_window_set_screen (GTK_WINDOW (dialog),
114 gtk_widget_get_screen (widget));
115
116 if (parent == NULL) {
117 /* we have no parent window */
118 gtk_window_set_skip_taskbar_hint (GTK_WINDOW (dialog),
119 FALSE);
120 gtk_window_set_title (GTK_WINDOW (dialog),
121 _("Error displaying help document"));
122 }
123
124 gtk_widget_show (dialog);
125 }
126}
