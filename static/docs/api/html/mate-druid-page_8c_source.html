132static void
133mate_druid_page_size_request (GtkWidget *widget,
134 GtkRequisition *requisition)
135{
136 GtkBin *bin;
137
138 g_return_if_fail (widget != NULL);
139 g_return_if_fail (MATE_IS_DRUID_PAGE (widget));
140 g_return_if_fail (requisition != NULL);
141 bin = GTK_BIN (widget);
142 requisition->width = GTK_CONTAINER (widget)->border_width * 2;
143 requisition->height = GTK_CONTAINER (widget)->border_width * 2;
144
145 if (bin->child && GTK_WIDGET_VISIBLE (bin->child)) {
146 GtkRequisition child_requisition;
147
148 gtk_widget_size_request (bin->child, &child_requisition);
149
150 requisition->width += child_requisition.width;
151 requisition->height += child_requisition.height;
152 }
153}
154static void
155mate_druid_page_size_allocate (GtkWidget *widget,
156 GtkAllocation *allocation)
157{
158 GtkBin *bin;
159 GtkAllocation child_allocation;
160
161 g_return_if_fail (widget != NULL);
162 g_return_if_fail (MATE_IS_DRUID_PAGE (widget));
163 g_return_if_fail (allocation != NULL);
164 widget->allocation = *allocation;
165 bin = GTK_BIN (widget);
166
167 child_allocation.x = 0;
168 child_allocation.y = 0;
169 child_allocation.width = MAX (allocation->width - GTK_CONTAINER (widget)->border_width * 2, 0);
170 child_allocation.height = MAX (allocation->height - GTK_CONTAINER (widget)->border_width * 2, 0);
171
172 if (GTK_WIDGET_REALIZED (widget)) {
173 gdk_window_move_resize (widget->window,
174 allocation->x + GTK_CONTAINER (widget)->border_width,
175 allocation->y + GTK_CONTAINER (widget)->border_width,
176 child_allocation.width,
177 child_allocation.height);
178 }
179 if (bin->child) {
180 gtk_widget_size_allocate (bin->child, &child_allocation);
181 }
182}

191static gint
192mate_druid_page_expose (GtkWidget *widget,
193 GdkEventExpose *event)
194{
195 g_return_val_if_fail (widget != NULL, FALSE);
196 g_return_val_if_fail (MATE_IS_DRUID_PAGE (widget), FALSE);
197 g_return_val_if_fail (event != NULL, FALSE);
198
199 if (!GTK_WIDGET_APP_PAINTABLE (widget))
200 mate_druid_page_paint (widget, &event->area);
201
202 if (GTK_WIDGET_DRAWABLE (widget)) {
203 return MATE_CALL_PARENT_WITH_DEFAULT
204 (GTK_WIDGET_CLASS, expose_event, (widget, event), FALSE);
205 }
206
207 return FALSE;

210static void
211mate_druid_page_realize (GtkWidget *widget)
212{
213 GdkWindowAttr attributes;
214 gint attributes_mask;
215 gint border_width;
216 g_return_if_fail (widget != NULL);
217 g_return_if_fail (MATE_IS_DRUID_PAGE (widget));
218
219 GTK_WIDGET_SET_FLAGS (widget, GTK_REALIZED);
220
221 border_width = GTK_CONTAINER (widget)->border_width;
222
223 attributes.x = widget->allocation.x + border_width;
224 attributes.y = widget->allocation.y + border_width;
225 attributes.width = widget->allocation.width - 2*border_width;
226 attributes.height = widget->allocation.height - 2*border_width;
227 attributes.window_type = GDK_WINDOW_CHILD;
228 attributes.wclass = GDK_INPUT_OUTPUT;
229 attributes.visual = gtk_widget_get_visual (widget);
230 attributes.colormap = gtk_widget_get_colormap (widget);
231 attributes.event_mask = gtk_widget_get_events (widget)
232 | GDK_BUTTON_MOTION_MASK
233 | GDK_BUTTON_PRESS_MASK
234 | GDK_BUTTON_RELEASE_MASK
235 | GDK_EXPOSURE_MASK
236 | GDK_ENTER_NOTIFY_MASK
237 | GDK_LEAVE_NOTIFY_MASK;
238
239 attributes_mask = GDK_WA_X | GDK_WA_Y | GDK_WA_VISUAL | GDK_WA_COLORMAP;
240
241 widget->window = gdk_window_new (gtk_widget_get_parent_window (widget), &attributes, attributes_mask);
242 gdk_window_set_user_data (widget->window, widget);
243 gdk_window_set_back_pixmap (widget->window, NULL, FALSE);
244
245 widget->style = gtk_style_attach (widget->style, widget->window);
246 gtk_style_set_background (widget->style, widget->window, GTK_STATE_NORMAL);
247}
