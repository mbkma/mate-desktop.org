214/* Renders the pixmap for the case of dithered or use_alpha */
215static void
216render_dither (MateColorPicker *cp)
217{
218 gint dark_r, dark_g, dark_b;
219 gint light_r, light_g, light_b;
220 gint i, j, rowstride;
221 gint c1[3], c2[3];
222 guchar *pixels;
223 guint8 insensitive_r = 0;
224 guint8 insensitive_g = 0;
225 guint8 insensitive_b = 0;
226
227 /* Compute dark and light check colors */
228
229 insensitive_r = GTK_WIDGET(cp)->style->bg[GTK_STATE_INSENSITIVE].red >> 8;
230 insensitive_g = GTK_WIDGET(cp)->style->bg[GTK_STATE_INSENSITIVE].green >> 8;
231 insensitive_b = GTK_WIDGET(cp)->style->bg[GTK_STATE_INSENSITIVE].blue >> 8;
232
233 if (cp->_priv->use_alpha) {
234 dark_r = (int) ((CHECK_DARK + (cp->_priv->r - CHECK_DARK) * cp->_priv->a) * 255.0 + 0.5);
235 dark_g = (int) ((CHECK_DARK + (cp->_priv->g - CHECK_DARK) * cp->_priv->a) * 255.0 + 0.5);
236 dark_b = (int) ((CHECK_DARK + (cp->_priv->b - CHECK_DARK) * cp->_priv->a) * 255.0 + 0.5);
237
238 light_r = (int) ((CHECK_LIGHT + (cp->_priv->r - CHECK_LIGHT) * cp->_priv->a) * 255.0 + 0.5);
239 light_g = (int) ((CHECK_LIGHT + (cp->_priv->g - CHECK_LIGHT) * cp->_priv->a) * 255.0 + 0.5);
240 light_b = (int) ((CHECK_LIGHT + (cp->_priv->b - CHECK_LIGHT) * cp->_priv->a) * 255.0 + 0.5);
241 } else {
242 dark_r = light_r = (int) (cp->_priv->r * 255.0 + 0.5);
243 dark_g = light_g = (int) (cp->_priv->g * 255.0 + 0.5);
244 dark_b = light_b = (int) (cp->_priv->b * 255.0 + 0.5);
245 }
246
247 /* Fill image buffer */
248
249 pixels = gdk_pixbuf_get_pixels (cp->_priv->pixbuf);
250 rowstride = gdk_pixbuf_get_rowstride (cp->_priv->pixbuf);
251 for (j = 0; j < COLOR_PICKER_HEIGHT; j++) {
252 if ((j / CHECK_SIZE) & 1) {
253 c1[0] = dark_r;
254 c1[1] = dark_g;
255 c1[2] = dark_b;
256
257 c2[0] = light_r;
258 c2[1] = light_g;
259 c2[2] = light_b;
260 } else {
261 c1[0] = light_r;
262 c1[1] = light_g;
263 c1[2] = light_b;
264
265 c2[0] = dark_r;
266 c2[1] = dark_g;
267 c2[2] = dark_b;
268 }
269
270 for (i = 0; i < COLOR_PICKER_WIDTH; i++) {
271 if (!GTK_WIDGET_SENSITIVE (GTK_WIDGET (cp)) && (i+j)%2) {
272 *(pixels + j * rowstride + i * 3) = insensitive_r;
273 *(pixels + j * rowstride + i * 3 + 1) = insensitive_g;
274 *(pixels + j * rowstride + i * 3 + 2) = insensitive_b;
275 } else if ((i / CHECK_SIZE) & 1) {
276 *(pixels + j * rowstride + i * 3) = c1[0];
277 *(pixels + j * rowstride + i * 3 + 1) = c1[1];
278 *(pixels + j * rowstride + i * 3 + 2) = c1[2];
279 } else {
280 *(pixels + j * rowstride + i * 3) = c2[0];
281 *(pixels + j * rowstride + i * 3 + 1) = c2[1];
282 *(pixels + j * rowstride + i * 3 + 2) = c2[2];
283 }
284 }
285 }
286 if (cp->_priv->drawing_area->window)
287 gdk_draw_pixbuf (cp->_priv->drawing_area->window,
288 cp->_priv->gc,
289 cp->_priv->pixbuf,
290 0, 0, 0, 0,
291 COLOR_PICKER_WIDTH,
292 COLOR_PICKER_HEIGHT,
293 GDK_RGB_DITHER_MAX,
294 0, 0);
295}

297/* Renders the pixmap with the contents of the color sample */
298static void
299render (MateColorPicker *cp)
300{
301 if (cp->_priv->dither || cp->_priv->use_alpha)
302 render_dither (cp);
303 else {
304 gint i, j, rowstride;
305 guint8 insensitive_r = 0;
306 guint8 insensitive_g = 0;
307 guint8 insensitive_b = 0;
308 guchar *pixels;
309
310 pixels = gdk_pixbuf_get_pixels (cp->_priv->pixbuf);
311 rowstride = gdk_pixbuf_get_rowstride (cp->_priv->pixbuf);
312 insensitive_r = GTK_WIDGET (cp)->style->bg[GTK_STATE_INSENSITIVE].red >> 8;
313 insensitive_g = GTK_WIDGET (cp)->style->bg[GTK_STATE_INSENSITIVE].green >> 8;
314 insensitive_b = GTK_WIDGET (cp)->style->bg[GTK_STATE_INSENSITIVE].blue >> 8;
315
316 for (i = 0; i < COLOR_PICKER_WIDTH; i++) {
317 for (j = 0; j < COLOR_PICKER_HEIGHT; j++) {
318 if (!GTK_WIDGET_SENSITIVE (GTK_WIDGET (cp)) && (i+j)%2) {
319 *(pixels + j * rowstride + i * 3) = insensitive_r;
320 *(pixels + j * rowstride + i * 3 + 1) = insensitive_g;
321 *(pixels + j * rowstride + i * 3 + 2) = insensitive_b;
322 } else {
323 *(pixels + j * rowstride + i * 3) = cp->_priv->r * 255.0 + 0.5;
324 *(pixels + j * rowstride + i * 3 + 1) = cp->_priv->g * 255.0 + 0.5;
325 *(pixels + j * rowstride + i * 3 + 2) = cp->_priv->b * 255.0 + 0.5;
326 }
327 }
328 }
329 }

333static gint
334expose_event (GtkWidget *widget, GdkEventExpose *event, gpointer data)
335{
336 MateColorPicker *cp;
337
338 cp = MATE_COLOR_PICKER (data);
339
340 gdk_draw_pixbuf (widget->window,
341 cp->_priv->gc,
342 cp->_priv->pixbuf,
343 event->area.x,
344 event->area.y,
345 event->area.x,
346 event->area.y,
347 event->area.width,
348 event->area.height,
349 GDK_RGB_DITHER_MAX,
350 event->area.x,
351 event->area.y);
352 return FALSE;

384static void
385drag_data_received (GtkWidget *widget,
386 GdkDragContext *context,
387 gint x,
388 gint y,
389 GtkSelectionData *selection_data,
390 guint info,
391 guint32 time,
392 MateColorPicker *cpicker)
393{
394 guint16 *dropped;
395
396 g_return_if_fail (cpicker != NULL);
397 g_return_if_fail (MATE_IS_COLOR_PICKER (cpicker));
398
399 if (selection_data->length < 0)
400 return;
401
402 if ((selection_data->format != 16) ||
403 (selection_data->length != 8)) {
404 g_warning (_("Received invalid color data\n"));
405 return;
406 }
407
408
409 dropped = (guint16 *)selection_data->data;
410
411 mate_color_picker_set_i16(cpicker, dropped[0], dropped[1],
412 dropped[2], dropped[3]);
413 g_signal_emit (cpicker, color_picker_signals[COLOR_SET], 0,
414 dropped[0], dropped[1], dropped[2], dropped[3]);
415
416}

418static void
419drag_data_get (GtkWidget *widget,
420 GdkDragContext *context,
421 GtkSelectionData *selection_data,
422 guint info,
423 guint time,
424 MateColorPicker *cpicker)
425{
426 gushort r, g, b, a;
427 guint16 dropped[4];
428
429 mate_color_picker_get_i16(cpicker, &r, &g, &b, &a);
430
431 dropped[0] = r;
432 dropped[1] = g;
433 dropped[2] = b;
434 dropped[3] = a;
435
436 gtk_selection_data_set (selection_data,
437 selection_data->target,
438 16/*fromat*/, (guchar *)dropped, 8/*length*/);
439}

441static void
442mate_color_picker_instance_init (MateColorPicker *cp)
443{
444 GtkWidget *alignment;
445 GtkWidget *frame;
446 const GtkTargetEntry drop_types[] = {
447 { "application/x-color", 0, 0 }
448 };
449
450 /* Create the widgets */
451 cp->_priv = g_new0(MateColorPickerPrivate, 1);
452
453 /*
454 * The application may very well override these.
455 */
456 _add_atk_name_desc (GTK_WIDGET (cp),
457 NULL,
458 _("Open a dialog to specify the color"));
459
460 alignment = gtk_alignment_new (0.5, 0.5, 0.0, 0.0);
461 gtk_container_set_border_width (GTK_CONTAINER (alignment), COLOR_PICKER_PAD);
462 gtk_container_add (GTK_CONTAINER (cp), alignment);
463 gtk_widget_show (alignment);
464
465 frame = gtk_frame_new (NULL);
466 gtk_frame_set_shadow_type (GTK_FRAME (frame), GTK_SHADOW_ETCHED_OUT);
467 gtk_container_add (GTK_CONTAINER (alignment), frame);
468 gtk_widget_show (frame);
469
470 gtk_widget_push_colormap (gdk_rgb_get_colormap ());
471
472 cp->_priv->drawing_area = gtk_drawing_area_new ();
473
474 gtk_widget_set_size_request (cp->_priv->drawing_area, COLOR_PICKER_WIDTH, COLOR_PICKER_HEIGHT);
475 g_signal_connect (cp->_priv->drawing_area, "expose_event",
476 G_CALLBACK (expose_event), cp);
477 gtk_container_add (GTK_CONTAINER (frame), cp->_priv->drawing_area);
478 gtk_widget_show (cp->_priv->drawing_area);
479
480 cp->_priv->title = g_strdup (_("Pick a color")); /* default title */
481
482 /* Create the buffer for the image so that we can create an image. Also create the
483 * picker's pixmap.
484 */
485
486 cp->_priv->pixbuf = gdk_pixbuf_new (GDK_COLORSPACE_RGB, FALSE, 8, COLOR_PICKER_WIDTH, COLOR_PICKER_HEIGHT);
487
488 cp->_priv->gc = NULL;
489 gtk_widget_pop_colormap ();
490
491 /* Start with opaque black, dither on, alpha disabled */
492
493 cp->_priv->r = 0.0;
494 cp->_priv->g = 0.0;
495 cp->_priv->b = 0.0;
496 cp->_priv->a = 1.0;
497 cp->_priv->dither = TRUE;
498 cp->_priv->use_alpha = FALSE;
499
500 gtk_drag_dest_set (GTK_WIDGET (cp),
501 GTK_DEST_DEFAULT_MOTION |
502 GTK_DEST_DEFAULT_HIGHLIGHT |
503 GTK_DEST_DEFAULT_DROP,
504 drop_types, G_N_ELEMENTS (drop_types), GDK_ACTION_COPY);
505 gtk_drag_source_set (GTK_WIDGET(cp),
506 GDK_BUTTON1_MASK|GDK_BUTTON3_MASK,
507 drop_types, G_N_ELEMENTS (drop_types),
508 GDK_ACTION_COPY);
509 g_signal_connect (cp, "drag_data_received",
510 G_CALLBACK (drag_data_received), cp);
511 g_signal_connect (cp, "drag_data_get",
512 G_CALLBACK (drag_data_get), cp);
513}

593/* Callback for when the OK button in the color selection dialog is clicked */
594static void
595cs_ok_clicked (GtkWidget *widget, gpointer data)
596{
597 MateColorPicker *cp;
598 GtkColorSelection *cs;
599 GdkColor color;
600 guint16 alpha;
601 gushort r, g, b, a;
602
603 cp = MATE_COLOR_PICKER (data);
604
605 cs = GTK_COLOR_SELECTION (GTK_COLOR_SELECTION_DIALOG (cp->_priv->cs_dialog)->colorsel);
606
607 gtk_color_selection_get_current_color (cs, &color);
608 alpha = gtk_color_selection_get_current_alpha (cs);
609
610 gtk_widget_destroy (cp->_priv->cs_dialog);
611
612 cp->_priv->r = color.red / 65535.0;
613 cp->_priv->g = color.green / 65535.0;
614 cp->_priv->b = color.blue / 65535.0;
615 cp->_priv->a = cp->_priv->use_alpha ? (alpha / 65535.0) : 1.0;
616
617 render (cp);
618 gtk_widget_queue_draw (cp->_priv->drawing_area);
619
620 /* Notify the world that the color was set */
621
622 mate_color_picker_get_i16 (cp, &r, &g, &b, &a);
623 g_signal_emit (cp, color_picker_signals[COLOR_SET], 0,
624 r, g, b, a);
625}

637static void
638mate_color_picker_clicked (GtkButton *button)
639{
640 MateColorPicker *cp;
641 GtkColorSelectionDialog *csd;
642 GdkColor color;
643 guint16 alpha;
644
645 g_return_if_fail (button != NULL);
646 g_return_if_fail (MATE_IS_COLOR_PICKER (button));
647
648 cp = MATE_COLOR_PICKER (button);
649
650 /*if dialog already exists, make sure it's shown and raised*/
651 if(cp->_priv->cs_dialog) {
652 csd = GTK_COLOR_SELECTION_DIALOG (cp->_priv->cs_dialog);
653 gtk_widget_show (cp->_priv->cs_dialog);
654 if (cp->_priv->cs_dialog->window)
655 gdk_window_raise(cp->_priv->cs_dialog->window);
656 } else {
657 /* Create the dialog and connects its buttons */
658 GtkWidget *parent;
659
660 parent = gtk_widget_get_toplevel(GTK_WIDGET(cp));
661
662 cp->_priv->cs_dialog = gtk_color_selection_dialog_new (cp->_priv->title);
663
664 if (parent)
665 gtk_window_set_transient_for(GTK_WINDOW(cp->_priv->cs_dialog),
666 GTK_WINDOW(parent));
667
668 csd = GTK_COLOR_SELECTION_DIALOG (cp->_priv->cs_dialog);
669 gtk_color_selection_set_has_palette
670 (GTK_COLOR_SELECTION (csd->colorsel), TRUE);
671 g_signal_connect (cp->_priv->cs_dialog, "destroy",
