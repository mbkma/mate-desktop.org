81void
82mate_message_box_construct (MateMessageBox *messagebox,
83 const gchar *message,
84 const gchar *message_box_type,
85 const gchar **buttons)
86{
87 GtkWidget *hbox;
88 GtkWidget *pixmap = NULL;
89 GtkWidget *alignment;
90 GtkWidget *label;
91 char *s;
92 const gchar* title_prefix = NULL;
93 const gchar* appname;
94
95 g_return_if_fail (messagebox != NULL);
96 g_return_if_fail (MATE_IS_MESSAGE_BOX (messagebox));
97 g_return_if_fail (message != NULL);
98 g_return_if_fail (message_box_type != NULL);
99
100 atk_object_set_role (gtk_widget_get_accessible (GTK_WIDGET (messagebox)), ATK_ROLE_ALERT);
101
102 /* Make noises, basically */
103 mate_triggers_vdo(message, message_box_type, NULL);
104
105 if (strcmp(MATE_MESSAGE_BOX_INFO, message_box_type) == 0)
106 {
107 title_prefix = _("Information");
108 pixmap = gtk_image_new_from_stock (GTK_STOCK_DIALOG_INFO, GTK_ICON_SIZE_DIALOG);
109 }
110 else if (strcmp(MATE_MESSAGE_BOX_WARNING, message_box_type) == 0)
111 {
112 title_prefix = _("Warning");
113 pixmap = gtk_image_new_from_stock (GTK_STOCK_DIALOG_WARNING, GTK_ICON_SIZE_DIALOG);
114
115 }
116 else if (strcmp(MATE_MESSAGE_BOX_ERROR, message_box_type) == 0)
117 {
118 title_prefix = _("Error");
119 pixmap = gtk_image_new_from_stock (GTK_STOCK_DIALOG_ERROR, GTK_ICON_SIZE_DIALOG);
120
121 }
122 else if (strcmp(MATE_MESSAGE_BOX_QUESTION, message_box_type) == 0)
123 {
124 title_prefix = _("Question");
125 pixmap = gtk_image_new_from_stock (GTK_STOCK_DIALOG_QUESTION, GTK_ICON_SIZE_DIALOG);
126 }
127 else
128 {
129 title_prefix = _("Message");
130 }
131
132 g_assert(title_prefix != NULL);
133 s = NULL;
134 appname = mate_program_get_human_readable_name(mate_program_get());
135 if (appname) {
136 s = g_strdup_printf("%s (%s)", title_prefix, appname);
137 }
138
139 mate_dialog_constructv (MATE_DIALOG (messagebox), s ? s : title_prefix, buttons);
140 g_free (s);
141
142 hbox = gtk_hbox_new (FALSE, 0);
143 gtk_box_pack_start (GTK_BOX(MATE_DIALOG(messagebox)->vbox),
144 hbox, TRUE, TRUE, 10);
145 gtk_widget_show (hbox);
146
147 if (pixmap) {
148 gtk_box_pack_start (GTK_BOX(hbox),
149 pixmap, FALSE, TRUE, 0);
150 gtk_widget_show (pixmap);
151 }
152
153 label = gtk_label_new (message);
154 gtk_label_set_justify (GTK_LABEL (label), GTK_JUSTIFY_LEFT);
155 gtk_label_set_line_wrap (GTK_LABEL (label), TRUE);
156 gtk_misc_set_padding (GTK_MISC (label), MATE_PAD, 0);
157 gtk_box_pack_start (GTK_BOX (hbox), label, TRUE, TRUE, 0);
158 gtk_widget_show (label);
159
160 /* Add some extra space on the right to balance the pixmap */
161 if (pixmap) {
162 alignment = gtk_alignment_new (0., 0., 0., 0.);
163 gtk_widget_set_size_request (alignment, MATE_PAD, -1);
164 gtk_widget_show (alignment);
165
166 gtk_box_pack_start (GTK_BOX (hbox), alignment, FALSE, FALSE, 0);
167 }
168
169 mate_dialog_set_close (MATE_DIALOG (messagebox),
170 TRUE );
171}

185GtkWidget*
186mate_message_box_new (const gchar *message,
187 const gchar *message_box_type, ...)
188{
189 va_list ap;
190 MateMessageBox *message_box;
191
192 g_return_val_if_fail (message != NULL, NULL);
193 g_return_val_if_fail (message_box_type != NULL, NULL);
194
195 va_start (ap, message_box_type);
196
197 message_box = g_object_new (MATE_TYPE_MESSAGE_BOX, NULL);
198
199 mate_message_box_construct (message_box, message,
200 message_box_type, NULL);
201
202 /* we need to add buttons by hand here */
203 while (TRUE) {
204 gchar * button_name;
205
206 button_name = va_arg (ap, gchar *);
207
208 if (button_name == NULL) {
209 break;
210 }
211
212 mate_dialog_append_button ( MATE_DIALOG(message_box),
213 button_name);
214 }
215
216 va_end (ap);
217
218 if (MATE_DIALOG (message_box)->buttons) {
219 gtk_widget_grab_focus(
220 g_list_last (MATE_DIALOG (message_box)->buttons)->data);
221 }
222
223 return GTK_WIDGET (message_box);
224}
