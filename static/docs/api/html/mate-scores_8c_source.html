156{
157 gchar tmp[10];
158 gchar tmp2[256];
159 guint i;
160
161 gs->_priv->n_scores = n_scores;
162
163 gs->_priv->label_names = g_malloc(sizeof(GtkWidget*) * n_scores);
164 gs->_priv->label_scores = g_malloc(sizeof(GtkWidget*) * n_scores);
165 gs->_priv->label_times = g_malloc(sizeof(GtkWidget*) * n_scores);
166
167 for(i=0; i < n_scores; i++) {
168 char *str_utf8;
169
170 gs->_priv->label_names[i] = gtk_label_new ( names[i] );
171 gtk_widget_show ( gs->_priv->label_names[i] );
172 gtk_table_attach_defaults ( gs->_priv->table, gs->_priv->label_names[i], 0, 1, i+1, i+2);
173
174 g_snprintf(tmp,sizeof(tmp),"%5.2f", scores[i]);
175 gs->_priv->label_scores[i] = gtk_label_new ( tmp );
176 gtk_widget_show ( gs->_priv->label_scores[i] );
177 gtk_table_attach_defaults ( gs->_priv->table, gs->_priv->label_scores[i], 1, 2, i+1, i+2);
178
179 /* TRANSLATORS NOTE: Date format used when showing scores in games.
180 * Please refer to manpage of strftime(3) for complete reference.
181 */
182 if(strftime(tmp2, sizeof(tmp2), _("%a %b %d %T %Y"),

184 /* according to docs, if the string does not fit, the

187 strcpy(tmp2, "???");

190
191 str_utf8 = g_locale_to_utf8 (tmp2, -1, NULL, NULL, NULL);
192 gs->_priv->label_times[i] = gtk_label_new (str_utf8 ? str_utf8 : NULL);
193 g_free (str_utf8);
194 gtk_widget_show ( gs->_priv->label_times[i] );

451GtkWidget* mate_scores_display(const gchar* title, const gchar* app_name, const gchar* level, int pos)
452{
453 GtkWidget *hs = NULL;
454 gchar **names = NULL;
455 gfloat *scores = NULL;
456 time_t *scoretimes = NULL;
457 gint top;
458
459 top = mate_score_get_notable(app_name, level, &names, &scores, &scoretimes);
460 if (top > 0) {
461 hs = mate_scores_new(top, names, scores, scoretimes, 0);
462 mate_scores_set_logo_label_title (MATE_SCORES(hs), title);
463 if(pos)
464 mate_scores_set_current_player(MATE_SCORES(hs), pos-1);
465
466 gtk_widget_show (hs);
467 }
468
469 g_strfreev(names);
470 g_free(scores);
471 g_free(scoretimes);
472
473 return hs;
474}
