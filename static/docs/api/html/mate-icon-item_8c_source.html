99static void
100send_focus_event (MateIconTextItem *iti, gboolean in)
101{
102 MateIconTextItemPrivate *priv;
103 GtkWidget *widget;
104 gboolean has_focus;
105 GdkEvent fake_event;
106
107 g_return_if_fail (in == FALSE || in == TRUE);
108
109 priv = iti->_priv;
110 if (priv->entry == NULL) {
111 g_assert (!in);
112 return;
113 }
114
115 widget = GTK_WIDGET (priv->entry);
116 has_focus = GTK_WIDGET_HAS_FOCUS (widget);
117 if (has_focus == in) {
118 return;
119 }
120
121 memset (&fake_event, 0, sizeof (fake_event));
122 fake_event.focus_change.type = GDK_FOCUS_CHANGE;
123 fake_event.focus_change.window = widget->window;
124 fake_event.focus_change.in = in;
125 gtk_widget_event (widget, &fake_event);
126
127 /* FIXME: this is failing */
128#if 0
129 g_return_if_fail (GTK_WIDGET_HAS_FOCUS (widget) == in);
130#endif
131}

133/* Updates the pango layout width */
134static void
135update_pango_layout (MateIconTextItem *iti)
136{
137 MateIconTextItemPrivate *priv;
138 PangoRectangle bounds;
139 const char *text;
140
141 priv = iti->_priv;
142
143 if (iti->editing) {
144 text = gtk_entry_get_text (GTK_ENTRY (priv->entry));
145 } else {
146 text = iti->text;
147 }
148
149 pango_layout_set_wrap (priv->layout, PANGO_WRAP_WORD_CHAR);
150 pango_layout_set_text (priv->layout, text, strlen (text));
151 pango_layout_set_width (priv->layout, iti->width * PANGO_SCALE);
152
153 /* In PANGO_WRAP_WORD mode, words wider than a line of text make
154 * PangoLayout overflow the layout width. If this happens, switch to
155 * character-based wrapping.
156 */
157
158 pango_layout_get_pixel_extents (iti->_priv->layout, NULL, &bounds);
159
160 priv->layout_width = bounds.width;
161 priv->layout_height = bounds.height;
162}

176/* Accepts the text in the off-screen entry of an icon text item */
177static void
178iti_edition_accept (MateIconTextItem *iti)
179{
180 MateIconTextItemPrivate *priv;
181 gboolean accept;
182
183 priv = iti->_priv;
184 accept = TRUE;
185
186 g_signal_emit (iti, iti_signals [TEXT_CHANGED], 0, &accept);
187
188 if (iti->editing){
189 if (accept) {
190 if (iti->is_text_allocated)
191 g_free (iti->text);
192
193 iti->text = g_strdup (gtk_entry_get_text (GTK_ENTRY(priv->entry)));
194 iti->is_text_allocated = 1;
195 }
196
197 iti_stop_editing (iti);
198 }
199 update_pango_layout (iti);
200 priv->need_text_update = TRUE;
201
202 mate_canvas_item_request_update (MATE_CANVAS_ITEM (iti));
203}

314static int
315get_layout_index (MateIconTextItem *iti, int x, int y)
316{
317 int index;
318 int trailing;
319 const char *cluster;
320 const char *cluster_end;
321
322 MateIconTextItemPrivate *priv;
323 PangoRectangle extents;
324
325 trailing = 0;
326 index = 0;
327 priv = iti->_priv;
328
329 pango_layout_get_extents (priv->layout, NULL, &extents);
330
331 x = (x * PANGO_SCALE) + extents.x;
332 y = (y * PANGO_SCALE) + extents.y;
333
334 pango_layout_xy_to_index (priv->layout, x, y, &index, &trailing);
335
336 cluster = gtk_entry_get_text (GTK_ENTRY (priv->entry)) + index;
337 cluster_end = cluster;
338 while (trailing) {
339 cluster_end = g_utf8_next_char (cluster_end);
340 --trailing;
341 }
342 index += (cluster_end - cluster);
343
344 return index;
345}

403/* Starts the editing state of an icon text item */
404static void
405iti_start_editing (MateIconTextItem *iti)
406{
407 MateIconTextItemPrivate *priv;
408
409 priv = iti->_priv;
410
411 if (iti->editing)
412 return;
413
414 /* We place an entry offscreen to handle events and selections. */
415 if (priv->entry_top == NULL) {
416 priv->entry = gtk_entry_new ();
417 g_signal_connect (priv->entry, "activate",
418 G_CALLBACK (iti_entry_activate), iti);
419
420
421 if (GTK_WIDGET_REALIZED (GTK_WIDGET (MATE_CANVAS_ITEM (iti)->canvas))) {
422 realize_cursor_gc (iti);
423 }
424
425 /* Make clipboard functions cause an update, since clipboard
426 * functions will change the offscreen entry */
427 g_signal_connect_after (priv->entry, "changed",
428 G_CALLBACK (iti_entry_text_changed_by_clipboard),
429 iti);
430 priv->entry_top = gtk_window_new (GTK_WINDOW_POPUP);
431 gtk_container_add (GTK_CONTAINER (priv->entry_top),
432 GTK_WIDGET (priv->entry));
433 gtk_window_move (GTK_WINDOW (priv->entry_top), 20000, 20000);
434 gtk_widget_show_all (priv->entry_top);
435 }
436
437 gtk_entry_set_text (GTK_ENTRY (priv->entry), iti->text);
438 gtk_editable_select_region (GTK_EDITABLE (priv->entry), 0, -1);
439
440 iti->editing = TRUE;
441 priv->need_state_update = TRUE;
442
443 send_focus_event (iti, TRUE);
444 mate_canvas_item_request_update (MATE_CANVAS_ITEM (iti));
445
446 g_signal_emit (iti, iti_signals[EDITING_STARTED], 0);
447}

449/* Recomputes the bounding box of an icon text item */
450static void
451recompute_bounding_box (MateIconTextItem *iti)
452{
453
454 MateCanvasItem *item;
455 double x1, y1, x2, y2;
456 int width_c, height_c;
457 int width_w, height_w;
458 int max_width_w;
459
460 item = MATE_CANVAS_ITEM (iti);
461
462 width_c = iti->_priv->layout_width + 2 * MARGIN_X;
463 height_c = iti->_priv->layout_height + 2 * MARGIN_Y;
464 width_w = ROUND (width_c / item->canvas->pixels_per_unit);
465 height_w = ROUND (height_c / item->canvas->pixels_per_unit);
466 max_width_w = ROUND (iti->width / item->canvas->pixels_per_unit);
467
468 x1 = iti->x;
469 y1 = iti->y;
470
471 mate_canvas_item_i2w (item, &x1, &y1);
472
473 x1 += ROUND ((max_width_w - width_w) / 2);
474 x2 = x1 + width_w;
475 y2 = y1 + height_w;
476
477 mate_canvas_w2c_d (item->canvas, x1, y1, &item->x1, &item->y1);
478 mate_canvas_w2c_d (item->canvas, x2, y2, &item->x2, &item->y2);
479}

481static void
482mate_icon_text_item_update (MateCanvasItem *item, double *affine, ArtSVP *clip_path, int flags)
483{
484 MateIconTextItem *iti;
485 MateIconTextItemPrivate *priv;
486
487 iti = MATE_ICON_TEXT_ITEM (item);
488 priv = iti->_priv;
489
490 MATE_CALL_PARENT (MATE_CANVAS_ITEM_CLASS, update, (item, affine, clip_path, flags));
491
492 /* If necessary, queue a redraw of the old bounding box */
493 if ((flags & MATE_CANVAS_UPDATE_VISIBILITY)
494 || (flags & MATE_CANVAS_UPDATE_AFFINE)
495 || priv->need_pos_update
496 || priv->need_font_update
497 || priv->need_text_update)
498 mate_canvas_request_redraw (item->canvas, item->x1, item->y1, item->x2, item->y2);
499
500 /* Compute new bounds */
501 if (priv->need_pos_update
502 || priv->need_font_update
503 || priv->need_text_update)
504 recompute_bounding_box (iti);
505
506 /* Queue redraw */
507 mate_canvas_request_redraw (item->canvas, item->x1, item->y1, item->x2, item->y2);
508
509 priv->need_pos_update = FALSE;
510 priv->need_font_update = FALSE;
511 priv->need_text_update = FALSE;
512 priv->need_state_update = FALSE;
513}

515static void
516iti_draw_cursor (MateIconTextItem *iti, GdkDrawable *drawable,
517 int x, int y)
518{
519 int stem_width;
520 int i;
521 int cursor_offset;
522 PangoRectangle pos;
523 GtkEntry *entry;
524
525 g_return_if_fail (iti->_priv->cursor_gc != NULL);
526
527 entry = GTK_ENTRY (iti->_priv->entry);
528 cursor_offset = gtk_editable_get_position (GTK_EDITABLE (entry));
529 pango_layout_get_cursor_pos (iti->_priv->layout,
530 g_utf8_offset_to_pointer (entry->text, cursor_offset) - entry->text,
531 &pos, NULL);
532 stem_width = PANGO_PIXELS (pos.height) / 30 + 1;
533 for (i = 0; i < stem_width; i++) {
534 gdk_draw_line (drawable, iti->_priv->cursor_gc,
535 x + PANGO_PIXELS (pos.x) + i - stem_width / 2,
536 y + PANGO_PIXELS (pos.y),
537 x + PANGO_PIXELS (pos.x) + i - stem_width / 2,
538 y + PANGO_PIXELS (pos.y) + PANGO_PIXELS (pos.height));
539 }
540}

542/* Draw method handler for the icon text item */
543static void
544mate_icon_text_item_draw (MateCanvasItem *item, GdkDrawable *drawable,
545 int x, int y, int width, int height)
546{
547 GtkWidget *widget;
548 GtkStyle *style;
549 int xofs, yofs;
550 int text_xofs, text_yofs;
551 int w, h;
552 MateIconTextItem *iti;
553 MateIconTextItemPrivate *priv;
554 GtkStateType state;
555
556 widget = GTK_WIDGET (item->canvas);
557 iti = MATE_ICON_TEXT_ITEM (item);
558 priv = iti->_priv;
559
560 style = GTK_WIDGET (MATE_CANVAS_ITEM (iti)->canvas)->style;
561
562 w = priv->layout_width + 2 * MARGIN_X;
563 h = priv->layout_height + 2 * MARGIN_Y;
564
565 xofs = item->x1 - x;
566 yofs = item->y1 - y;
567
568 text_xofs = xofs - (iti->width - priv->layout_width - 2 * MARGIN_X) / 2;
569 text_yofs = yofs + MARGIN_Y;
570
571 if (iti->selected && GTK_WIDGET_HAS_FOCUS (widget))
572 state = GTK_STATE_SELECTED;
573 else if (iti->selected)
574 state = GTK_STATE_ACTIVE;
575 else
576 state = GTK_STATE_NORMAL;
577
578 if (iti->selected && !iti->editing)
579 gdk_draw_rectangle (drawable,
580 style->base_gc[state],
581 TRUE,
582 xofs + 1, yofs + 1,
583 w - 2, h - 2);
584
585 if (GTK_WIDGET_HAS_FOCUS (widget) && iti->focused && ! iti->editing) {
586 GdkRectangle r;
587
588 r.x = xofs;
589 r.y = yofs;
590 r.width = w - 1;
591 r.height = h - 1;
592 gtk_paint_focus (style, drawable,
593 iti->selected ? GTK_STATE_SELECTED: GTK_STATE_NORMAL,
594 &r, widget, NULL,
595 xofs, yofs, w - 1, h - 1);
596#if 0
597 gtk_draw_focus (style,
598 drawable,
599 xofs, yofs,
600 w - 1, h - 1);
601#endif
602 }
603
604 if (iti->editing) {
605 /* FIXME: are these the right graphics contexts? */
606 gdk_draw_rectangle (drawable,
607 style->base_gc[GTK_STATE_NORMAL],
608 TRUE,
609 xofs, yofs, w - 1, h - 1);
610 gdk_draw_rectangle (drawable,
611 style->fg_gc[GTK_STATE_NORMAL],
612 FALSE,
613 xofs, yofs, w - 1, h - 1);
614 }
615
616 gdk_draw_layout (drawable,
617 style->text_gc[iti->editing ? GTK_STATE_NORMAL : state],
618 text_xofs,
619 text_yofs,
620 priv->layout);
621
622 if (iti->editing) {
623 int range[2];
624 if (gtk_editable_get_selection_bounds (GTK_EDITABLE (priv->entry), &range[0], &range[1])) {
625 GdkRegion *clip_region;
626 GdkColor *selection_color, *text_color;
627 guint8 state;
628
629 range[0] = g_utf8_offset_to_pointer (GTK_ENTRY (priv->entry)->text, range[0]) - GTK_ENTRY (priv->entry)->text;
630 range[1] = g_utf8_offset_to_pointer (GTK_ENTRY (priv->entry)->text, range[1]) - GTK_ENTRY (priv->entry)->text;
631
632 state = GTK_WIDGET_HAS_FOCUS (widget) ? GTK_STATE_SELECTED : GTK_STATE_ACTIVE;
633 selection_color = &widget->style->base[state];
634 text_color = &widget->style->text[state];
635 clip_region = gdk_pango_layout_get_clip_region
636 (priv->layout,
637 text_xofs,
638 text_yofs,
639 range, 1);
640 gdk_gc_set_clip_region (widget->style->black_gc,
641 clip_region);
642 gdk_draw_layout_with_colors (drawable,
643 widget->style->black_gc,
644 text_xofs,
645 text_yofs,
646 priv->layout,
647 text_color,
648 selection_color);
649 gdk_gc_set_clip_region (widget->style->black_gc, NULL);
650 gdk_region_destroy (clip_region);
651 } else {
652 iti_draw_cursor (iti, drawable, text_xofs, text_yofs);
653 }
654 }
655}

657static void
658draw_pixbuf_aa (GdkPixbuf *pixbuf, MateCanvasBuf *buf, double affine[6], int x_offset, int y_offset)
659{
660 void (* affine_function)
661 (art_u8 *dst, int x0, int y0, int x1, int y1, int dst_rowstride,
662 const art_u8 *src, int src_width, int src_height, int src_rowstride,
663 const double affine[6],
664 ArtFilterLevel level,
665 ArtAlphaGamma *alpha_gamma);
666
667 affine[4] += x_offset;
668 affine[5] += y_offset;
669
670 affine_function = gdk_pixbuf_get_has_alpha (pixbuf)
671 ? art_rgb_rgba_affine
672 : art_rgb_affine;
673
674 (* affine_function)
675 (buf->buf,
676 buf->rect.x0, buf->rect.y0,
677 buf->rect.x1, buf->rect.y1,
678 buf->buf_rowstride,
679 gdk_pixbuf_get_pixels (pixbuf),
680 gdk_pixbuf_get_width (pixbuf),
681 gdk_pixbuf_get_height (pixbuf),
682 gdk_pixbuf_get_rowstride (pixbuf),
683 affine,
684 ART_FILTER_NEAREST,
685 NULL);
686
687 affine[4] -= x_offset;
688 affine[5] -= y_offset;
689}

691static void
692mate_icon_text_item_render (MateCanvasItem *item, MateCanvasBuf *buffer)
693{
694 GdkVisual *visual;
695 GdkPixmap *pixmap;
696 GdkPixbuf *text_pixbuf;
697 double affine[6];
698 int width, height;
699
700 visual = gdk_rgb_get_visual ();
701 art_affine_identity(affine);
702 width = ROUND (item->x2 - item->x1);
703 height = ROUND (item->y2 - item->y1);
704
705 pixmap = gdk_pixmap_new (NULL, width, height, visual->depth);
706

754/* Point method handler for the icon text item */
755static double
756mate_icon_text_item_point (MateCanvasItem *item, double x, double y, int cx, int cy, MateCanvasItem **actual_item)
757{
758 double dx, dy;
759
760 *actual_item = item;
761
762 if (cx < item->x1)
763 dx = item->x1 - cx;
764 else if (cx > item->x2)
765 dx = cx - item->x2;
766 else
767 dx = 0.0;
768
769 if (cy < item->y1)
770 dy = item->y1 - cy;
771 else if (cy > item->y2)
772 dy = cy - item->y2;
773 else
774 dy = 0.0;
775
776 return sqrt (dx * dx + dy * dy);
777}

779static gboolean
780mate_icon_text_item_event (MateCanvasItem *item, GdkEvent *event)
781{
782 MateIconTextItem *iti;
783 MateIconTextItemPrivate *priv;
784 int idx;
785 double cx, cy;
786
787 iti = MATE_ICON_TEXT_ITEM (item);
788 priv = iti->_priv;
789
790 switch (event->type) {
791 case GDK_KEY_PRESS:
792 if (!iti->editing) {
793 break;
794 }
795
796 switch(event->key.keyval) {
797
798 /* Pass these events back to parent */
799 case GDK_Escape:
800 case GDK_Return:
801 case GDK_KP_Enter:
802 return FALSE;
803
804 /* Handle up and down arrow keys. GdkEntry does not know
805 * how to handle multi line items */
806 case GDK_Up:
807 case GDK_Down:
808 iti_handle_arrow_key_event(iti, event);
809 break;
810
811 default:
812 /* Check for control key operations */
813 if (event->key.state & GDK_CONTROL_MASK) {
814 return FALSE;
815 }
816
817 /* Handle any events that reach us */
818 gtk_widget_event (GTK_WIDGET (priv->entry), event);
819 break;
820 }
821
822 /* Update text item to reflect changes */
823 update_pango_layout (iti);
824 priv->need_text_update = TRUE;
825 mate_canvas_item_request_update (item);
826 return TRUE;
827
828 case GDK_BUTTON_PRESS:
829 if (!iti->editing) {
830 break;
831 }
832
833 if (event->button.button == 1) {
834 mate_canvas_w2c_d (item->canvas, event->button.x, event->button.y, &cx, &cy);
835 idx = get_layout_index (iti,
836 (cx - item->x1) + MARGIN_X,
837 (cy - item->y1) + MARGIN_Y);
838 iti_start_selecting (iti, idx, event->button.time);
839 }
840 return TRUE;
841 case GDK_MOTION_NOTIFY:
842 if (!priv->selecting)
843 break;
844
845 gtk_widget_event (GTK_WIDGET (priv->entry), event);
846 mate_canvas_w2c_d (item->canvas, event->button.x, event->button.y, &cx, &cy);
847 idx = get_layout_index (iti,
848 floor ((cx - (item->x1 + MARGIN_X)) + .5),
849 floor ((cy - (item->y1 + MARGIN_Y)) + .5));
850
851 iti_selection_motion (iti, idx);
852 return TRUE;
853
854 case GDK_BUTTON_RELEASE:
855 if (priv->selecting && event->button.button == 1)
856 iti_stop_selecting (iti, event->button.time);
857 else
858 break;
859 return TRUE;
860 default:
861 break;
862 }
863
864 return FALSE;
865}
