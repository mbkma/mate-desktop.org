236char *
237mate_icon_theme_lookup_icon (MateIconTheme *icon_theme,
238 const char *icon_name,
239 int size,
240 const MateIconData **icon_data_out,
241 int *base_size)
242{
243 MateIconThemePrivate *priv;
244 int i;
245 char *icon;
246 GtkIconInfo *info;
247 GtkIconLookupFlags flags;
248 GdkRectangle rectangle;
249 GdkPoint *points;
250 int n_points;
251 MateIconData *icon_data;
252 priv = icon_theme->priv;
253
254 if (icon_data_out)
255 *icon_data_out = NULL;
256
257 if (priv->allow_svg)
258 flags = GTK_ICON_LOOKUP_FORCE_SVG;
259 else
260 flags = GTK_ICON_LOOKUP_NO_SVG;
261
262 info = gtk_icon_theme_lookup_icon (get_gtk_theme (icon_theme),
263 icon_name, size, flags);
264
265 if (info == NULL)
266 return NULL;
267
268 icon = g_strdup (gtk_icon_info_get_filename (info));
269
270 if (base_size != NULL)
271 *base_size = gtk_icon_info_get_base_size (info);
272
273 /* Free the old info */
274
275 icon_data = &priv->icon_data;
276 g_free (icon_data->display_name);
277 g_free (icon_data->attach_points);
278 memset (icon_data, 0, sizeof (MateIconData));
279
280 icon_data->display_name = g_strdup (gtk_icon_info_get_display_name (info));
281
282 gtk_icon_info_set_raw_coordinates (info, TRUE);
283 icon_data->has_embedded_rect = gtk_icon_info_get_embedded_rect (info, &rectangle);
284 if (icon_data->has_embedded_rect)
285 {
286 icon_data->x0 = rectangle.x;
287 icon_data->y0 = rectangle.y;
288 icon_data->x1 = rectangle.x + rectangle.width;
289 icon_data->y1 = rectangle.y + rectangle.height;
290 }
291
292 if (gtk_icon_info_get_attach_points (info, &points, &n_points))
293 {
294 icon_data->n_attach_points = n_points;
295 icon_data->attach_points = g_new (MateIconDataPoint, n_points);
296 for (i = 0; i < n_points; i++)
297 {
298 icon_data->attach_points[i].x = points[i].x;
299 icon_data->attach_points[i].y = points[i].y;
300 }
301 g_free (points);
302 }
303
304 if (icon_data_out != NULL &&
305 (icon_data->has_embedded_rect ||
306 icon_data->attach_points != NULL ||
307 icon_data->display_name != NULL))
308 *icon_data_out = icon_data;
309
310
311 gtk_icon_info_free (info);
312
313 return icon;
314}
