31/* Accessibility support routines for libmateui */
32static gint
33is_gail_loaded (GtkWidget *widget)
34{
35 AtkObject *aobj;
36 if (gail_loaded == -1) {
37 aobj = gtk_widget_get_accessible (widget);
38 if (!GTK_IS_ACCESSIBLE (aobj))
39 gail_loaded = 0;
40 else
41 gail_loaded = 1;
42 }
43 return gail_loaded;
44}

74 AtkRelationType w1_to_w2, AtkRelationType w2_to_w1)
75{
76 AtkObject *atk_widget1;
77 AtkObject *atk_widget2;
78 AtkRelationSet *relation_set;
79 AtkRelation *relation;
80 AtkObject *targets[1];
81
82 atk_widget1 = gtk_widget_get_accessible (widget1);
83 atk_widget2 = gtk_widget_get_accessible (widget2);
84
85 /* Create the widget1 -> widget2 relation */
86 relation_set = atk_object_ref_relation_set (atk_widget1);
87 targets[0] = atk_widget2;
88 relation = atk_relation_new (targets, 1, w1_to_w2);
89 atk_relation_set_add (relation_set, relation);
90 g_object_unref (relation);
91 g_object_unref (relation_set);
92
93 /* Create the widget2 -> widget1 relation */
94 relation_set = atk_object_ref_relation_set (atk_widget2);
95 targets[0] = atk_widget1;
96 relation = atk_relation_new (targets, 1, w2_to_w1);
97 atk_relation_set_add (relation_set, relation);
98 g_object_unref (relation);
99 g_object_unref (relation_set);

233GType
234_accessibility_create_derived_type (const char *type_name,
235 GType existing_gobject_with_proxy,
236 _AccessibilityClassInitFn class_init)
237{
238 GType type;
239 GType parent_atk_type;
240 GTypeInfo tinfo = { 0 };
241 GTypeQuery query;
242 AtkObjectFactory *factory;
243
244 if ((type = g_type_from_name (type_name))) {
245 return type;
246 }
247
248 factory = atk_registry_get_factory
249 (atk_get_default_registry (),
250 existing_gobject_with_proxy);
251 if (!factory) {
252 return G_TYPE_INVALID;
253 }
254
255 parent_atk_type = atk_object_factory_get_accessible_type (factory);
256 if (!parent_atk_type) {
257 return G_TYPE_INVALID;
258 }
259
260 /*
261 * Figure out the size of the class and instance
262 * we are deriving from
263 */
264 g_type_query (parent_atk_type, &query);
265
266 if (class_init) {
267 tinfo.class_init = (GClassInitFunc) class_init;
268 }
269
270 tinfo.class_size = query.class_size;
271 tinfo.instance_size = query.instance_size;
272
273 /* Register the type */
274 type = g_type_register_static (
275 parent_atk_type, type_name, &tinfo, 0);
276
277 return type;
278}
