13static gdouble
14quality_func (MatePasswordDialog *dialog, const char *text, gpointer ptr)
15{
16 const char *p;
17 gsize length;
18 int uppercase = 0, symbols = 0, numbers = 0, strength;
19 gunichar uc;
20
21 if (text == NULL) return 0.0;
22
23 /* Get the length */
24 length = g_utf8_strlen (text, -1);
25
26 /* Count the number of number, symbols and uppercase chars */
27 for (p = text; *p; p = g_utf8_find_next_char (p, NULL))
28 {
29 uc = g_utf8_get_char (p);
30
31 if (g_unichar_isdigit (uc))
32 {
33 numbers++;
34 }
35 else if (g_unichar_isupper (uc))
36 {
37 uppercase++;
38 }
39 else if (g_unichar_islower (uc))
40 {
41 /* Not counted */
42 }
43 else if (g_unichar_isgraph (uc))
44 {
45 symbols++;
46 }
47 }
48
49 if (length > 5) length = 5;
50 if (numbers > 3) numbers = 3;
51 if (symbols > 3) symbols = 3;
52 if (uppercase > 3) uppercase = 3;
53
54 strength = (length * 10 - 20) + (numbers * 10) + (symbols * 15) + (uppercase * 10);
55 strength = CLAMP (strength, 0, 100);
56
57 return (double) strength / 100.0;
58}
