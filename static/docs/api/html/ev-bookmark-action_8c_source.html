82GtkAction *
83ev_bookmark_action_new (EvBookmark *bookmark)
84{
85 GtkAction *action;
86 gchar *name;
87
88 g_return_val_if_fail (bookmark->title != NULL, NULL);
89
90 name = g_strdup_printf ("EvBookmark%u", bookmark->page);
91 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
92 action = GTK_ACTION (g_object_new (EV_TYPE_BOOKMARK_ACTION,
93 "name", name,
94 "label", bookmark->title,
95 "page", bookmark->page,
96 NULL));
97 G_GNUC_END_IGNORE_DEPRECATIONS;
98 g_free (name);
99
100 return action;
101}
