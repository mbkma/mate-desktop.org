38static gboolean
39applet_main (MatePanelApplet* applet_widget)
40{
41 GError *error = NULL;
42 GvcApplet *applet;
43 GApplication *app = NULL;
44
45 app = g_application_new (GVC_APPLET_DBUS_NAME, G_APPLICATION_FLAGS_NONE);
46
47 if (!g_application_register (app, NULL, &error)) {
48 g_warning ("%s", error->message);
49 g_error_free (error);
50 return FALSE;
51 }
52 if (g_application_get_is_remote (app)) {
53 g_warning ("Applet is already running, exiting");
54 return TRUE;
55 }
56 if (mate_mixer_init () == FALSE) {
57 g_warning ("libmatemixer initialization failed, exiting");
58 return FALSE;
59 }
60
61 gtk_icon_theme_append_search_path (gtk_icon_theme_get_default (),
62 ICON_DATA_DIR);
63
64 applet = gvc_applet_new ();
65
66 gvc_applet_fill (applet, applet_widget);
67 gvc_applet_start (applet);
68
69 g_object_unref (app);
70
71 return TRUE;
72}

80 if (!g_strcmp0 (iid, "GvcApplet"))
