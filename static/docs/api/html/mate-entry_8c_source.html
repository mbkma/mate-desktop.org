160mate_entry_class_init (MateEntryClass *class)
161{
162 GtkObjectClass *object_class;
163 GtkWidgetClass *widget_class;
164 GObjectClass *gobject_class;
165
166 parent_class = g_type_class_peek_parent (class);

259static void
260entry_activated (GtkWidget *widget, gpointer data)
261{
262 MateEntry *gentry;
263 const gchar *text;
264
265 gentry = data;
266
267 text = gtk_entry_get_text (GTK_ENTRY (widget));
268
269 if (!gentry->_priv->changed || (strcmp (text, "") == 0)) {
270 gentry->_priv->changed = FALSE;
271 } else {
272 mate_entry_prepend_history (gentry, TRUE, gtk_entry_get_text (GTK_ENTRY (widget)));
273 }
274
275 g_signal_emit (gentry, mate_entry_signals[ACTIVATE_SIGNAL], 0);
276}

338static void
339mate_entry_destroy (GtkObject *object)
340{
341 MateEntry *gentry;
342
343 /* Note: destroy can run multiple times */
344
345 g_return_if_fail (object != NULL);
346 g_return_if_fail (MATE_IS_ENTRY (object));
347
348 gentry = MATE_ENTRY (object);
349
350 if (gentry->_priv->mateconf_client != NULL) {
351 gchar *key;
352
353 if (gentry->_priv->mateconf_notify_id != 0) {
354 mateconf_client_notify_remove
355 (gentry->_priv->mateconf_client,
356 gentry->_priv->mateconf_notify_id);
357 gentry->_priv->mateconf_notify_id = 0;
358 }
359
360 key = build_mateconf_key (gentry);
361 mateconf_client_remove_dir (gentry->_priv->mateconf_client,
362 key, NULL);
363 g_free (key);
364
365 g_object_unref (G_OBJECT (gentry->_priv->mateconf_client));
366
367 gentry->_priv->mateconf_client = NULL;
368 }
369
370 MATE_CALL_PARENT (GTK_OBJECT_CLASS, destroy, (object));
371}

411static void
412mate_entry_history_changed (MateConfClient* client,
413 guint cnxn_id,
414 MateConfEntry *entry,
415 gpointer user_data)
416{
417 MateEntry *gentry;
418
419 GDK_THREADS_ENTER();
420
421 gentry = MATE_ENTRY (user_data);
422
423 /* If we're getting a notification from saving our own
424 * history, don't reload it.
425 */
426 if (gentry->_priv->saving_history) {
427 gentry->_priv->saving_history = FALSE;
428
429 goto end;
430 }
431
432 mate_entry_load_history (gentry);
433
434 end:
435 GDK_THREADS_LEAVE();
436}

448void
449mate_entry_set_history_id (MateEntry *gentry, const gchar *history_id)
450{
451 gchar *key;
452
453 g_return_if_fail (gentry != NULL);
454 g_return_if_fail (MATE_IS_ENTRY (gentry));
455
456 if (gentry->_priv->history_id != NULL) {
457 g_warning ("The program is trying to change an existing "
458 "MateEntry history id. This operation is "
459 "not allowed.");
460
461 return;
462 }
463
464 if (history_id == NULL)
465 return;
466
467 gentry->_priv->history_id = g_strdup (history_id);
468
469 /* Register with mateconf */
470 key = build_mateconf_key (gentry);
471
472 mateui_mateconf_lazy_init ();
473 gentry->_priv->mateconf_client = mateconf_client_get_default ();
474
475 mateconf_client_add_dir (gentry->_priv->mateconf_client,
476 key,
477 MATECONF_CLIENT_PRELOAD_NONE,
478 NULL);
479
480 gentry->_priv->mateconf_notify_id = mateconf_client_notify_add (gentry->_priv->mateconf_client,
481 key,
482 mate_entry_history_changed,
483 gentry,
484 NULL, NULL);
485
486 g_free (key);
487}

564static void
565set_combo_items (MateEntry *gentry)
566{
567 MateEntryPrivate *priv;
568 GList *l;
569 char *text;
570
571 priv = gentry->_priv;
572
573 /* Save the contents of the entry so that setting the new list of
574 * strings will not overwrite it.
575 */
576
577 text = gtk_editable_get_chars (GTK_EDITABLE (GTK_COMBO (gentry)->entry), 0, -1);
578
579 /* Build the list of strings out of our items */
580
581 if (priv->items) {
582 GList *strings = NULL;
583
584 for (l = priv->items; l; l = l->next) {
585 struct item *item = l->data;
586
587 strings = g_list_prepend (strings, item->text);
588 }
589
590 strings = g_list_reverse (strings);
591
592 gtk_combo_set_popdown_strings (GTK_COMBO (gentry), strings);
593
594 g_list_free (strings);
595 }
596 else
597 {
598 gtk_list_clear_items (GTK_LIST (GTK_COMBO (gentry)->list), 0, -1);
599 }
600
601 /* Restore the text in the entry and clear our changed flag. */
602
603 gtk_entry_set_text (GTK_ENTRY (GTK_COMBO (gentry)->entry), text);
604 g_free (text);
605
606 priv->changed = FALSE;
607}

609static void
610mate_entry_add_history (MateEntry *gentry, gboolean save,
611 const gchar *text, gboolean append)
612{
613 MateEntryPrivate *priv;
614 struct item *item;
615 GList *list;
616 gboolean changed;
617
618 g_return_if_fail (gentry != NULL);
619 g_return_if_fail (MATE_IS_ENTRY (gentry));
620 g_return_if_fail (text != NULL); /* FIXME: should we just return without warning? */
621
622 priv = gentry->_priv;
623
624 item = g_new (struct item, 1);
625 item->save = save;
626 item->text = g_strdup (text);
627
628 /* Remove duplicates */
629 changed = FALSE;
630 list = priv->items;
631
632 while (list != NULL) {
633 struct item * data = list->data;
634
635 if (strcmp (data->text, text) == 0) {
636 free_item (data, NULL);
637 priv->items = g_list_delete_link (priv->items, list);
638
639 changed = TRUE;
640 break;
641 }
642
643 list = g_list_next (list);
644 }
645
646 /* Really add the history item */
647 if (append)
648 priv->items = g_list_append (priv->items, item);
649 else
650 priv->items = g_list_prepend (priv->items, item);
651
652 /* Update combo list */
653 set_combo_items (gentry);
654
655 /* Save the history if needed */
656 if (changed || save)
657 mate_entry_save_history (gentry);
658}

712static void
713mate_entry_load_history (MateEntry *gentry)
714{
715 struct item *item;
716 gchar *key;
717 GSList *mateconf_items, *items;
718
719 g_return_if_fail (gentry != NULL);
720 g_return_if_fail (MATE_IS_ENTRY (gentry));
721
722 if (mate_program_get_app_id (mate_program_get()) == NULL ||
723 gentry->_priv->history_id == NULL)
724 return;
725
726 g_return_if_fail (gentry->_priv->mateconf_client != NULL);
727
728 free_items (gentry);
729
730 key = build_mateconf_key (gentry);
731
732 mateconf_items = mateconf_client_get_list (gentry->_priv->mateconf_client,
733 key, MATECONF_VALUE_STRING, NULL);
734 g_free (key);
735
736 for (items = mateconf_items; items; items = items->next) {
737
738 item = g_new (struct item, 1);
739 item->save = TRUE;
740 item->text = items->data;
741
742 gentry->_priv->items = g_list_append (gentry->_priv->items, item);
743 }
744
745 set_combo_items (gentry);
746
747 g_slist_free (mateconf_items);
748}

769static gboolean
770check_for_duplicates (GSList *mateconf_items,
771 const struct item *item)
772{
773
774 for (; mateconf_items; mateconf_items = mateconf_items->next) {
775
776 if (strcmp ((gchar *) mateconf_items->data, item->text) == 0) {
777 return FALSE;
778 }
779 }
780
781 return TRUE;
782}

785static void
786mate_entry_save_history (MateEntry *gentry)
787{
788 GList *items;
789 GSList *mateconf_items;
790 struct item *item;
791 gchar *key;
792 gint n;
793
794 g_return_if_fail (gentry != NULL);
795 g_return_if_fail (MATE_IS_ENTRY (gentry));
796
797 if (mate_program_get_app_id (mate_program_get ()) == NULL ||
798 gentry->_priv->history_id == NULL)
799 return;
800
801 g_return_if_fail (gentry->_priv->mateconf_client != NULL);
802
803 key = build_mateconf_key (gentry);
804 mateconf_items = NULL;
805
806 for (n = 0, items = gentry->_priv->items;
807 items && n < gentry->_priv->max_saved;
808 items = items->next, n++) {
809 item = items->data;
810
811 if (item->save && check_for_duplicates (mateconf_items, item)) {
812 mateconf_items = g_slist_prepend (mateconf_items, item->text);
813 }
814 }
815
816 mateconf_items = g_slist_reverse (mateconf_items);
817
818 /* Save the list */
819 gentry->_priv->saving_history = TRUE;
820 mateconf_client_set_list (gentry->_priv->mateconf_client, key, MATECONF_VALUE_STRING, mateconf_items, NULL);
821
822 g_slist_free (mateconf_items);
823 g_free (key);
824}
