138static void
139day_selected (GtkCalendar *calendar, MateDateEdit *gde)
140{
141 char buffer [256];
142 guint year, month, day;
143 struct tm mtm = {0};
144 char *str_utf8;
145
146 gtk_calendar_get_date (calendar, &year, &month, &day);
147
148 mtm.tm_mday = day;
149 mtm.tm_mon = month;
150 if (year > 1900)
151 mtm.tm_year = year - 1900;
152 else
153 mtm.tm_year = year;
154
155 if (strftime (buffer, sizeof (buffer),
156 strftime_date_format, &mtm) == 0)
157 strcpy (buffer, "???");
158 buffer[sizeof(buffer)-1] = '\0';
159
160 /* FIXME: what about set time */
161
162 str_utf8 = g_locale_to_utf8 (buffer, -1, NULL, NULL, NULL);
163 gtk_entry_set_text (GTK_ENTRY (gde->_priv->date_entry),
164 str_utf8 ? str_utf8 : "");
165 g_free (str_utf8);
166 g_signal_emit (gde, date_edit_signals [DATE_CHANGED], 0);

201/* This function is yanked from gtkcombo.c */
202static gint
203button_press_popup (GtkWidget *widget, GdkEventButton *event, gpointer data)
204{
205 MateDateEdit *gde;
206 GtkWidget *child;
207
208 gde = data;
209
210 child = gtk_get_event_widget ((GdkEvent *) event);
211
212 /* We don't ask for button press events on the grab widget, so
213 * if an event is reported directly to the grab widget, it must
214 * be on a window outside the application (and thus we remove
215 * the popup window). Otherwise, we check if the widget is a child
216 * of the grab widget, and only remove the popup window if it
217 * is not.
218 */
219 if (child != widget) {
220 while (child) {
221 if (child == widget)
222 return FALSE;
223 child = child->parent;
224 }
225 }
226
227 hide_popup (gde);
228
229 return TRUE;
230}

232static void
233position_popup (MateDateEdit *gde)
234{
235 gint x, y;
236 gint bwidth, bheight;
237 GtkRequisition req;
238
239 gtk_widget_size_request (gde->_priv->cal_popup, &req);
240
241 gdk_window_get_origin (gde->_priv->date_button->window, &x, &y);
242
243 x += gde->_priv->date_button->allocation.x;
244 y += gde->_priv->date_button->allocation.y;
245 bwidth = gde->_priv->date_button->allocation.width;
246 bheight = gde->_priv->date_button->allocation.height;
247
248 x += bwidth - req.width;
249 y += bheight;
250
251 if (x < 0)
252 x = 0;
253
254 if (y < 0)
255 y = 0;
256
257 gtk_window_move (GTK_WINDOW (gde->_priv->cal_popup), x, y);
258}

260static gboolean
261popup_grab_on_window (GdkWindow *window,
262 guint32 activate_time)
263{
264 if ((gdk_pointer_grab (window, TRUE,
265 GDK_BUTTON_PRESS_MASK | GDK_BUTTON_RELEASE_MASK |
266 GDK_POINTER_MOTION_MASK,
267 NULL, NULL, activate_time) == 0)) {
268 if (gdk_keyboard_grab (window, TRUE,
269 activate_time) == 0)
270 return TRUE;
271 else {
272 gdk_pointer_ungrab (activate_time);
273 return FALSE;
274 }
275 }
276 return FALSE;
277}

287 + * do this so that the grab (with owner events == TRUE) affects

290 + */
291 if (!popup_grab_on_window (widget->window,
292 gtk_get_current_event_time ()))
293 return;
294
295 str = gtk_entry_get_text (GTK_ENTRY (gde->_priv->date_entry));
296
297 date = g_date_new ();
298 g_date_set_parse (date, str);
299 /* GtkCalendar expects month to be in 0-11 range (inclusive) */
300 month = g_date_get_month (date) - 1;
301
302 gtk_calendar_select_month (GTK_CALENDAR (gde->_priv->calendar),
303 CLAMP (month, 0, 11),
304 g_date_get_year (date));
305 gtk_calendar_select_day (GTK_CALENDAR (gde->_priv->calendar),
306 g_date_get_day (date));
307
308 g_date_free (date);
309
310 /* FIXME: the preceeding needs further checking to see if it's correct,
311 * the following is so utterly wrong that it doesn't even deserve to be
312 * just commented out, but I didn't want to cut it right now */
313#if 0
314 struct tm mtm = {0};
315 /* This code is pretty much just copied from gtk_date_edit_get_date */
316 sscanf (gtk_entry_get_text (GTK_ENTRY (gde->_priv->date_entry)), "%d/%d/%d",
317 &mtm.tm_mon, &mtm.tm_mday, &mtm.tm_year);
318
319 mtm.tm_mon = CLAMP (mtm.tm_mon, 1, 12);
320 mtm.tm_mday = CLAMP (mtm.tm_mday, 1, 31);
321
322 mtm.tm_mon--;
323
324 /* Hope the user does not actually mean years early in the A.D. days...

326 */
327 if (mtm.tm_year >= 1900)
328 mtm.tm_year -= 1900;
329
330 gtk_calendar_select_month (GTK_CALENDAR (gde->_priv->calendar), mtm.tm_mon, 1900 + mtm.tm_year);
331 gtk_calendar_select_day (GTK_CALENDAR (gde->_priv->calendar), mtm.tm_mday);
332#endif
333
334 position_popup (gde);

341 /* Now transfer our grabs to the popup window; this
342 * should always succeed.
343 */

345 gtk_get_current_event_time ());

619 const GValue *value,
620 GParamSpec *pspec)
621{
622 MateDateEdit *self;
623

625

636 break;
637 case PROP_UPPER_HOUR:
638 mate_date_edit_set_popup_range(self, self->_priv->lower_hour,
639 g_value_get_int (value));
640 break;
641
642 default:
643 break;
644 }
645}

688void
689mate_date_edit_set_time (MateDateEdit *gde, time_t the_time)
690{
691 struct tm *mytm;
692 char buffer [256];
693 char *str_utf8;
694
695 g_return_if_fail(gde != NULL);
696
697 if (the_time == 0)
698 the_time = time (NULL);
699 gde->_priv->initial_time = the_time;
700
701 mytm = localtime (&the_time);
702
703 /* Set the date */
704 if (strftime (buffer, sizeof (buffer), strftime_date_format, mytm) == 0)
705 strcpy (buffer, "???");
706 buffer[sizeof(buffer)-1] = '\0';
707
708 str_utf8 = g_locale_to_utf8 (buffer, -1, NULL, NULL, NULL);
709 gtk_entry_set_text (GTK_ENTRY (gde->_priv->date_entry), str_utf8 ? str_utf8 : "");
710 g_free (str_utf8);
711
712 /* Set the time */
713 if (gde->_priv->flags & MATE_DATE_EDIT_24_HR) {
714 if (gde->_priv->flags & MATE_DATE_EDIT_DISPLAY_SECONDS) {
715 if (strftime (buffer, sizeof (buffer), "%H:%M:%S", mytm) == 0)
716 strcpy (buffer, "???");
717 } else {
718 if (strftime (buffer, sizeof (buffer), "%H:%M", mytm) == 0)
719 strcpy (buffer, "???");
720 }
721 } else {
722 if (gde->_priv->flags & MATE_DATE_EDIT_DISPLAY_SECONDS) {
723 if (strftime (buffer, sizeof (buffer), "%I:%M:%S %p", mytm) == 0)
724 strcpy (buffer, "???");
725 } else {
726 if (strftime (buffer, sizeof (buffer), "%I:%M %p", mytm) == 0)
727 strcpy (buffer, "???");
728 }
729 }
730 buffer[sizeof(buffer)-1] = '\0';
731
732 str_utf8 = g_locale_to_utf8 (buffer, -1, NULL, NULL, NULL);
733 gtk_entry_set_text (GTK_ENTRY (gde->_priv->time_entry), str_utf8 ? str_utf8 : "");
734 g_free (str_utf8);
735}
