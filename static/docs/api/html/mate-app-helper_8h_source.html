202typedef struct {
203 MateUIInfoType type; /* Type of item */
204 gchar const *label; /* String to use in item's label */
205 gchar const *hint; /* Tooltip for toolbar items, status
206 bar message for menu items. */
207 gpointer moreinfo; /* Extra information; depends on the
208 type. */
209 gpointer user_data; /* User data sent to the callback. */
210 gpointer unused_data; /* Should be NULL (reserved). */
211 MateUIPixmapType pixmap_type; /* Type of pixmap for this item. */
212 gconstpointer pixmap_info; /* Pointer to pixmap information. */
213 guint accelerator_key; /* Accelerator key, or 0 for none. */
214 GdkModifierType ac_mods; /* Mask of modifier keys for the
215 accelerator. */
216 GtkWidget *widget; /* Filled in by the mate_app_create*
217 functions. */
218} MateUIInfo;
