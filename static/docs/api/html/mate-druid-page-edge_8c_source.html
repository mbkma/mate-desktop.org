100static void
101mate_druid_page_edge_instance_init (MateDruidPageEdge *page)
102{
103 GtkWidget *table;
104 GtkWidget *box;
105 MateDruidPageEdgePrivate *priv;
106
107 priv = g_new0(MateDruidPageEdgePrivate, 1);
108 page->_priv = priv;
109
110 /* FIXME: we should be a windowed widget, so we can set the
111 * background style on ourself rather than doing this eventbox
112 * hack */
113 priv->background = gtk_event_box_new ();
114 gtk_container_add (GTK_CONTAINER (page), priv->background);
115 gtk_widget_show (priv->background);
116
117 table = gtk_table_new (2, 2, FALSE);
118 gtk_container_add (GTK_CONTAINER (priv->background), table);
119 gtk_widget_show (table);
120
121 /* top bar watermark */
122 priv->top_watermark = gtk_image_new ();
123 gtk_misc_set_alignment (GTK_MISC (priv->top_watermark), 0.0, 0.0);
124 gtk_widget_show (priv->top_watermark);
125
126 /* title label */
127 priv->title_label = gtk_label_new (NULL);
128 gtk_misc_set_alignment (GTK_MISC (priv->title_label), 0.0, 0.5);
129 gtk_widget_show (priv->title_label);
130
131 /* top bar logo */
132 priv->logo = gtk_image_new ();
133 gtk_misc_set_alignment (GTK_MISC (priv->logo), 1.0, 0.5);
134 gtk_widget_show (priv->logo);
135
136 box = gtk_hbox_new (FALSE, 0);
137 gtk_widget_show (box);
138
139 /* side watermark */
140 priv->side_watermark = gtk_image_new ();
141 gtk_misc_set_alignment (GTK_MISC (priv->side_watermark), 0.0, 1.0);
142 gtk_widget_set_size_request (priv->side_watermark, DRUID_PAGE_LEFT_WIDTH, -1);
143 gtk_box_pack_start (GTK_BOX (box), priv->side_watermark,
144 FALSE, TRUE, 0);
145 gtk_widget_show (priv->side_watermark);
146
147 /* contents event box (used for styles) */
148 priv->contents = gtk_event_box_new ();
149 gtk_box_pack_start (GTK_BOX (box), priv->contents,
150 TRUE, TRUE, 0);
151 gtk_widget_show (priv->contents);
152
153 /* contents label */
154 priv->text_label = gtk_label_new (NULL);
155 gtk_label_set_line_wrap (GTK_LABEL (priv->text_label), TRUE);
156 gtk_misc_set_alignment (GTK_MISC (priv->text_label), 0.5, 0.5);
157 gtk_container_add (GTK_CONTAINER (priv->contents), priv->text_label);
158 gtk_widget_show (priv->text_label);
159
160 gtk_table_attach (GTK_TABLE (table),
161 priv->logo,
162 1, 2, 0, 1,
163 GTK_FILL, GTK_FILL,
164 5, 5);
165
166 gtk_table_attach (GTK_TABLE (table),
167 priv->title_label,
168 0, 1, 0, 1,
169 GTK_EXPAND | GTK_FILL,
170 GTK_FILL,
171 5, 5);
172
173 gtk_table_attach (GTK_TABLE (table),
174 priv->top_watermark,
175 0, 2, 0, 1,
176 GTK_EXPAND | GTK_FILL,
177 GTK_FILL,
178 0, 0);
179
180 gtk_table_attach (GTK_TABLE (table),
181 box,
182 0, 2, 1, 2,
183 GTK_EXPAND | GTK_FILL,
184 GTK_EXPAND | GTK_FILL,
185 0, 0);
186

203void
204mate_druid_page_edge_construct (MateDruidPageEdge *druid_page_edge,
205 MateEdgePosition position,
206 gboolean antialiased,
207 const gchar *title,
208 const gchar *text,

230
231 if (top_watermark)
232 mate_druid_page_edge_set_top_watermark (druid_page_edge, top_watermark);
233}

269static void
270mate_druid_page_edge_set_color (MateDruidPageEdge *page)
271{
272 MateDruidPageEdgePrivate *priv = page->_priv;
273 GtkWidget *widget = GTK_WIDGET (page);
274
275 if (!priv->background_color_set)
276 page->background_color = widget->style->bg[GTK_STATE_SELECTED];
277
278 if (!priv->textbox_color_set)
279 page->textbox_color = widget->style->bg[GTK_STATE_NORMAL];
280
281 if (!priv->logo_background_color_set)
282 page->logo_background_color = widget->style->bg[GTK_STATE_SELECTED];
283
284 if (!priv->title_color_set)
285 page->title_color = widget->style->fg[GTK_STATE_SELECTED];
286
287 if (!priv->text_color_set)
288 page->text_color = widget->style->fg[GTK_STATE_NORMAL];
289
290 gtk_widget_modify_bg (priv->background, GTK_STATE_NORMAL, &page->background_color);
291 gtk_widget_modify_bg (priv->contents, GTK_STATE_NORMAL, &page->textbox_color);
292 gtk_widget_modify_bg (priv->logo, GTK_STATE_NORMAL, &page->logo_background_color);
293 gtk_widget_modify_fg (priv->title_label, GTK_STATE_NORMAL, &page->title_color);
294 gtk_widget_modify_fg (priv->text_label, GTK_STATE_NORMAL, &page->text_color);
295}

297static void
298mate_druid_page_edge_prepare (MateDruidPage *page,
299 GtkWidget *druid)
300{
301 switch (MATE_DRUID_PAGE_EDGE (page)->position) {
302 case MATE_EDGE_START:
303 mate_druid_set_buttons_sensitive (MATE_DRUID (druid), FALSE, TRUE, TRUE, TRUE);
304 mate_druid_set_show_finish (MATE_DRUID (druid), FALSE);
305 if (GTK_IS_WINDOW (gtk_widget_get_toplevel (druid)))
306 gtk_widget_grab_default (MATE_DRUID (druid)->next);
307 break;
308 case MATE_EDGE_FINISH:
309 mate_druid_set_buttons_sensitive (MATE_DRUID (druid), TRUE, FALSE, TRUE, TRUE);
310 mate_druid_set_show_finish (MATE_DRUID (druid), TRUE);
311 if (GTK_IS_WINDOW (gtk_widget_get_toplevel (druid)))
312 gtk_widget_grab_default (MATE_DRUID (druid)->finish);
313 break;
314 case MATE_EDGE_OTHER:
315 mate_druid_set_buttons_sensitive (MATE_DRUID (druid), TRUE, TRUE, TRUE, TRUE);
316 mate_druid_set_show_finish (MATE_DRUID (druid), FALSE);
317 default:
318 break;
319 }
