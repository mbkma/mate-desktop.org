79/* the default set_label function: returns a GtkLabel with child->name
80 * if you provide your own, it should return a new widget if its old_label
81 * parameter is NULL and modify and return the old widget otherwise. it
82 * should (obviously) NOT call the parent class handler!
83 */
84static GtkWidget *mate_mdi_child_set_label (MateMDIChild *child, GtkWidget *old_label, gpointer data)
85{
86#ifdef MATE_ENABLE_DEBUG
87 g_message("MateMDIChild: default set_label handler called!\n");
88#endif
89
90 if(old_label) {
91 gtk_label_set_text(GTK_LABEL(old_label), child->name);
92 return old_label;
93 }
94 else {
95 GtkWidget *label;
96
97 label = gtk_label_new(child->name);
98 gtk_misc_set_alignment(GTK_MISC(label), 0.0, 0.5);
99
100 return label;
101 }
102}

121static void mate_mdi_child_destroy (GtkObject *obj)
122{
123 MateMDIChild *mdi_child;
124
125#ifdef MATE_ENABLE_DEBUG
126 g_message("MateMDIChild: destroying!\n");
127#endif
128
129 mdi_child = MATE_MDI_CHILD(obj);
130
131 while(mdi_child->views)
132 mate_mdi_child_remove_view(mdi_child, GTK_WIDGET(mdi_child->views->data));
133
134 if(GTK_OBJECT_CLASS(parent_class)->destroy)
135 (* GTK_OBJECT_CLASS(parent_class)->destroy)(obj);
136}

151GtkWidget *mate_mdi_child_add_view (MateMDIChild *mdi_child)
152{
153 GtkWidget *view = NULL;
154
155 view = mate_mdi_child_create_view(mdi_child);
156
157 if(view) {
158 mdi_child->views = g_list_append(mdi_child->views, view);
159
160 g_object_set_data(G_OBJECT(view), "MateMDIChild", mdi_child);
161 }
162
163 return view;

192void mate_mdi_child_set_name(MateMDIChild *mdi_child, const gchar *name)
193{
194 gchar *old_name = mdi_child->name;
195
196 if(mdi_child->parent)
197 _mate_mdi_child_list_menu_remove_item(MATE_MDI(mdi_child->parent), mdi_child);
198
199 mdi_child->name = (gchar *)g_strdup(name);
200 g_free(old_name);
201
202 if(mdi_child->parent) {
203 _mate_mdi_child_list_menu_add_item(MATE_MDI(mdi_child->parent), mdi_child);
204 mate_mdi_update_child(MATE_MDI(mdi_child->parent), mdi_child);
205 }
206}
